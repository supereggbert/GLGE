141 /**
142 * Gets the absolute path given an import path and the path it's relative to
143 * @param {string} path the path to get the absolute path for
144 * @param {string} relativeto the path the supplied path is relativeto
145 * @returns {string} absolute path
146 * @private
147 */
148 GLGE.MD2.prototype.getAbsolutePath=function(path,relativeto){
149 	if(path.substr(0,7)=="http://" || path.substr(0,7)=="file://"  || path.substr(0,7)=="https://"){
150 		return path;
151 	}
152 	else
153 	{
154 		if(!relativeto){
155 			relativeto=window.location.href;
156 		}
157 		if (relativeto.indexOf("://")==-1){
158 			return relativeto.slice(0,relativeto.lastIndexOf("/"))+"/"+path;
159 		}
160 		//find the path compoents
161 		var bits=relativeto.split("/");
162 		var domain=bits[2];
163 		var proto=bits[0];
164 		var initpath=[];
165 		for(var i=3;i<bits.length-1;i++){
166 			initpath.push(bits[i]);
167 		}
168 		//relative to domain
169 		if(path.substr(0,1)=="/"){
170 			initpath=[];
171 		}
172 		var locpath=path.split("/");
173 		for(var i=0;i<locpath.length;i++){
174 			if(locpath[i]=="..") initpath.pop();
175 				else if(locpath[i]!="") initpath.push(locpath[i]);
176 		}
177 		return proto+"//"+domain+"/"+initpath.join("/");
178 	}
179 }

227 /**
228 * Sets the MD2 frame number
229 * @param {string} frame the frame to display
230 */
231 GLGE.MD2.prototype.setMD2Frame=function(frame){
232 	var totalframes=this.MD2EndFrame-this.MD2StartFrame+1;
233 	if(totalframes==1) return;
234 	if(this.MD2Loop){
235 		frame=frame%totalframes;
236 		var frame2=((Math.floor(frame)+1)%totalframes);
237 	}else{
238 		frame=Math.min(totalframes-1,frame);
239 		frame2=Math.min(totalframes-1,Math.floor(frame)+1);
240 		if(frame==(totalframes-1) && !this.MD2AnimFinished){
241 			this.MD2AnimFinished=true;
242 			this.fireEvent("md2AnimFinished",{});
243 		}
244 	}
245 	var framefrac=frame%1;
246 	if(frame<1 && this.MD2LastAnimFrame!=undefined){
247 		frame=this.MD2LastAnimFrame-this.MD2StartFrame;
248 	}else{
249 		this.MD2LastAnimFrame=null;
250 		this.lastMD2Frame=Math.floor(frame)+this.MD2StartFrame;
251 	}
252 	this.setMeshFrame1(Math.floor(frame)+this.MD2StartFrame);
253 	this.setMeshFrame2(frame2+this.MD2StartFrame);
254 	this.setMeshBlendFactor(framefrac);
255 }

258 	if(!now) now=+new Date;

262 	}

266 /**
267 * Sets the url of the MD2 model
268 * @param {string} url the url to the MD2 file
269 */
270 GLGE.MD2.prototype.setSrc=function(url,relativeTo){
271 	if(relativeTo) url=this.getAbsolutePath(url,relativeTo);
272 	this.url=url;
273 	
274 	//prevent the same model parsing multiple times
275 	if(this.loadingCache[this.url] && !this.headersCache[url]){
276 		var that=this;
277 		setTimeout(function(){that.setSrc(url)},15);
278 		return;
279 	}
280 	
281 	this.loadingCache[this.url]=true;
282 	if(this.headersCache[url]){
283 		this.header=this.headersCache[url];
284 		this.setMesh(this.meshCache[url]);
285 		if(this.MD2Anim) this.setMD2Animation(this.MD2Anim);
286 		this.fireEvent("loaded",{url:this.url});
287 		return;
288 	}
289 	
290 	var that=this;
291 	var xhr = new XMLHttpRequest();
292 	xhr.overrideMimeType('text/plain; charset=x-user-defined');
293 	xhr.open("GET", url, true);
294 	xhr.send(null);
295 	this.verts=[];
296 	this.normals=[];
297 	
298 	xhr.onreadystatechange = function (aEvt) {
299 		if (xhr.readyState == 4) {
300 			if(xhr.status == 200 || xhr.status == 0){
301 				response = xhr.responseText;
302 				if (response) {
303 					var buffer = new ArrayBuffer(response.length);
304 					var byteArray = new Uint8Array(buffer);
305 					var byteArray = [];
306 					for(var i=0;i<response.length;i++){
307 						byteArray[i]=response.charCodeAt(i) & 0xff;
308 					}
309 					that.bufferLoaded(byteArray);
310 				}
311 			}else{
312 				alert("Error loading page\n");
313 			}
314 		}
315 	};
316 }

331 /**
332 * Extract header info
333 * @private
334 */
335 GLGE.MD2.prototype.parseHeader=function(){
336 	this.header={};
337 	for (var i=0; i<this.headerNames.length; i++) {
338 		this.header[this.headerNames[i]]=this.getUint16At(i*4);
339 	}
340 	this.headersCache[this.url]=this.header;
341 }
342 /**
343 * get 16 bit int at location
344 * @private
345 */
346 GLGE.MD2.prototype.getUint16At=function(index){
347 	return this.byteArray[index]+this.byteArray[index+1]*256;
348 }
349 /**
350 * get 32 bit float at location
351 * @private
352 */
353 GLGE.MD2.prototype.getFloat32At=function(index){
354 	var b3=this.byteArray[index];
355 	var b2=this.byteArray[index+1];
356 	var b1=this.byteArray[index+2];
357 	var b0=this.byteArray[index+3];
358 	sign = 1 - (2 * (b0 >> 7)),
359 	exponent = (((b0 << 1) & 0xff) | (b1 >> 7)) - 127,
360 	mantissa = ((b1 & 0x7f) << 16) | (b2 << 8) | b3;
361 
362 	if (mantissa == 0 && exponent == -127) {
363 		return 0.0;
364 	}
365 
366 	if (exponent == -127) { // Denormalized
367 		return sign * mantissa * Math.pow(2, -126 - 23);
368 	}
369 
370 	return sign * (1 + mantissa * Math.pow(2, -23)) * Math.pow(2, exponent);
371 }
372 /**
373 * process the frame data
374 * @private
375 */
376 GLGE.MD2.prototype.parseFrames=function(){
377 	
378 	var vertsArray = this.byteArray;
379 	var startFrame=0;
380 	var MD2Animations={};
381 	for(var j=0;j<this.header.num_frames;j++){
382 		var scaleTrans=[
383 			this.getFloat32At(this.header.ofs_frames+j*this.header.framesize),
384 			this.getFloat32At(this.header.ofs_frames+4+j*this.header.framesize),
385 			this.getFloat32At(this.header.ofs_frames+8+j*this.header.framesize),
386 			this.getFloat32At(this.header.ofs_frames+12+j*this.header.framesize),
387 			this.getFloat32At(this.header.ofs_frames+16+j*this.header.framesize),
388 			this.getFloat32At(this.header.ofs_frames+20+j*this.header.framesize)
389 		];	
390 		var verts=[];
391 		var normals=[];
392 		var start=this.header.ofs_frames+24+j*this.header.framesize;
393 		var frameName="";
394 		for(var i=start;i<start+16;i++){
395 			if(vertsArray[i]==0) break;
396 			frameName+=String.fromCharCode(vertsArray[i]);
397 		}
398 		frameName=frameName.replace(/[0-9]/g,'');
399 		if(lastFrameName && frameName!=lastFrameName){
400 			MD2Animations[lastFrameName]=[startFrame,j-1];
401 			startFrame=j;
402 		}
403 		var lastFrameName=frameName;
404 		start=this.header.ofs_frames+40+j*this.header.framesize;
405 		for(var i=start;i<start+this.header.framesize-40;i=i+12){
406 			verts.push(vertsArray[i]*scaleTrans[0]+scaleTrans[3]);
407 			verts.push(vertsArray[i+1]*scaleTrans[1]+scaleTrans[4]);
408 			verts.push(vertsArray[i+2]*scaleTrans[2]+scaleTrans[5]);
409 			verts.push(vertsArray[i+4]*scaleTrans[0]+scaleTrans[3]);
410 			verts.push(vertsArray[i+5]*scaleTrans[1]+scaleTrans[4]);
411 			verts.push(vertsArray[i+6]*scaleTrans[2]+scaleTrans[5]);
412 			verts.push(vertsArray[i+8]*scaleTrans[0]+scaleTrans[3]);
413 			verts.push(vertsArray[i+9]*scaleTrans[1]+scaleTrans[4]);
414 			verts.push(vertsArray[i+10]*scaleTrans[2]+scaleTrans[5]);
415 			var n=this.preNormals[vertsArray[i+3]];
416 			if(!n) n=[0,0,1]; //sanity check
417 			normals.push(n[0]);normals.push(n[1]);normals.push(n[2]);
418 			n=this.preNormals[vertsArray[i+7]];
419 			if(!n) n=[0,0,1]; //sanity check
420 			normals.push(n[0]);normals.push(n[1]);normals.push(-n[2]);
421 			n=this.preNormals[vertsArray[i+11]];
422 			if(!n) n=[0,0,1]; //sanity check
423 			normals.push(n[0]);normals.push(n[1]);normals.push(n[2]);
424 		}
425 		this.verts[j]=verts;
426 		this.normals[j]=normals;
427 	}
428 	MD2Animations[lastFrameName]=[startFrame,j-2];
429 	this.MD2Animations[this.url]=MD2Animations;

431 /**
432 * Process the UV data
433 * @private
434 */
435 GLGE.MD2.prototype.parseUVs=function(){
436 	var uvs=[];
437 	var byteArray=this.byteArray;
438 	var start=this.header.ofs_st;
439 	for(var i=start;i<start+this.header.num_st*4;i=i+4){
440 		uvs.push(this.getUint16At(i)/this.header.skinwidth);
441 		uvs.push(1-this.getUint16At(i+2)/this.header.skinheight);
442 	}
443 	this.globaluvs=uvs;
444 }
445 /**
446 * parses the face data in the md2 file
447 * @private
448 */
449 GLGE.MD2.prototype.parseFaces=function(){
450 	var start=this.header.ofs_tris;
451 	var len=start+this.header.num_tris*12;
452 	var faces=[];
453 	var uvs=[];
454 	var verts=[];
455 	var normals=[];
456 	var idx=0;
457 	for(var i=start;i<len;i=i+12){
458 		faces.push(idx++);
459 		faces.push(idx++);
460 		faces.push(idx++);
461 		var n1=this.getUint16At(i);
462 		var n2=this.getUint16At(i+2);
463 		var	n3=this.getUint16At(i+4);
464 		for(var j=0;j<this.verts.length;j++){
465 			if(!verts[j]){verts[j]=[];normals[j]=[];}
466 			var v=this.verts[j];
467 			var n=this.normals[j];
468 			verts[j].push(v[n1*3]);
469 			verts[j].push(v[n1*3+1]);
470 			verts[j].push(v[n1*3+2]);
471 			normals[j].push(n[n1*3]);
472 			normals[j].push(n[n1*3+1]);
473 			normals[j].push(n[n1*3+2]);
474 			verts[j].push(v[n2*3]);
475 			verts[j].push(v[n2*3+1]);
476 			verts[j].push(v[n2*3+2]);
477 			normals[j].push(n[n2*3]);
478 			normals[j].push(n[n2*3+1]);
479 			normals[j].push(n[n2*3+2]);
480 			verts[j].push(v[n3*3]);
481 			verts[j].push(v[n3*3+1]);
482 			verts[j].push(v[n3*3+2]);
483 			normals[j].push(n[n3*3]);
484 			normals[j].push(n[n3*3+1]);
485 			normals[j].push(n[n3*3+2]);
486 		}
487 		uvs.push(this.globaluvs[this.getUint16At(i+6)*2]);
488 		uvs.push(this.globaluvs[this.getUint16At(i+6)*2+1]);
489 		uvs.push(this.globaluvs[this.getUint16At(i+8)*2]);
490 		uvs.push(this.globaluvs[this.getUint16At(i+8)*2+1]);
491 		uvs.push(this.globaluvs[this.getUint16At(i+10)*2]);
492 		uvs.push(this.globaluvs[this.getUint16At(i+10)*2+1]);
493 	}
494 	this.normals=normals;
495 	this.verts=verts;
496 	this.uvs=uvs;
497 	this.faces=faces;
498 	this.createMesh()
499 }

501 /**
502 * creates the mesh
503 * @private
504 */
505 GLGE.MD2.prototype.createMesh=function(){
506 	var m=new GLGE.Mesh;
507 	var verts=this.verts;
508 	var normals=this.normals;
509 	var uvs=this.uvs;
510 	var faces=this.faces;
511 	for(var i=0;i<verts.length;i++){
512 		m.setPositions(verts[i],i).setNormals(normals[i],i);
513 	}
514 	if(this.doTangents){
515 		m.setUV(uvs).setFaces(faces);
516 	}else{
517 		m.setFaces(faces).setUV(uvs);
518 	}
519 	this.setMesh(m);
520 	this.meshCache[this.url]=m;
521 	this.fireEvent("loaded",{url:this.url});
522 }
