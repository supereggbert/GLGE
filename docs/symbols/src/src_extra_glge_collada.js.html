 70 /**
 71 * Gets the absolute path given an import path and the path it's relative to
 72 * @param {string} path the path to get the absolute path for
 73 * @param {string} relativeto the path the supplied path is relativeto
 74 * @returns {string} absolute path
 75 * @private
 76  77 */
 78 GLGE.Collada.prototype.getAbsolutePath=function(path,relativeto){
 79 	if(path.substr(0,7)=="http://" || path.substr(0,7)=="file://"  || path.substr(0,7)=="https://"){
 80 		return path;
 81 	}
 82 	else
 83 	{
 84 		if(!relativeto){
 85 			relativeto=window.location.href;
 86 		}
 87 		if (relativeto.indexOf("://")==-1){
 88 			return relativeto.slice(0,relativeto.lastIndexOf("/"))+"/"+path;
 89 		}
 90 		//find the path compoents
 91 		var bits=relativeto.split("/");
 92 		var domain=bits[2];
 93 		var proto=bits[0];
 94 		var initpath=[];
 95 		for(var i=3;i<bits.length-1;i++){
 96 			initpath.push(bits[i]);
 97 		}
 98 		//relative to domain
 99 		if(path.substr(0,1)=="/"){
100 			initpath=[];
101 		}
102 		var locpath=path.split("/");
103 		for(i=0;i<locpath.length;i++){
104 			if(locpath[i]=="..") initpath.pop();
105 				else if(locpath[i]!="") initpath.push(locpath[i]);
106 		}
107 		return proto+"//"+domain+"/"+initpath.join("/");
108 	}

135 	var value;

138 	var output=[];
139 	var currentArray;
140 	var i;
141 	while(child){
142 		currentArray=(prev+child.nodeValue).replace(/\s+/g," ").replace(/^\s+/g,"").split(" ");
143 		child=child.nextSibling;
144 		if(currentArray[0]=="") currentArray.unshift();

146 		for(i=0;i<currentArray.length;i++) if(currentArray[i]!="") output.push(currentArray[i]);

165             var authoring=contributor[j].getElementsByTagName("authoring_tool");

171                     return true;
172                 }
173             }
174         }        
175     }

558 					for(k=0;k<8;k++){
559 						newjoints.push(tmp[k].joint);
560 						newweights.push(tmp[k].weight);
561 					}
562 				}
563 				vertexJoints=newjoints;
564 				vertexWeights=newweights;
565 				skeletonData.count=8;
566 			}
567             for (var index=0;index<nummesh;++index) {			
568 			    trimesh[index].setJoints(skeletonData.joints);
569 			    trimesh[index].setInvBindMatrix(skeletonData.inverseBindMatrix);
570                 var maxval=min(MAXVERTS*(index+1)*skeletonData.count,vertexJoints.length);
571                 var minval=MAXVERTS*index*skeletonData.count;
572 			    trimesh[index].setVertexJoints(vertexJoints.slice(minval,maxval),skeletonData.count);
573 			    trimesh[index].setVertexWeights(vertexWeights.slice(minval,maxval),skeletonData.count);
574             }
575 		}
576         for (var index=0;index<nummesh;++index) {		
577 		    trimesh[index].setFaces(faces.slice(0,min(MAXVERTS*(index+1),faces.length)-MAXVERTS*(index)));
578 		    trimesh[index].matName=triangles[i].getAttribute("material");
579             
580 		    meshes.push(trimesh[index]);
581         }
582 	}
583 	meshCache[this.url][id]=meshes;
584 	return meshes;
585 };
