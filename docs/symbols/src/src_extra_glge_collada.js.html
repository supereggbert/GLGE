 69 /**
 70 * Gets the absolute path given an import path and the path it's relative to
 71 * @param {string} path the path to get the absolute path for
 72 * @param {string} relativeto the path the supplied path is relativeto
 73 * @returns {string} absolute path
 74 * @private
 75 */
 76 GLGE.Collada.prototype.getAbsolutePath=function(path,relativeto){
 77 	if(path.substr(0,7)=="http://" || path.substr(0,7)=="file://"  || path.substr(0,7)=="https://"){
 78 		return path;
 79 	}
 80 	else
 81 	{
 82 		if(!relativeto){
 83 			relativeto=window.location.href;
 84 		}
 85 		if (relativeto.indexOf("://")==-1){
 86 			return relativeto.slice(0,relativeto.lastIndexOf("/"))+"/"+path;
 87 		}
 88 		//find the path compoents
 89 		var bits=relativeto.split("/");
 90 		var domain=bits[2];
 91 		var proto=bits[0];
 92 		var initpath=[];
 93 		for(var i=3;i<bits.length-1;i++){
 94 			initpath.push(bits[i]);
 95 		}
 96 		//relative to domain
 97 		if(path.substr(0,1)=="/"){
 98 			initpath=[];
 99 		}
100 		var locpath=path.split("/");
101 		for(i=0;i<locpath.length;i++){
102 			if(locpath[i]=="..") initpath.pop();
103 				else if(locpath[i]!="") initpath.push(locpath[i]);
104 		}
105 		return proto+"//"+domain+"/"+initpath.join("/");
106 	}

133 	var value;

136 	var output=[];
137 	var currentArray;
138 	var i;
139 	while(child){
140 		currentArray=(prev+child.nodeValue).replace(/\s+/g," ").replace(/^\s+/g,"").split(" ");

142 		if(currentArray[0]=="") currentArray.unshift();

163             var authoring=contributor[j].getElementsByTagName("authoring_tool");

169                     return true;
170                 }
171             }
172         }        
173     }

265 	var element=this.xml.getElementById(id);

267         return []

269 		var value;
270 		if(element.tagName=="vertices"){

273 			for(var i=0;i<inputs.length;i++){
274 				value[i]=this.getSource(inputs[i].getAttribute("source").substr(1));

277 		}else{
278 			var accessor=element.getElementsByTagName("technique_common")[0].getElementsByTagName("accessor")[0];
279 			var sourceArray=this.xml.getElementById(accessor.getAttribute("source").substr(1));
280 			var type=sourceArray.tagName;

283 			offset=parseInt(accessor.getAttribute("offset"));

294 	}

556 						newjoints.push(tmp[k].joint);
557 						newweights.push(tmp[k].weight);
558 					}
559 				}
560 				vertexJoints=newjoints;
561 				vertexWeights=newweights;
562 				skeletonData.count=8;
563 			}
564             for (var index=0;index<nummesh;++index) {			
565 			    trimesh[index].setJoints(skeletonData.joints);
566 			    trimesh[index].setInvBindMatrix(skeletonData.inverseBindMatrix);

568                 var minval=MAXVERTS*index*skeletonData.count;
569 			    trimesh[index].setVertexJoints(vertexJoints.slice(minval,maxval),skeletonData.count);
570 			    trimesh[index].setVertexWeights(vertexWeights.slice(minval,maxval),skeletonData.count);
571             }
572 		}
573         for (var index=0;index<nummesh;++index) {		
574 		    trimesh[index].setFaces(faces.slice(0,min(MAXVERTS*(index+1),faces.length)-MAXVERTS*(index)));
575 		    trimesh[index].matName=triangles[i].getAttribute("material");
576             
577 		    meshes.push(trimesh[index]);
578         }
579 	}
580 	meshCache[this.url][id]=meshes;
581 	return meshes;
582 };
