 35 (function(GLGE){
 36 
 37 /**
 38 * @class A wrapping class for jiglib triangle mesh
 39 * @augments GLGE.PhysicsAbstract
 40 */
 41 GLGE.PhysicsMesh=function(uid){
 42 	this.jigLibObj=new jigLib.JTriangleMesh(null, 100, 0.1);
 43 	this.jigLibObj.GLGE=this;
 44 	this.jigLibObj.addEventListener(jigLib.JCollisionEvent.COLLISION, function(event){this.GLGE.fireEvent("collision",{obj:event.collisionBody.GLGE,impulse:event.collisionImpulse})});
 45 	this.dirty=true;
 46 	this.addEventListener("matrixUpdate",this.makeDirty);
 47 	this.addEventListener("childMatrixUpdate",this.makeDirty);
 48 	this.addEventListener("childAdded",this.makeDirty);
 49 	this.addEventListener("childRemoved",this.makeDirty);
 50 	
 51 	GLGE.PhysicsAbstract.call(this,uid);
 52 }
 53 GLGE.augment(GLGE.PhysicsAbstract,GLGE.PhysicsMesh);
 54 
 55 
 56 GLGE.PhysicsMesh.prototype.className="PhysicsMesh";
 57 /**
 58 * Forces and update of the triangle mesh
 59 */
 60 GLGE.PhysicsMesh.prototype.forceUpdate=function(){
 61 	this.dirty=true;
 62 	return this;
 63 }
 64 
 65 /**
 66 * flag to regenerate trimesh and redo octtree
 67 * @private
 68 */
 69 GLGE.PhysicsMesh.prototype.makeDirty=function(e){
 70 	this.dirty=true;
 71 }
 72 /**
 73 * called before a system intergrate
 74 * @private
 75 */
 76 GLGE.PhysicsMesh.prototype.preProcess=function(){
 77 	//recreate mesh and build octree
 78 	if(this.dirty){
 79 		var triangles=this.getTriangles();
 80 		this.jigLibObj.createMesh(triangles.verts, triangles.faces);
 81 		this.dirty=false;
 82 	}
 83 }
 84 /**
 85 * Creates the jiglib triangle arrays from the containing objects
 86 * @private
 87 */
 88 GLGE.PhysicsMesh.prototype.getTriangles=function(){
 89 	var objs=this.getObjects();
 90 	var verts=[];
 91 	var faces=[];
 92 	for(var i=0;i<objs.length;i++){
 93 		if(objs[i].multimaterials){
 94 			var matrix=objs[i].getModelMatrix();
 95 			for(var j=0;j<objs[i].multimaterials.length;j++){
 96 				var mesh=objs[i].multimaterials[j].getMesh();
 97 				var vertcnt=verts.length;
 98 				if(mesh){
 99 					for(var k=0;k<mesh.positions.length;k=k+3){
100 						var vert=[mesh.positions[k],mesh.positions[k+1],mesh.positions[k+2],1];
101 						var v=GLGE.mulMat4Vec4(matrix,vert);
102 						verts.push([v[0],v[1],v[2],1]);
103 					}
104 					var mfaces=mesh.faces.data
105 					if(mfaces){
106 						var len=mfaces.length;
107 						len=((len/3)|0)*3;
108 						for(var k=0;k<len;k=k+3){
109 							faces.push([+mfaces[k]+vertcnt,+mfaces[k+1]+vertcnt,+mfaces[k+2]+vertcnt]);
110 						}
111 					}else{
112 						for(var k=0;k<mesh.positions.length/3;k=k+3){
113 							faces.push([k+vertcnt,k+1+vertcnt,k+2+vertcnt]);
114 						}
115 					}
116 				}
117 			}
118 		}
119 	}
120 	
121 	return {verts:verts,faces:faces};
122 }
