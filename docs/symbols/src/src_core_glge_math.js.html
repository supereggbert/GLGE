282 * @param {Array} a the first vector
283 * @param {Array} b the second vector
284 */
285 GLGE.angleVec3=function(a,b){
286     a=GLGE.toUnitVec3(a);
287     b=GLGE.toUnitVec3(b);
288     d=GLGE.dotVec3(a,b);
289     if (d<-1)
290         d=-1;
291     if (d>1)
292         d=1;
293     return Math.acos(d);
