 66 /**
 67 * @class Group class to allow object transform hierarchies 
 68 * @augments GLGE.Animatable
 69 * @augments GLGE.Placeable
 70 * @augments GLGE.QuickNotation
 71 * @augments GLGE.JSONLoader

 73 GLGE.Group=function(uid){
 74 	this.children=[];
 75     var that=this;
 76     this.downloadComplete=function(){
 77  78         if(that.isComplete()) that.fireEvent("downloadComplete");
 79     }
 80 	GLGE.Assets.registerAsset(this,uid);
 81 }
 82 GLGE.augment(GLGE.Placeable,GLGE.Group);
 83 GLGE.augment(GLGE.Animatable,GLGE.Group);
 84 GLGE.augment(GLGE.QuickNotation,GLGE.Group);
 85 GLGE.augment(GLGE.JSONLoader,GLGE.Group);

 87 GLGE.Group.prototype.className="Group";
 88 GLGE.Group.prototype.type=GLGE.G_NODE;
 89 GLGE.Group.prototype.visible=true;

110 /**
111 * Checks  if resources have finished downloading
112 * @returns {boolean}
113 */
114 GLGE.Group.prototype.isComplete=function(){
115     for(var i=0;i<this.children.length;i++){
116         if(this.children[i].isComplete && !this.children[i].isComplete()){
117             return false;
118         }
119     }
120     return true;
121 }

132 /**
133 * Gets the name of the object and names of any sub objects
134 * @returns an object of name
135 */
136 GLGE.Group.prototype.getNames=function(names){
137 	if(!names) names={};
138 	var thisname=this.getName();
139 	if(thisname!="") names[thisname]=this;
140 	for(var i=0;i<this.children.length;i++){
141 		if(this.children[i].getNames){
142 			this.children[i].getNames(names);
143 		}
144 	}
145 	return names;
146 }
147 /**
148 * Gets the bounding volume for this group
149 * @returns {GLGE.BoundingVolume} 
150 */
151 GLGE.Group.prototype.getBoundingVolume=function(local){
152 	this.boundingVolume=null;
153 	for(var i=0; i<this.children.length;i++){
154 		if(this.children[i].getBoundingVolume){
155 			if(!this.boundingVolume) {
156 				this.boundingVolume=this.children[i].getBoundingVolume(true).clone();
157 			}else{
158 				this.boundingVolume.addBoundingVolume(this.children[i].getBoundingVolume(true));
159 			}
160 		}
161 	}
162 	if(!this.boundingVolume) this.boundingVolume=new GLGE.BoundingVolume(0,0,0,0,0,0);
163 	if(local){
164 		this.boundingVolume.applyMatrix(this.getLocalMatrix());
165 	}else{
166 		this.boundingVolume.applyMatrix(this.getModelMatrix());
167 	}
168 	
169 	return this.boundingVolume;
170 }
171 /**
172 * Gets a list of all objects in this group
173 * @param {array} pointer to an array [optional]
174 * @returns {GLGE.Object[]} an array of GLGE.Objects
175 */
176 GLGE.Group.prototype.getObjects=function(objects){
177 	if(this.lookAt) this.Lookat(this.lookAt);
178 	if(this.animation) this.animate();
179 
180 	if(!objects) objects=[];
181 	for(var i=0; i<this.children.length;i++){
182 		if(this.children[i].className=="Object" || this.children[i].className=="Text" || this.children[i].toRender){
183 			if(this.children[i].visible || this.children[i].visible==undefined){
184 				if(this.children[i].renderFirst) objects.unshift(this.children[i]);
185 					else	objects.push(this.children[i]);
186 			}
187 		}else if(this.children[i].getObjects){
188 			if(this.children[i].visible || this.children[i].visible==undefined){
189 				this.children[i].getObjects(objects);
190 			}
191 		}
192 	}
193 	return objects;
194 }
195 /**
196 * Gets a list of all lights in this group
197 * @param {array} pointer to an array [optional]
198 * @returns {GLGE.Lights[]} an array of GLGE.Lights
199 */
200 GLGE.Group.prototype.getLights=function(lights){
201 	if(!lights) lights=[];
202 	for(var i=0; i<this.children.length;i++){
203 		if(this.children[i].className=="Light"){
204 			lights.push(this.children[i]);
205 		}else if(this.children[i].getLights){
206 			this.children[i].getLights(lights);
207 		}
208 	}
209 	return lights;
210 }

212 /**
213 * Forces an update of all shaders and programs in this group
214 */
215 GLGE.Group.prototype.updateAllPrograms=function(){
216 	var objects=this.getObjects();
217 	for(var i=0;i<objects.length;i++){
218 		if(objects[i].updateProgram) objects[i].updateProgram();
219 	}
220 }

222 /**
223 * Adds a new object to this group
224 * @param {object} object the object to add to this group
225 */
226 GLGE.Group.prototype.addChild=function(object){
227 	if(object.parent) object.parent.removeChild(object);
228 	if(this.noCastShadows!=null && object.noCastShadows==null && object.setCastShadows) object.setCastShadows(!this.noCastShadows);
229 	
230 	GLGE.reuseMatrix4(object.matrix);
231 	object.matrix=null; //clear any cache
232 	object.parent=this;
233 	this.children.push(object);
234 	//if the child added contains lights or is a light then we'll need to update shader programs
235 	if((object.getLights && object.getLights().length>0) || object.className=="Light"){
236 		var root=object;
237 238 		while(root.parent) root=root.parent;
239 		root.updateAllPrograms();
240 	}	
241 	if(object.addEventListener){
242 		object.addEventListener("shaderupdate",function(){
243 			var root=this;
244 			while(root.parent) root=root.parent;
245 			root.updateAllPrograms();
246 		});
247 		object.addEventListener("downloadComplete",this.downloadComplete);
248 	}
249 	this.fireEvent("childAdded",{obj:object});
250 	if(object.fireEvent) object.fireEvent("appened",{obj:this});
251 	this.fireEvent("childAdded",{obj:object});
252 	//fire child added event for all parents as well
253 	var o=this;
254 	while(o=o.parent) o.fireEvent("childAdded",{obj:object,target:this});
255 	return this;
256 }

267 /**
268 * Removes an object or sub group from this group
269 * @param {child} object or index the item to remove
270 */
271 GLGE.Group.prototype.removeChild=function(child){
272 	var object;
273 	if (typeof child == 'object') {
274 		for(var i=0;i<this.children.length;i++){
275 			if(this.children[i]==child) {
276 				child = i;
277 				object = child;
278 				break;
279 			}
280 		}
281 	} else {
282 		if (this.children.length <= child)
283 			return;
284 		
285 		object = this.children[child];
286 	}
287 	
288 	if(this.children[child].removeEventListener){
289 		this.children[child].removeEventListener("downloadComplete",this.downloadComplete);
290 	}
291 	this.children.splice(child, 1);
292 	if(this.scene && this.scene["remove"+object.className]){
293 		this.scene["remove"+object.className](object);
294 	}
295 	if(object.fireEvent) object.fireEvent("removed",{obj:this});
296 	this.fireEvent("childRemoved",{obj:object});
297 	//fire child removed event for all parents as well
298 	var o=this;
299 	while(o=o.parent) o.fireEvent("childRemoved",{obj:object,target:this});
300 }
