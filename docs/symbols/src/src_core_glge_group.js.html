 66 /**
 67 * @class Group class to allow object transform hierarchies 
 68 * @augments GLGE.Animatable
 69 * @augments GLGE.Placeable
 70 * @augments GLGE.QuickNotation

 73 GLGE.Group=function(uid){
 74 	this.children=[];
 75     var that=this;
 76     this.downloadComplete=function(){
 77         if(that.isComplete()) that.fireEvent("downloadComplete");
 78     }
 79 	GLGE.Assets.registerAsset(this,uid);
 80 }
 81 GLGE.augment(GLGE.Placeable,GLGE.Group);

 83 GLGE.augment(GLGE.QuickNotation,GLGE.Group);

 87 GLGE.Group.prototype.type=GLGE.G_NODE;

109 /**
110 * Checks  if resources have finished downloading
111 * @returns {boolean}
112 */
113 GLGE.Group.prototype.isComplete=function(){
114     for(var i=0;i<this.children.length;i++){
115         if(this.children[i].isComplete && !this.children[i].isComplete()){
116             return false;
117         }
118     }
119     return true;

131 /**
132 * Gets the name of the object and names of any sub objects
133 * @returns an object of name
134 */
135 GLGE.Group.prototype.getNames=function(names){
136 	if(!names) names={};
137 	var thisname=this.getName();
138 	if(thisname!="") names[thisname]=this;
139 	for(var i=0;i<this.children.length;i++){
140 		if(this.children[i].getNames){
141 			this.children[i].getNames(names);
142 		}
143 	}
144 	return names;
145 }
146 /**
147 * Gets the bounding volume for this group
148 * @returns {GLGE.BoundingVolume} 
149 */
150 GLGE.Group.prototype.getBoundingVolume=function(local){
151 	this.boundingVolume=null;
152 	for(var i=0; i<this.children.length;i++){
153 		if(this.children[i].getBoundingVolume){
154 			if(!this.boundingVolume) {
155 				this.boundingVolume=this.children[i].getBoundingVolume(true).clone();
156 			}else{
157 				this.boundingVolume.addBoundingVolume(this.children[i].getBoundingVolume(true));
158 			}
159 		}
160 	}
161 	if(!this.boundingVolume) this.boundingVolume=new GLGE.BoundingVolume(0,0,0,0,0,0);
162 	if(local){
163 		this.boundingVolume.applyMatrix(this.getLocalMatrix());
164 	}else{
165 		this.boundingVolume.applyMatrix(this.getModelMatrix());
166 	}
167 	
168 	return this.boundingVolume;
169 }
170 /**
171 * Gets a list of all objects in this group
172 * @param {array} pointer to an array [optional]
173 * @returns {GLGE.Object[]} an array of GLGE.Objects
174 */
175 GLGE.Group.prototype.getObjects=function(objects){
176 	if(this.lookAt) this.Lookat(this.lookAt);
177 	if(this.animation) this.animate();
178 
179 	if(!objects) objects=[];
180 	for(var i=0; i<this.children.length;i++){
181 		if(this.children[i].className=="Object" || this.children[i].className=="Text" || this.children[i].toRender){
182 			if(this.children[i].visible || this.children[i].visible==undefined){
183 				if(this.children[i].renderFirst) objects.unshift(this.children[i]);
184 					else	objects.push(this.children[i]);
185 			}
186 		}else if(this.children[i].getObjects){
187 			if(this.children[i].visible || this.children[i].visible==undefined){
188 				this.children[i].getObjects(objects);
189 			}
190 		}
191 	}
192 	return objects;
193 }
194 /**
195 * Gets a list of all lights in this group
196 * @param {array} pointer to an array [optional]
197 * @returns {GLGE.Lights[]} an array of GLGE.Lights
198 */
199 GLGE.Group.prototype.getLights=function(lights){
200 	if(!lights) lights=[];
201 	for(var i=0; i<this.children.length;i++){
202 		if(this.children[i].className=="Light"){
203 			lights.push(this.children[i]);
204 		}else if(this.children[i].getLights){
205 			this.children[i].getLights(lights);
206 		}
207 	}
208 	return lights;
209 }

211 /**
212 * Forces an update of all shaders and programs in this group
213 */
214 GLGE.Group.prototype.updateAllPrograms=function(){
215 	var objects=this.getObjects();
216 	for(var i=0;i<objects.length;i++){
217 		if(objects[i].updateProgram) objects[i].updateProgram();
218 	}
219 }

221 /**
222 * Adds a new object to this group
223 * @param {object} object the object to add to this group
224 */
225 GLGE.Group.prototype.addChild=function(object){
226 	if(object.parent) object.parent.removeChild(object);
227 	if(this.noCastShadows!=null && object.noCastShadows==null && object.setCastShadows) object.setCastShadows(!this.noCastShadows);
228 	
229 	GLGE.reuseMatrix4(object.matrix);
230 	object.matrix=null; //clear any cache
231 	object.parent=this;
232 	this.children.push(object);
233 	//if the child added contains lights or is a light then we'll need to update shader programs
234 	if((object.getLights && object.getLights().length>0) || object.className=="Light"){
235 		var root=object;
236 		while(root.parent) root=root.parent;
237 		root.updateAllPrograms();
238 	}	
239 	if(object.addEventListener){
240 		object.addEventListener("shaderupdate",function(){
241 			var root=this;
242 			while(root.parent) root=root.parent;
243 			root.updateAllPrograms();
244 		});
245 		object.addEventListener("downloadComplete",this.downloadComplete);
246 	}
247 	this.fireEvent("childAdded",{obj:object});
248 	if(object.fireEvent) object.fireEvent("appened",{obj:this});
249 	this.fireEvent("childAdded",{obj:object});
250 	//fire child added event for all parents as well
251 	var o=this;
252 	while(o=o.parent) o.fireEvent("childAdded",{obj:object,target:this});
253 	return this;
254 }

265 /**
266 * Removes an object or sub group from this group
267 * @param {object} object the item to remove
268 */
269 GLGE.Group.prototype.removeChild=function(object){
270 	for(var i=0;i<this.children.length;i++){
271 		if(this.children[i]==object){
272 			if(this.children[i].removeEventListener){
273 				this.children[i].removeEventListener("downloadComplete",this.downloadComplete);
274 			}
275 			this.children.splice(i, 1);
276 			if(this.scene && this.scene["remove"+object.className]){
277 				this.scene["remove"+object.className](object);
278 			}
279 			if(object.fireEvent) object.fireEvent("removed",{obj:this});
280 			this.fireEvent("childRemoved",{obj:object});
281 			//fire child removed event for all parents as well
282 			var o=this;
283 			while(o=o.parent) o.fireEvent("childRemoved",{obj:object,target:this});
284 			break;
285 		}
286 	}
287 }
