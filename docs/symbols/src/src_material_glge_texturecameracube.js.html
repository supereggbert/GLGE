209 /**
210 * registers the render passes
211 * @private
212 **/
213 GLGE.TextureCameraCube.prototype.doTexture=function(gl,object){
214 
215 	this.gl=gl;
216 	var objMatrix=object.getModelMatrix();
217 	
218 	var height=this.bufferHeight;
219 	var width=this.bufferWidth;
220 
221 	if(!this.cubeBuffers.length || this.update){
222 		this.createFrameBuffers(gl);
223 		this.update=false;
224 		return false;
225 	}
226 
227 	gl.bindTexture(gl.TEXTURE_CUBE_MAP, this.glTexture);
228 	if(!this.rendered || this.autoUpdate){
229 		for(var i=0;i<6;i++){
230 			var matrix=this.cameraMatries[i].slice(0);
231 			var m=GLGE.mulMat4(matrix,objMatrix);
232 			var v=GLGE.mulMat4Vec3(m,[this.offsetX,this.offsetY,this.offsetZ,1]);
233 			matrix[3]=-v[0];
234 			matrix[7]=-v[1];
235 			matrix[11]=-v[2];
236 			gl.scene.addRenderPass(this.cubeBuffers[i],matrix, this.pMatrix,width,height,object,true);
237 		}
238 		this.rendered=true;
239 	}
240 	return true;
241 
242 
243 }
244 GLGE.TextureCameraCube.prototype.registerPasses=GLGE.TextureCameraCube.prototype.doTexture;

246 /**
247 * Creates the frame buffer and texture
248 * @private
249 */
250 GLGE.TextureCameraCube.prototype.createFrameBuffers=function(gl){
251 	var height=this.bufferHeight;
252 	var width=this.bufferWidth;
253 	
254 	
255 	var renderBuffer = gl.createRenderbuffer();
256 	this.glTexture=gl.createTexture();
257 	gl.bindTexture(gl.TEXTURE_CUBE_MAP, this.glTexture);
258 	gl.bindRenderbuffer(gl.RENDERBUFFER, renderBuffer);
259 	gl.renderbufferStorage(gl.RENDERBUFFER, gl.DEPTH_COMPONENT16,width, height);
260 
261 	gl.texImage2D(gl.TEXTURE_CUBE_MAP_POSITIVE_X, 0, gl.RGBA, width,height, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);
262 	gl.texImage2D(gl.TEXTURE_CUBE_MAP_NEGATIVE_X, 0, gl.RGBA, width,height, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);
263 	gl.texImage2D(gl.TEXTURE_CUBE_MAP_POSITIVE_Y, 0, gl.RGBA, width,height, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);
264 	gl.texImage2D(gl.TEXTURE_CUBE_MAP_NEGATIVE_Y, 0, gl.RGBA, width,height, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);
265 	gl.texImage2D(gl.TEXTURE_CUBE_MAP_POSITIVE_Z, 0, gl.RGBA, width,height, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);
266 267 	gl.texImage2D(gl.TEXTURE_CUBE_MAP_NEGATIVE_Z, 0, gl.RGBA, width,height, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);	
268     
269 	gl.texParameteri(gl.TEXTURE_CUBE_MAP, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
270 	gl.texParameteri(gl.TEXTURE_CUBE_MAP, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
271 	gl.texParameteri(gl.TEXTURE_CUBE_MAP, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
272 	gl.texParameteri(gl.TEXTURE_CUBE_MAP, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
273 	
274 	var frameBuffer
275 	frameBuffer = gl.createFramebuffer();
276 	gl.bindFramebuffer(gl.FRAMEBUFFER, frameBuffer);
277 	gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, renderBuffer);
278 	gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_CUBE_MAP_NEGATIVE_X, this.glTexture, 0);	
279 	this.cubeBuffers.push(frameBuffer);	
280 	
281 	frameBuffer = gl.createFramebuffer();
282 	gl.bindFramebuffer(gl.FRAMEBUFFER, frameBuffer);
283 	gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, renderBuffer);
284 	gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_CUBE_MAP_POSITIVE_X, this.glTexture, 0);	
285 	this.cubeBuffers.push(frameBuffer);	
286 		
287 	frameBuffer = gl.createFramebuffer();
288 	gl.bindFramebuffer(gl.FRAMEBUFFER, frameBuffer);
289 	gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, renderBuffer);
290 	gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_CUBE_MAP_NEGATIVE_Y, this.glTexture, 0);	
291 	this.cubeBuffers.push(frameBuffer);
292 		
293 	frameBuffer = gl.createFramebuffer();
294 	gl.bindFramebuffer(gl.FRAMEBUFFER, frameBuffer);
295 	gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, renderBuffer);
296 	gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_CUBE_MAP_POSITIVE_Y, this.glTexture, 0);	
297 	this.cubeBuffers.push(frameBuffer);
298 
299 	frameBuffer = gl.createFramebuffer();
300 	gl.bindFramebuffer(gl.FRAMEBUFFER, frameBuffer);
301 	gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, renderBuffer);
302 	gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_CUBE_MAP_NEGATIVE_Z, this.glTexture, 0);	
303 	this.cubeBuffers.push(frameBuffer);
304 
305 	frameBuffer = gl.createFramebuffer();
306 	gl.bindFramebuffer(gl.FRAMEBUFFER, frameBuffer);
307 	gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, renderBuffer);
308 	gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_CUBE_MAP_POSITIVE_Z, this.glTexture, 0);	
309 	this.cubeBuffers.push(frameBuffer);
