208 /**
209 * registers the render passes
210 * @private
211 **/
212 GLGE.TextureCameraCube.prototype.doTexture=function(gl,object){
213 
214 	this.gl=gl;
215 	var objMatrix=object.getModelMatrix();
216 	
217 	var height=this.bufferHeight;
218 	var width=this.bufferWidth;
219 
220 	if(!this.cubeBuffers.length || this.update){
221 		this.createFrameBuffers(gl);
222 		this.update=false;
223 		return false;
224 	}
225 
226 	gl.bindTexture(gl.TEXTURE_CUBE_MAP, this.glTexture);
227 	if(!this.rendered || this.autoUpdate){
228 		for(var i=0;i<6;i++){
229 			var matrix=this.cameraMatries[i].slice(0);
230 			var m=GLGE.mulMat4(matrix,objMatrix);
231 			var v=GLGE.mulMat4Vec3(m,[this.offsetX,this.offsetY,this.offsetZ,1]);
232 			matrix[3]=-v[0];
233 			matrix[7]=-v[1];
234 			matrix[11]=-v[2];
235 			gl.scene.addRenderPass(this.cubeBuffers[i],matrix, this.pMatrix,width,height,object,true);
236 		}
237 		this.rendered=true;
238 	}
239 	return true;
240 
241 
242 }
243 GLGE.TextureCameraCube.prototype.registerPasses=GLGE.TextureCameraCube.prototype.doTexture;

245 /**
246 * Creates the frame buffer and texture
247 * @private
248 */
249 GLGE.TextureCameraCube.prototype.createFrameBuffers=function(gl){
250 	var height=this.bufferHeight;
251 	var width=this.bufferWidth;
252 	
253 	
254 	var renderBuffer = gl.createRenderbuffer();
255 	this.glTexture=gl.createTexture();
256 	gl.bindTexture(gl.TEXTURE_CUBE_MAP, this.glTexture);
257 	gl.bindRenderbuffer(gl.RENDERBUFFER, renderBuffer);
258 	gl.renderbufferStorage(gl.RENDERBUFFER, gl.DEPTH_COMPONENT16,width, height);
259 
260 	gl.texImage2D(gl.TEXTURE_CUBE_MAP_POSITIVE_X, 0, gl.RGBA, width,height, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);
261 	gl.texImage2D(gl.TEXTURE_CUBE_MAP_NEGATIVE_X, 0, gl.RGBA, width,height, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);
262 	gl.texImage2D(gl.TEXTURE_CUBE_MAP_POSITIVE_Y, 0, gl.RGBA, width,height, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);
263 	gl.texImage2D(gl.TEXTURE_CUBE_MAP_NEGATIVE_Y, 0, gl.RGBA, width,height, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);
264 	gl.texImage2D(gl.TEXTURE_CUBE_MAP_POSITIVE_Z, 0, gl.RGBA, width,height, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);
265 	gl.texImage2D(gl.TEXTURE_CUBE_MAP_NEGATIVE_Z, 0, gl.RGBA, width,height, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);	
266     
267 	gl.texParameteri(gl.TEXTURE_CUBE_MAP, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
268 	gl.texParameteri(gl.TEXTURE_CUBE_MAP, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
269 	gl.texParameteri(gl.TEXTURE_CUBE_MAP, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
270 	gl.texParameteri(gl.TEXTURE_CUBE_MAP, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
271 	
272 	var frameBuffer
273 	frameBuffer = gl.createFramebuffer();
274 	gl.bindFramebuffer(gl.FRAMEBUFFER, frameBuffer);
275 	gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, renderBuffer);
276 	gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_CUBE_MAP_NEGATIVE_X, this.glTexture, 0);	
277 	this.cubeBuffers.push(frameBuffer);	
278 	
279 	frameBuffer = gl.createFramebuffer();
280 	gl.bindFramebuffer(gl.FRAMEBUFFER, frameBuffer);
281 	gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, renderBuffer);
282 	gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_CUBE_MAP_POSITIVE_X, this.glTexture, 0);	
283 	this.cubeBuffers.push(frameBuffer);	
284 		
285 	frameBuffer = gl.createFramebuffer();
286 	gl.bindFramebuffer(gl.FRAMEBUFFER, frameBuffer);
287 	gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, renderBuffer);
288 	gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_CUBE_MAP_NEGATIVE_Y, this.glTexture, 0);	
289 	this.cubeBuffers.push(frameBuffer);
290 		
291 	frameBuffer = gl.createFramebuffer();
292 	gl.bindFramebuffer(gl.FRAMEBUFFER, frameBuffer);
293 	gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, renderBuffer);
294 	gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_CUBE_MAP_POSITIVE_Y, this.glTexture, 0);	
295 	this.cubeBuffers.push(frameBuffer);
296 
297 	frameBuffer = gl.createFramebuffer();
298 	gl.bindFramebuffer(gl.FRAMEBUFFER, frameBuffer);
299 	gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, renderBuffer);
300 	gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_CUBE_MAP_NEGATIVE_Z, this.glTexture, 0);	
301 	this.cubeBuffers.push(frameBuffer);
302 
303 	frameBuffer = gl.createFramebuffer();
304 	gl.bindFramebuffer(gl.FRAMEBUFFER, frameBuffer);
305 	gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, renderBuffer);
306 	gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_CUBE_MAP_POSITIVE_Z, this.glTexture, 0);	
307 	this.cubeBuffers.push(frameBuffer);
308
