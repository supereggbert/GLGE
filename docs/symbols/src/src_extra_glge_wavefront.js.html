 51 /**
 52 * Gets the absolute path given an import path and the path it's relative to
 53 * @param {string} path the path to get the absolute path for
 54 * @param {string} relativeto the path the supplied path is relativeto
 55 * @returns {string} absolute path
 56 * @private
 57 */
 58 GLGE.Wavefront.prototype.getAbsolutePath=function(path,relativeto){
 59 	if(path.substr(0,7)=="http://" || path.substr(0,7)=="file://"  || path.substr(0,7)=="https://"){
 60 		return path;
 61 	}
 62 	else
 63 	{
 64 		if(!relativeto){
 65 			relativeto=window.location.href;
 66 		}
 67 		if(relativeto.indexOf("?")>0){
 68 			relativeto=relativeto.substr(0,relativeto.indexOf("?"));
 69 		}
 70 		//find the path compoents
 71 		var bits=relativeto.split("/");
 72 		var domain=bits[2];
 73 		var proto=bits[0];
 74 		var initpath=[];
 75 		for(var i=3;i<bits.length-1;i++){
 76 			initpath.push(bits[i]);
 77 		}
 78 		//relative to domain
 79 		if(path.substr(0,1)=="/"){
 80 			initpath=[];
 81 		}
 82 		var locpath=path.split("/");
 83 		for(i=0;i<locpath.length;i++){
 84 			if(locpath[i]=="..") initpath.pop();
 85 				else if(locpath[i]!="") initpath.push(locpath[i]);
 86 		}
 87 		return proto+"//"+domain+"/"+initpath.join("/");
 88 	}
 89 };

 93 /**
 94 * Loads a material file from a url
 95 * @param {string} url the url of the material file
 96 * @private
 97 */
 98 GLGE.Wavefront.prototype.loadMaterials=function(url){
 99 	if(!this.loading){
100 		this.loadFile(url,null,function(url,text){
101 			this.parseMaterials(text.split("\n"));
102 			if(this.queue.length>0){
103 				var matUrl=this.queue.pop();
104 				this.loadMaterials(matUrl,this.src);
105 			}else{
106 				this.parseMesh();
107 				this.fireEvent("loaded",{});
108 			}
109 		});
110 	}else{
111 		this.queue.push(url);
112 	}
113 
114 };
115 /**
116 * creates the GLGE materials from a mtl file
117 * @param {string} file the file to parse
118 * @private
119 */
120 GLGE.Wavefront.prototype.parseMaterials=function(file){
121 	//loop though all lines and look for matlibs
122 	var j = 0;
123 	var i = 0;
124 	var index = 0;
125 	var idNameMaterial;
126 	while(i<file.length)
127 	{
128 		//newmtl
129 		if(file[i].substr(0,6)=="newmtl")
130 		{
131 			var data=file[i].replace(/^\s+|\s+$/g,"").replace(/\s+/g," ").split(" ");
132 			var material=new GLGE.Material;
133 			idNameMaterial = file[j].substr(7);
134 			j=i+1;
135 			
136 			while(file[j].substr(0,6) != "newmtl")
137 			{
138 				
139 				data=file[j].replace(/^\s+|\s+$/g,"").replace(/\s+/g," ").split(" ");
140 				if(data.length>1)
141 				{
142 					switch(data[0]){
143 						case "Kd":
144 							material.setColorR(parseFloat(data[1]));
145 							material.setColorG(parseFloat(data[2]));
146 							material.setColorB(parseFloat(data[3]));
147 							break;
148 						case "Ks":
149 							material.setSpecularColor({r:parseFloat(data[1]),g:parseFloat(data[2]),b:parseFloat(data[3])});
150 							break;
151 						case "Ns":
152 							material.setShininess(parseFloat(data[1]));
153 							break;
154 						case "d":
155 							this.setZtransparent(true);
156 							material.setAlpha(parseFloat(data[1]));
157 							break;
158 						case "map_Kd":
159 							var ml=new GLGE.MaterialLayer;
160 							ml.setMapto(GLGE.M_COLOR);
161 							ml.setMapinput(GLGE.UV1);
162 							var tex=new GLGE.Texture;
163 							var k=1;
164 							while(data[k][0]=="-") k=k+2;
165 							tex.setSrc(this.getAbsolutePath(data[k],this.relativeTo));
166 							material.addTexture(tex);
167 							ml.setTexture(tex);
168 							material.addMaterialLayer(ml);
169 							break;
170 						case "map_Ks":
171 						case "map_spec":
172 							var ml=new GLGE.MaterialLayer;
173 							ml.setMapto(GLGE.M_SPECULAR);
174 							ml.setMapinput(GLGE.UV1);
175 							var tex=new GLGE.Texture;
176 							var k=1;
177 							while(data[k][0]=="-") k=k+2;
178 							tex.setSrc(this.getAbsolutePath(data[k],this.relativeTo));
179 							material.addTexture(tex);
180 							ml.setTexture(tex);
181 							material.addMaterialLayer(ml);
182 							break;
183 						case "bump":
184 						case "map_bump":
185 							var ml=new GLGE.MaterialLayer;
186 							ml.setMapto(GLGE.M_NOR);
187 							ml.setMapinput(GLGE.UV1);
188 							var tex=new GLGE.Texture;
189 							var k=1;
190 							while(data[k][0]=="-") k=k+2;
191 							tex.setSrc(this.getAbsolutePath(data[k],this.relativeTo));
192 							material.addTexture(tex);
193 							ml.setTexture(tex);
194 							material.addMaterialLayer(ml);
195 							break;
196 					}
197 				}
198 				j++;
199 				if(j>=file.length)
200 					break;
201 			}
202 			i=j-1;
203 			this.materials[index]=material;
204 			this.idMaterials.push(idNameMaterial);
205 			index++;
206 		}
207 		i++;
208 	}
209 };
210 /**
211 * loads a resource from a url
212 * @param {string} url the url of the resource to load
213 * @param {string} relativeTo the url to load relative to
214 * @param {function} callback thefunction to call once the file is loaded
215 * @private
216 */
217 GLGE.Wavefront.prototype.loadFile=function(url,relativeTo,callback){
218 	this.loading=true;
219 	if(!callback) callback=this.loaded;
220 	if(!relativeTo && this.relativeTo) relativeTo=this.relativeTo;
221 	url=this.getAbsolutePath(url,relativeTo);
222 	if(!this.relativeTo) this.relativeTo=url;
223 	var req = new XMLHttpRequest();
224 	var that=this;
225 	if(req) {
226 		req.overrideMimeType("text/plain")
227 		req.onreadystatechange = function() {
228 			if(this.readyState  == 4)
229 			{
230 				if(this.status  == 200 || this.status==0){
231 					that.loading=false;
232 					callback.call(that,url,this.responseText);
233 				}else{ 
234 					GLGE.error("Error loading Document: "+url+" status "+this.status);
235 				}
236 			}
237 		};
238 		req.open("GET", url, true);
239 		req.send("");
240 	}	
241 }

257 GLGE.Wavefront.prototype.loaded=function(url,objfile){
258 	this.file=objArray=objfile.split("\n");
259 	var hasMaterial=false;
260 	//loop through the file and load the Materials
261 	for(var i=0;i<objArray.length;i++){
262 		var data=objArray[i].split(" ");
263 		if(data.length>1){
264 			if(data[0]=="mtllib"){
265 				hasMaterial=true;
266 				this.loadMaterials(data[1]);
267 			}
268 		}
269 	}
270 	if(!hasMaterial){
271 		this.parseMesh();
272 		this.fireEvent("loaded",{});
273 	}
274

276 /**
277 * creates a new multimaterial
278 * @private
279 */
280 GLGE.Wavefront.prototype.createMultiMaterial=function(idxDataOrig,idxDataOrigMap,verts,norms,texCoords,faces,material,smooth){
281 	//loop though the indexes to produce streams
282 	var positions=[];
283 	var normals=[];
284 	var uv=[];
285 	var newfaces=[];
286 	var idxData=[];
287 	var idxDataMap={};
288 	for(var i=0;i<faces.length;i++){
289 		var data=idxDataOrig[faces[i]];
290 		var idx=idxDataMap[data];
291 		if((typeof idx === "undefined") || !smooth){
292 			idxData.push(data);
293 			idxDataMap[data]=idxData.length-1;
294 			newfaces.push(idxData.length-1);
295 		}else{
296 			newfaces.push(idxDataMap[data]);
297 		}
298 	}
299 	faces=newfaces;
300 	for(i=0;i<idxData.length;i++){
301 		if(idxData[i].indexOf("/")>0) var vertData=idxData[i].split("/");
302 			else var vertData=[idxData[i]];
303 		if(!verts[vertData[0]-1]) GLGE.error(vertData[0]);
304 		positions.push(verts[vertData[0]-1][1]);
305 		positions.push(verts[vertData[0]-1][2]);
306 		positions.push(verts[vertData[0]-1][3]);
307 		if(vertData[1]){
308 			uv.push(texCoords[vertData[1]-1][1]);
309 			uv.push(texCoords[vertData[1]-1][2]);
310 		}
311 		if(vertData[2]){
312 			normals.push(norms[vertData[2]-1][1]);
313 			normals.push(norms[vertData[2]-1][2]);
314 			normals.push(norms[vertData[2]-1][3]);
315 		}
316 	}
317 	if(positions.length/3>65024){
318 		var newPositions=[];
319 		var newNormals=[];
320 		var newUVs=[];
321 		for(var i=0;i<faces.length;i++){
322 			newPositions.push(positions[faces[i]*3],positions[faces[i]*3+1],positions[faces[i]*3+2]);
323 			if(normals.length>0) newNormals.push(normals[faces[i]*3],normals[faces[i]*3+1],normals[faces[i]*3+2]);
324 			if(uv.length>0) newUVs.push(normals[faces[i]*2],normals[faces[i]*2+1]);
325 		}
326 		positions=newPositions;
327 		normals=newNormals;
328 		uv=newUVs;
329 		faces=[];
330 	}
331 	var multiMat=new GLGE.MultiMaterial;
332 	var mesh=new GLGE.Mesh;
333 	
334 	mesh.setPositions(positions);
335 	if(normals.length>0) mesh.setNormals(normals);
336 	if(uv.length>0) mesh.setUV(uv);
337 	if(faces.length>0) mesh.setFaces(faces);
338 	multiMat.setMesh(mesh);
339 	multiMat.setMaterial(material);
340 	this.addMultiMaterial(multiMat);
341 
342 }
343 /**
344 * Parses the mesh
345 * @private
346 */
347 GLGE.Wavefront.prototype.parseMesh=function(){
348 	objArray=this.file;
349 	var texCoords=[];
350 	var verts=[];
351 	var norms=[];
352 	var faces=[];
353 	var idxData=[];
354 	var idxDataMap={};
355 	var vertoffset=0;
356 	var smooth=true;
357 	var material=new GLGE.Material;
358 	for(var i=0;i<objArray.length;i++){
359 		if(objArray[i][0]!="#"){
360 			var data=objArray[i].replace(/^\s+|\s+$/g,"").replace(/\s+/g," ").split(" ");
361 			if(data.length>0){
362 				switch(data[0]){
363 					case "s":
364 						if(data[1]=="1") smooth=true;
365 							else smooth=false;
366 					case "o":
367 						if(faces.length>0){
368 							this.createMultiMaterial(idxData,idxDataMap,verts,norms,texCoords,faces,material,smooth);
369 							faces=[];
370 							material=new GLGE.Material;
371 						}
372 						break;
373 					case "usemtl":
374 						if(faces.length>0){
375 							this.createMultiMaterial(idxData,idxDataMap,verts,norms,texCoords,faces,material,smooth);
376 							faces=[];
377 						}
378 						if(this.idMaterials.indexOf(data[1]) == -1)//Material no name 
379 							material=this.materials[0];//default
380 						else
381 							material=this.materials[this.idMaterials.indexOf(data[1])];//get Idname material
382 						break;
383 					case "v":
384 						verts.push(data);
385 						break;
386 					case "vt":
387 						texCoords.push(data);
388 						break;
389 					case "vn":
390 						norms.push(data);
391 						break;
392 					case "f":
393 						var tmpface=[];
394 						for(var j=1;j<data.length;j++){
395 							var idx=idxDataMap[data[j]];
396 							if((typeof idx === "undefined") || !smooth){
397 								idxData.push(data[j]);
398 								idx=idxData.length-1;
399 								idxDataMap[data[j]]=idx;
400 							}
401 							tmpface.push(idx);
402 						}
403 						for(j=0;j<tmpface.length-2;j++){
404 							faces.push(tmpface[0]-vertoffset);
405 							faces.push(tmpface[1+j]-vertoffset);
406 							faces.push(tmpface[2+j]-vertoffset);
407 						}
408 						break;
409 				}
410 			}
411 		}
412 	}
413 	this.createMultiMaterial(idxData,idxDataMap,verts,norms,texCoords,faces,material,smooth);
414 };
