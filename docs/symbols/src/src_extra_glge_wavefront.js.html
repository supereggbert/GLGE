 36 /**
 37 * @class parses and displays a warefront object file with mtl material
 38 * @param {string} uid the unique id for this object
 39 * @augments GLGE.Object
 40 */
 41 GLGE.Wavefront=function(uid){
 42 	this.multimaterials=[];
 43 	this.materials={};
 44 	this.instances=[];
 45 	this.queue=[];
 46 	GLGE.Object.call(this,uid);
 47 	GLGE.Assets.registerAsset(this,uid);
 48 }
 49 GLGE.augment(GLGE.Object,GLGE.Wavefront);
 50 /**
 51 * Gets the absolute path given an import path and the path it's relative to
 52 * @param {string} path the path to get the absolute path for
 53 * @param {string} relativeto the path the supplied path is relativeto
 54 * @returns {string} absolute path
 55 * @private
 56 */
 57 GLGE.Wavefront.prototype.getAbsolutePath=function(path,relativeto){
 58 	if(path.substr(0,7)=="http://" || path.substr(0,7)=="file://"  || path.substr(0,7)=="https://"){
 59 		return path;
 60 	}
 61 	else
 62 	{
 63 		if(!relativeto){
 64 			relativeto=window.location.href;
 65 		}
 66 		if(relativeto.indexOf("?")>0){
 67 			relativeto=relativeto.substr(0,relativeto.indexOf("?"));
 68 		}
 69 		//find the path compoents
 70 		var bits=relativeto.split("/");
 71 		var domain=bits[2];
 72 		var proto=bits[0];
 73 		var initpath=[];
 74 		for(var i=3;i<bits.length-1;i++){
 75 			initpath.push(bits[i]);
 76 		}
 77 		//relative to domain
 78 		if(path.substr(0,1)=="/"){
 79 			initpath=[];
 80 		}
 81 		var locpath=path.split("/");
 82 		for(i=0;i<locpath.length;i++){
 83 			if(locpath[i]=="..") initpath.pop();
 84 				else if(locpath[i]!="") initpath.push(locpath[i]);
 85 		}
 86 		return proto+"//"+domain+"/"+initpath.join("/");
 87 	}
 88 };

 92 /**
 93 * Loads a material file from a url
 94 * @param {string} url the url of the material file
 95 * @private
 96 */
 97 GLGE.Wavefront.prototype.loadMaterials=function(url){
 98 	if(!this.loading){
 99 		this.loadFile(url,null,function(url,text){
100 			this.parseMaterials(text.split("\n"));
101 			if(this.queue.length>0){
102 				var matUrl=this.queue.pop();
103 				this.loadMaterials(matUrl,this.src);
104 			}else{
105 				this.parseMesh();
106 				this.fireEvent("loaded",{});
107 			}
108 		});
109 	}else{
110 		this.queue.push(url);
111 	}
112 
113 };
114 /**
115 * creates the GLGE materials from a mtl file
116 * @param {string} file the file to parse
117 * @private
118 */
119 GLGE.Wavefront.prototype.parseMaterials=function(file){
120 	//loop though all lines and look for matlibs
121 	for(var i=0;i<file.length;i++){
122 		//newmtl
123 		if(file[i].substr(0,6)=="newmtl"){
124 			var data=file[i].replace(/^\s+|\s+$/g,"").replace(/\s+/g," ").split(" ");
125 			var material=new GLGE.Material;
126 			this.materials[data[1].replace(/\s+$/,"").replace("\r","").replace("\t","")]=material;
127 			i++;
128 		}
129 		var data=file[i].replace(/^\s+|\s+$/g,"").replace(/\s+/g," ").split(" ");
130 		if(data.length>1){
131 			switch(data[0]){
132 				case "Kd":
133 					material.setColorR(parseFloat(data[1]));
134 					material.setColorG(parseFloat(data[2]));
135 					material.setColorB(parseFloat(data[3]));
136 					break;
137 				case "Ks":
138 					material.setSpecularColor({r:parseFloat(data[1]),g:parseFloat(data[2]),b:parseFloat(data[3])});
139 					break;
140 				case "Ns":
141 					material.setShininess(parseFloat(data[1]));
142 					break;
143 				case "d":
144 					this.setZtransparent(true);
145 					material.setAlpha(parseFloat(data[1]));
146 					break;
147 				case "map_Kd":
148 					var ml=new GLGE.MaterialLayer;
149 					ml.setMapto(GLGE.M_COLOR);
150 					ml.setMapinput(GLGE.UV1);
151 					var tex=new GLGE.Texture;
152 					var k=1;
153 					while(data[k][0]=="-") k=k+2;
154 					tex.setSrc(this.getAbsolutePath(data[k],this.relativeTo));
155 					material.addTexture(tex);
156 					ml.setTexture(tex);
157 					material.addMaterialLayer(ml);
158 				case "map_Ks":
159 				case "map_spec":
160 					var ml=new GLGE.MaterialLayer;
161 					ml.setMapto(GLGE.M_SPECULAR);
162 					ml.setMapinput(GLGE.UV1);
163 					var tex=new GLGE.Texture;
164 					var k=1;
165 					while(data[k][0]=="-") k=k+2;
166 					tex.setSrc(this.getAbsolutePath(data[k],this.relativeTo));
167 					material.addTexture(tex);
168 					ml.setTexture(tex);
169 					material.addMaterialLayer(ml);
170 				case "bump":
171 				case "map_bump":
172 					var ml=new GLGE.MaterialLayer;
173 174 					ml.setMapto(GLGE.M_NOR);
175 					ml.setMapinput(GLGE.UV1);
176 					var tex=new GLGE.Texture;
177 					var k=1;
178 					while(data[k][0]=="-") k=k+2;
179 					tex.setSrc(this.getAbsolutePath(data[k],this.relativeTo));
180 					material.addTexture(tex);
181 					ml.setTexture(tex);
182 					material.addMaterialLayer(ml);
183 			}
184 		}
185 	}
186 };
187 /**
188 * loads a resource from a url
189 * @param {string} url the url of the resource to load
190 * @param {string} relativeTo the url to load relative to
191 * @param {function} callback thefunction to call once the file is loaded
192 * @private
193 */
194 GLGE.Wavefront.prototype.loadFile=function(url,relativeTo,callback){
195 	this.loading=true;
196 	if(!callback) callback=this.loaded;
197 	if(!relativeTo && this.relativeTo) relativeTo=this.relativeTo;
198 199 	url=this.getAbsolutePath(url,relativeTo);
200 	if(!this.relativeTo) this.relativeTo=url;
201 	var req = new XMLHttpRequest();
202 	var that=this;
203 	if(req) {
204 		req.overrideMimeType("text/plain")
205 		req.onreadystatechange = function() {
206 			if(this.readyState  == 4)
207 			{
208 				if(this.status  == 200 || this.status==0){
209 					that.loading=false;
210 					callback.call(that,url,this.responseText);
211 				}else{ 
212 					GLGE.error("Error loading Document: "+url+" status "+this.status);
213 				}
214 			}
215 		};
216 		req.open("GET", url, true);
217 		req.send("");
218 	}	
219 }

229 /**
230 * loads a resource from a url
231 * @param {string} url the url of the resource loaded
232 * @param {string} objfile the loaded file
233 * @private
234 */
235 GLGE.Wavefront.prototype.loaded=function(url,objfile){
236 	this.file=objArray=objfile.split("\n");
237 	var hasMaterial=false;
238 	//loop through the file and load the Materials
239 	for(var i=0;i<objArray.length;i++){
240 		var data=objArray[i].split(" ");
241 		if(data.length>1){
242 			if(data[0]=="mtllib"){
243 				hasMaterial=true;
244 				this.loadMaterials(data[1]);
245 			}
246 		}
247 	}
248 	if(!hasMaterial){
249 		this.parseMesh();
250 		this.fireEvent("loaded",{});
251 	}
252 	
253 };
254 /**
255 * creates a new multimaterial
256 * @private
257 */
258 GLGE.Wavefront.prototype.createMultiMaterial=function(idxDataOrig,verts,norms,texCoords,faces,material,smooth){
259 	//loop though the indexes to produce streams
260 	var positions=[];
261 	var normals=[];
262 	var uv=[];
263 	var newfaces=[];
264 	var idxData=[];
265 	for(var i=0;i<faces.length;i++){
266 		var data=idxDataOrig[faces[i]];
267 		if(idxData.indexOf(data)==-1 || !smooth){
268 			idxData.push(data);
269 			newfaces.push(idxData.length-1);
270 		}else{
271 			newfaces.push(idxData.indexOf(data));
272 		}
273 	}
274 	faces=newfaces;
275 	for(i=0;i<idxData.length;i++){
276 		if(idxData[i].indexOf("/")>0) var vertData=idxData[i].split("/");
277 			else var vertData=[idxData[i]];
278 		if(!verts[vertData[0]-1]) GLGE.error(vertData[0]);
279 		positions.push(verts[vertData[0]-1][1]);
280 		positions.push(verts[vertData[0]-1][2]);
281 		positions.push(verts[vertData[0]-1][3]);
282 		if(vertData[1]){
283 			uv.push(texCoords[vertData[1]-1][1]);
284 			uv.push(texCoords[vertData[1]-1][2]);
285 		}
286 		if(vertData[2]){
287 			normals.push(norms[vertData[2]-1][1]);
288 			normals.push(norms[vertData[2]-1][2]);
289 			normals.push(norms[vertData[2]-1][3]);
290 		}
291 	}
292 	var multiMat=new GLGE.MultiMaterial;
293 	var mesh=new GLGE.Mesh;
294 	
295 	mesh.setPositions(positions);
296 	if(uv.length>0) mesh.setUV(uv);
297 	if(normals.length>0) mesh.setNormals(normals);
298 	mesh.setFaces(faces);
299 	multiMat.setMesh(mesh);
300 	multiMat.setMaterial(material);
301 	this.addMultiMaterial(multiMat);
302 }
303 /**
304 * Parses the mesh
305 * @private
306 */
307 GLGE.Wavefront.prototype.parseMesh=function(){
308 	objArray=this.file;
309 	var texCoords=[];
310 	var verts=[];
311 	var norms=[];
312 	var faces=[];
313 	var idxData=[];
314 	var vertoffset=0;
315 	var smooth=true;
316 	var material=new GLGE.Material;
317 	for(var i=0;i<objArray.length;i++){
318 		if(objArray[i][0]!="#"){
319 			var data=objArray[i].replace(/^\s+|\s+$/g,"").replace(/\s+/g," ").split(" ");
320 			if(data.length>0){
321 				switch(data[0]){
322 					case "s":
323 						if(data[1]=="1") smooth=true;
324 							else smooth=false;
325 					case "o":
326 						if(faces.length>0){
327 							this.createMultiMaterial(idxData,verts,norms,texCoords,faces,material,smooth);
328 							faces=[];
329 							material=new GLGE.Material;
330 						}
331 						break;
332 					case "usemtl":
333 						if(faces.length>0){
334 							this.createMultiMaterial(idxData,verts,norms,texCoords,faces,material,smooth);
335 							faces=[];
336 						}
337 						material=this.materials[data[1]];
338 						break;
339 					case "v":
340 						verts.push(data);
341 						break;
342 					case "vt":
343 						texCoords.push(data);
344 						break;
345 					case "vn":
346 						norms.push(data);
347 						break;
348 					case "f":
349 						var tmpface=[];
350 						for(var j=1;j<data.length;j++){
351 							var idx=idxData.indexOf(data[j]);
352 							if(idx==-1 || !smooth){
353 								idxData.push(data[j]);
354 								idx=idxData.length-1;
355 							}
356 							tmpface.push(idx);
357 						}
358 						for(j=0;j<tmpface.length-2;j++){
359 							faces.push(tmpface[0]-vertoffset);
360 							faces.push(tmpface[1+j]-vertoffset);
361 							faces.push(tmpface[2+j]-vertoffset);
362 						}
363 						break;
364 				}
365 			}
366 		}
367 	}
368 	this.createMultiMaterial(idxData,verts,norms,texCoords,faces,material,smooth);
369 };

371 /**
372 * Parses the dom element and creates a texture
373 * @param {domelement} ele the element to create the objects from
374 * @private
375 */
376 GLGE.Document.prototype.getWavefront=function(ele){
377 	if(!ele.object){
378 		var rel=this.getAbsolutePath(this.rootURL,null);
379 		ele.object=new GLGE[this.classString(ele.tagName)];
380 		//ele.object.setSrc(this.getAbsolutePath(ele.getAttribute("src"),rel));
381 		ele.object.setSrc(ele.getAttribute("src"),rel);
382 		ele.removeAttribute("src");
383 		this.setProperties(ele);
384 	}
385 	return ele.object;
386 }
