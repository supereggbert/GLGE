 36 (function(GLGE){
 37 
 38 /**
 39 * @class A class to load json fragments from remote location or string
 40 **/
 41 GLGE.JSONLoader=function(){
 42 }
 43 GLGE.JSONLoader.prototype.downloadPriority=0;
 44 /**
 45 * Loads a json fragment from a url
 46 * @param {string} url The URL to load
 47 **/
 48 GLGE.JSONLoader.prototype.setJSONSrc=function(url){
 49 	var GLGEObj=this;
 50 	GLGE.Message.messageLoader(url,function(text){
 51 		GLGEObj.setJSONString(text);
 52 	},this.downloadPriority);
 53 }
 54 /**
 55 * Loads a json fragment from a string
 56 * @param {string} string The URL to load
 57 **/
 58 GLGE.JSONLoader.prototype.setJSONString=function(string){
 59 	var message = JSON.parse(string);
 60 	//check to make sure this is the correct class type
 61 	if(message.type==this.className){
 62 		message.uid=this.uid;
 63 		//we don't want to create a new one we want to update this one
 64 		message.command="update";
 65 		GLGE.Message.parseMessage(message);
 66 	}
 67 }
 68 /**
 69 * Sets the download priority
 70 * @param {number} value The download priority
 71 **/
 72 GLGE.JSONLoader.prototype.setDownloadPriority=function(value){
 73 	this.downloadPriority=value;
 74 }
 75 /**
 76 * Gets the download priority
 77 * @returns {number} The download priority
 78 **/
 79 GLGE.JSONLoader.prototype.getDownloadPriority=function(){
 80 	return this.downloadPriority;
 81 }
 82 
 83 })(GLGE);
