220 /**
221 * Method gets the current projection matrix of this camera
222 * @return {Matrix} Returns the camera projection matrix
223 */
224 GLGE.Camera.prototype.getProjectionMatrix=function(){
225 	if(!this.pMatrix){
226 		if(this.pMatrixOveride){
227 			this.pMatrix=this.pMatrixOveride;
228 		}else{
229 			switch(this.type){
230 				case GLGE.C_PERSPECTIVE:
231 					this.pMatrix=GLGE.makePerspective(this.fovy, this.aspect, this.near, this.far);
232 					break;
233 				case GLGE.C_ORTHO:
234 					this.pMatrix=GLGE.makeOrtho(-this.orthoscale*this.aspect,this.orthoscale*this.aspect,-this.orthoscale,this.orthoscale, this.near, this.far);
235 					break;
236 			}
237 		}
238 	}
239 	return this.pMatrix;
240 };
241 /**
242 * Method generates the projection matrix based on the 
243 * camera paramaters
244 * @param {Matrix} projection The new projection matrix
245 */
246 GLGE.Camera.prototype.setProjectionMatrix=function(projection){
247 	this.pMatrix=projection;
248 	return this;
249 };
250 /**
251 * Method sets a custom projection matrix
252 * @param {Matrix} projection The new projection matrix
253 */
254 GLGE.Camera.prototype.setCustomProjectionMatrix=function(projection){
255 	this.pMatrix=projection;
256 	this.pMatrixOveride=projection;
257 	return this;
258 };
259 /**
260 * Method generates the cameras view matrix
261 * @return Returns the view matrix based on this camera
262 * @type Matrix
263 */
264 GLGE.Camera.prototype.updateMatrix=function(){
265 266 	var position=this.getPosition();
267 	var vMatrix=GLGE.translateMatrix(position.x,position.y,position.z);
268 	vMatrix=GLGE.mulMat4(vMatrix,this.getRotMatrix());
269 	if(this.parent) vMatrix=GLGE.mulMat4(this.parent.getModelMatrix(),vMatrix);
270 	this.location=[vMatrix[3],vMatrix[7],vMatrix[11]];
271 	this.matrix=GLGE.inverseMat4(vMatrix);
272 273 };
274 /**
275 * Method generates the cameras view matrix
276 * @return Returns the view matrix based on this camera
277 * @type Matrix
278 */
279 GLGE.Camera.prototype.getViewMatrix=function(){
280 	if(!this.matrix || !this.rotmatrix) this.updateMatrix();
281 	return this.matrix;
282 };
283 
284 /**
285 * Method generates the cameras view projection matrix
286 * @return Returns the view projection  matrix based on this camera
287 * @type Matrix
288 */
289 GLGE.Camera.prototype.getViewProjection=function(){
290 	var projectionMatrix=this.getProjectionMatrix();
291 	var viewMatrix=this.getViewMatrix();
292 	if(projectionMatrix!=this.vpProjectionMatrix || viewMatrix!=this.vpViewMatrix){
293 		this.cameraViewProjection=GLGE.mulMat4(projectionMatrix,viewMatrix);
294 		this.vpProjectionMatrix=projectionMatrix;
295 		this.vpViewMatrix=viewMatrix;
296 	}
297 	return this.cameraViewProjection;
298 };
