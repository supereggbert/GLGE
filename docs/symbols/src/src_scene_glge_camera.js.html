219 /**
220 * Method gets the current projection matrix of this camera
221 * @return {Matrix} Returns the camera projection matrix
222 */
223 GLGE.Camera.prototype.getProjectionMatrix=function(){
224 	if(!this.pMatrix){
225 		if(this.pMatrixOveride){
226 			this.pMatrix=this.pMatrixOveride;
227 		}else{
228 			switch(this.type){
229 				case GLGE.C_PERSPECTIVE:
230 					this.pMatrix=GLGE.makePerspective(this.fovy, this.aspect, this.near, this.far);
231 					break;
232 				case GLGE.C_ORTHO:
233 					this.pMatrix=GLGE.makeOrtho(-this.orthoscale*this.aspect,this.orthoscale*this.aspect,-this.orthoscale,this.orthoscale, this.near, this.far);
234 					break;
235 			}
236 		}
237 	}
238 	return this.pMatrix;
239 };
240 /**
241 * Method generates the projection matrix based on the 
242 * camera paramaters
243 * @param {Matrix} projection The new projection matrix
244 */
245 GLGE.Camera.prototype.setProjectionMatrix=function(projection){
246 	this.pMatrix=projection;
247 	return this;
248 };
249 /**
250 * Method sets a custom projection matrix
251 * @param {Matrix} projection The new projection matrix
252 */
253 GLGE.Camera.prototype.setCustomProjectionMatrix=function(projection){
254 	this.pMatrix=projection;
255 	this.pMatrixOveride=projection;
256 	return this;
257 };
258 /**
259 * Method generates the cameras view matrix
260 * @return Returns the view matrix based on this camera
261 * @type Matrix
262 */
263 GLGE.Camera.prototype.updateMatrix=function(){
264 	var position=this.getPosition();
265 	var vMatrix=GLGE.translateMatrix(position.x,position.y,position.z);
266 	vMatrix=GLGE.mulMat4(vMatrix,this.getRotMatrix());
267 	if(this.parent) vMatrix=GLGE.mulMat4(this.parent.getModelMatrix(),vMatrix);
268 	this.location=[vMatrix[3],vMatrix[7],vMatrix[11]];
269 	this.matrix=GLGE.inverseMat4(vMatrix);
270 };
271 /**
272 * Method generates the cameras view matrix
273 * @return Returns the view matrix based on this camera
274 * @type Matrix
275 */
276 GLGE.Camera.prototype.getViewMatrix=function(){
277 	if(!this.matrix || !this.rotmatrix) this.updateMatrix();
278 	return this.matrix;
279 };
280 
281 /**
282 * Method generates the cameras view projection matrix
283 * @return Returns the view projection  matrix based on this camera
284 * @type Matrix
285 */
286 GLGE.Camera.prototype.getViewProjection=function(){
287 	var projectionMatrix=this.getProjectionMatrix();
288 	var viewMatrix=this.getViewMatrix();
289 	if(projectionMatrix!=this.vpProjectionMatrix || viewMatrix!=this.vpViewMatrix){
290 		this.cameraViewProjection=GLGE.mulMat4(projectionMatrix,viewMatrix);
291 		this.vpProjectionMatrix=projectionMatrix;
292 		this.vpViewMatrix=viewMatrix;
293 	}
294 	return this.cameraViewProjection;
295 };
