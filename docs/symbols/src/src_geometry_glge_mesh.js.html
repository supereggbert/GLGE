 91 /**
 92 * Gets the bounding volume for the mesh
 93 * @returns {GLGE.BoundingVolume} 
 94 */
 95 GLGE.Mesh.prototype.getBoundingVolume=function(){
 96 	if(!this.positions) return new GLGE.BoundingVolume(0,0,0,0,0,0);
 97 	if(!this.boundingVolume){
 98 		var minX,maxX,minY,maxY,minZ,maxZ;
 99 		var positions=this.positions;
100 		for(var i=0;i<positions.length;i=i+3){
101 			if(i==0){
102 				minX=maxX=positions[i];
103 				minY=maxY=positions[i+1];
104 				minZ=maxZ=positions[i+2];
105 			}else{
106 				minX=Math.min(minX,positions[i]);
107 				maxX=Math.max(maxX,positions[i]);
108 				minY=Math.min(minY,positions[i+1]);
109 				maxY=Math.max(maxY,positions[i+1]);
110 				minZ=Math.min(minZ,positions[i+2]);
111 				maxZ=Math.max(maxZ,positions[i+2]);
112 			}
113 		}
114 		this.boundingVolume=new GLGE.BoundingVolume(minX,maxX,minY,maxY,minZ,maxZ);
115 	}
116 	return this.boundingVolume;
117 }

187 	}else{

327 /**
328 * gets a vert tangent
329 * @private
330 */
331 GLGE.Mesh.prototype.tangentFromUV=function(p1,p2,p3,uv1,uv2,uv3,n){
332 	var toUnitVec3=GLGE.toUnitVec3;
333 	var subVec3=GLGE.subVec3;
334 	var scaleVec3=GLGE.scaleVec3;
335 	var dotVec3=GLGE.dotVec3;
336 	var crossVec3=GLGE.crossVec3;
337 	
338 	uv21=[uv2[0]-uv1[0],uv2[1]-uv1[1]];
339 	uv31=[uv3[0]-uv1[0],uv3[1]-uv1[1]];
340 	
341 	p21=GLGE.subVec3(p2,p1);
342 	p31=GLGE.subVec3(p3,p1);
343 	var s=(uv21[0]*uv31[1]-uv31[0]*uv21[1]);
344 
345 	if(s!=0){
346 		s=1/s;
347 		var t=subVec3(scaleVec3(p21,uv31[1]*s),scaleVec3(p31,uv21[1]*s));
348 		var b=subVec3(scaleVec3(p31,uv21[0]*s),scaleVec3(p21,uv31[0]*s));
349 	}else{
350 		t=[0,0,0];
351 		b=[0,0,0];
352 	}
353 	if(GLGE.dotVec3(GLGE.crossVec3(p21,p31),n)>0){
354 		t=scaleVec3(t,-1);
355 		b=scaleVec3(b,-1);
356 	}
357 	return [t,b];
358 }

495 /**
496 * Calculates the normals for this mesh
497 * @private
498 */
499 GLGE.Mesh.prototype.calcNormals=function(){
500 	for(var n=0;n<this.framePositions.length;n++){
501 		var normals=[];
502 		var positions=this.framePositions[n];
503 		var faces=this.faces.data;
504 		if(!faces){
505 			faces=[];
506 			for(var i=0;i<positions.length/3;i++) faces[i]=i;
507 		}
508 		for(var i=0;i<faces.length;i=i+3){
509 			var v1=[positions[faces[i]*3],positions[faces[i]*3+1],positions[faces[i]*3+2]];
510 			var v2=[positions[faces[i+1]*3],positions[faces[i+1]*3+1],positions[faces[i+1]*3+2]];
511 			var v3=[positions[faces[i+2]*3],positions[faces[i+2]*3+1],positions[faces[i+2]*3+2]];
512 			var vec1=GLGE.subVec3(v2,v1);
513 			var vec2=GLGE.subVec3(v3,v1);
514 			var norm=GLGE.toUnitVec3(GLGE.crossVec3(vec1,vec2));
515 			if(normals[faces[i]]==undefined) normals[faces[i]]=[];
516 			normals[faces[i]].push(norm);
517 			if(normals[faces[i+1]]==undefined) normals[faces[i+1]]=[];
518 			normals[faces[i+1]].push(norm);
519 			if(normals[faces[i+2]]==undefined) normals[faces[i+2]]=[];
520 			normals[faces[i+2]].push(norm);
521 		}
522 		var norms=[];
523 		for(i=0;i<normals.length;i++){
524 			var x=0,y=0,z=0;
525 			if(normals[i]!=undefined){
526 				for(var j=0;j<normals[i].length;j++){
527 					x+=normals[i][j][0];
528 					y+=normals[i][j][1];
529 					z+=normals[i][j][2];
530 				}
531 				x/=normals[i].length;
532 				y/=normals[i].length;
533 				z/=normals[i].length;
534 				norms[i*3]=x;
535 				norms[i*3+1]=y;
536 				norms[i*3+2]=z;
537 			}
538 		}
539 		this.setNormals(norms,n);
540 	}

547 	
548 	//calculate ambient color based on vertex angles
549 	var verts=this.positions;
550 	var faces=this.faces.data;
551 	var normals=this.normals;
552 	
553 	var idx=[];
554 	var len=verts.length/3
555 	for(var i=0;i<len;i++){
556 		idx.push([]);

562 		idx[faces[i+1]].push(faces[i+2]);
563 		idx[faces[i+2]].push(faces[i]);
564 		idx[faces[i+2]].push(faces[i+1]);
565 	}
566 	var ao=[];
567 	for(var i=0;i<len;i++){
568 		var AOfactor=0;
569 		var normal=[normals[i*3],normals[i*3+1],normals[i*3+2]];
570 		for(var j=0;j<idx[i].length;j++){
571 			var f=idx[i][j];
572 			var v=[verts[f*3]-verts[i*3],verts[f*3+1]-verts[i*3+1],verts[f*3+2]-verts[i*3+2]];
573 			v=GLGE.toUnitVec3(v);
574 			AOfactor+=v[0]*normal[0]+v[1]*normal[1]+v[2]*normal[2];
575 		}
576 		AOfactor/=idx[i].length;
577 		AOfactor=1.0-(AOfactor+1)*0.5;
578 		ao.push(AOfactor);
579 		ao.push(AOfactor);
580 		ao.push(AOfactor);
581 		ao.push(1);
582 	}
583 	this.setVertexColors(ao);
584 }
585 /**
586 * optimize geometry
587 * @private
588 */
589 GLGE.Mesh.prototype.optimize=function(){
590 	var verts=this.positions;
591 	var normals=this.normals;
592 	var faces=this.faces.data;
593 	var tangents=this.tangents;
594 	var uv1=this.uv1set;
595 	var uv2=this.uv2set;
596 	//expand out the faces
597 	var vertsTemp=[];
598 	var normalsTemp=[];
599 	var uv1Temp=[];
600 	var uv2Temp=[];
601 	var tangentsTemp=[];
602 	if(faces){
603 		for(var i=0;i<faces.length;i++){
604 			vertsTemp.push(verts[faces[i]*3]);
605 			vertsTemp.push(verts[faces[i]*3+1]);
606 			vertsTemp.push(verts[faces[i]*3+2]);
607 			normalsTemp.push(normals[faces[i]*3]);
608 			normalsTemp.push(normals[faces[i]*3+1]);
609 			normalsTemp.push(normals[faces[i]*3+2]);
610 			if(tangents && tangents.length>0){
611 				tangentsTemp.push(tangents[faces[i]*3]);
612 				tangentsTemp.push(tangents[faces[i]*3+1]);
613 				tangentsTemp.push(tangents[faces[i]*3+2]);
614 			}
615 			if(uv1){
616 				uv1Temp.push(uv1[faces[i]*2]);
617 				uv1Temp.push(uv1[faces[i]*2+1]);
618 			}
619 			if(uv2){
620 				uv2Temp.push(uv2[faces[i]*2]);
621 				uv2Temp.push(uv2[faces[i]*2+1]);
622 			}
623 		}
624 	}else{
625 		vertsTemp=verts;
626 		normalsTemp=normals;
627 		tangentsTemp=tangents;
628 		uv1Temp=uv1;
629 		uv2Temp=uv2;
630 	}
631 
632 	var newVerts=[];
633 	var newNormals=[];
634 	var newFaces=[];
635 	var newUV1s=[];
636 	var newUV2s=[];
637 	var newTangents=[];
638 	var stack=[];
639 	
640 	for(var i=0;i<vertsTemp.length;i=i+3){
641 		if(uv1 && uv2){
642 			var idx=[vertsTemp[i],vertsTemp[i+1],vertsTemp[i+2],normalsTemp[i],normalsTemp[i+1],normalsTemp[i+2],uv1Temp[i/3*2],uv1Temp[i/3*2+1]].join(" ");
643 		}else if(uv1){
644 			var idx=[vertsTemp[i],vertsTemp[i+1],vertsTemp[i+2],normalsTemp[i],normalsTemp[i+1],normalsTemp[i+2],uv1Temp[i/3*2],uv1Temp[i/3*2+1]].join(" ");
645 		}else{
646 			var idx=[vertsTemp[i],vertsTemp[i+1],vertsTemp[i+2],normalsTemp[i],normalsTemp[i+1],normalsTemp[i+2]].join(" ");
647 		}
648 		var vertIdx=stack.indexOf(idx);
649 		if(vertIdx<0){
650 			stack.push(idx);
651 			vertIdx=stack.length-1;
652 			newVerts.push(vertsTemp[i]);
653 			newVerts.push(vertsTemp[i+1]);
654 			newVerts.push(vertsTemp[i+2]);
655 			newNormals.push(normalsTemp[i]);
656 			newNormals.push(normalsTemp[i+1]);
657 			newNormals.push(normalsTemp[i+2]);
658 			if(tangents && tangents.length>0){
659 				newTangents.push(tangentsTemp[i]);
660 				newTangents.push(tangentsTemp[i+1]);
661 				newTangents.push(tangentsTemp[i+2]);
662 			}
663 			if(uv1){
664 				newUV1s.push(uv1Temp[i/3*2]);
665 				newUV1s.push(uv1Temp[i/3*2+1]);
666 			}
667 			if(uv2){
668 				newUV2s.push(uv2Temp[i/3*2]);
669 				newUV2s.push(uv2Temp[i/3*2+1]);
670 			}
671 		}
672 		newFaces.push(vertIdx);
673 	}
674 	this.setPositions(newVerts).setNormals(newNormals).setFaces(newFaces).setUV(newUV1s).setUV2(newUV2s).setTangents(newTangents);
675 }
