 62 	var tags=this.getElementsByTagName("*");
 63 	for(var i=0; i<tags.length;i++){
 64 		if(tags[i].getAttribute("id")==id){

 67 		}
 68 	}

 71 /**
 72 * Gets the absolute path given an import path and the path it's relative to
 73 * @param {string} path the path to get the absolute path for
 74 * @param {string} relativeto the path the supplied path is relativeto
 75 * @returns {string} absolute path
 76 * @private
 77 */
 78 GLGE.Document.prototype.getAbsolutePath=function(path,relativeto){
 79 	if(path.substr(0,7)=="http://" || path.substr(0,7)=="file://"  || path.substr(0,8)=="https://"){
 80 		return path;
 81 	}
 82 	else
 83 	{
 84 		if(!relativeto){
 85 			relativeto=window.location.href;
 86 		}
 87 		//find the path compoents
 88 		var bits=relativeto.split("/");
 89 		var domain=bits[2];
 90 		var proto=bits[0];
 91 		var initpath=[];
 92 		for(var i=3;i<bits.length-1;i++){
 93 			initpath.push(bits[i]);
 94 		}
 95 		//relative to domain
 96 		if(path.substr(0,1)=="/"){
 97 			initpath=[];
 98 		}
 99 		var locpath=path.split("/");
100 		for(var i=0;i<locpath.length;i++){
101 			if(locpath[i]=="..") initpath.pop();
102 				else if(locpath[i]!="") initpath.push(locpath[i]);
103 		}
104 		return proto+"//"+domain+"/"+initpath.join("/");
105 	}
106 }
107 /**
108 * Loads the root document
109 * @param {string} url URL of the resource to load
110 * @param {object} preload Decides if a preloader is used. true: default preloader, object: specialized preloader
111 */
112 GLGE.Document.prototype.load=function(url, preload){
113 	
114 	if(preload)
115 		this.usePreloader(preload);
116 	
117 	this.documents=[];
118 	this.rootURL=url;
119 	this.loadDocument(url,null);
120 }
121 /**
122 * Loads an additional documents into the collection
123 * @param {string} url URL of the resource to load
124 * @param {string} relativeto the path the URL is relative to, null for default
125 */
126 GLGE.Document.prototype.loadDocument=function(url,relativeto){
127 	this.loadCount++;
128 	url=this.getAbsolutePath(url,relativeto);
129 	
130 	if(this.preloader)
131 	{
132 		this.preloader.loadXMLFile(url);
133 	}
134 	else
135 	{
136 		var req = new XMLHttpRequest();
137 		if(req) {
138 			req.docurl=url;
139 			req.docObj=this;
140 			req.overrideMimeType("text/xml");
141 			req.onreadystatechange = function() {
142 				if(this.readyState  == 4)
143 				{
144 					if(this.status  == 200 || this.status==0){
145 						this.responseXML.getElementById=this.docObj.getElementById;
146 						this.docObj.loaded(this.docurl,this.responseXML);
147 					}else{ 
148 						GLGE.error("Error loading Document: "+this.docurl+" status "+this.status);
149 					}
150 				}
151 			};
152 			req.open("GET", url, true);
153 			req.send("");
154 		}
155 	}	
156 }
157 /**
158 * Trigered when a document has finished loading
159 * @param {string} url the absolute url of the document that has loaded
160 * @param {XMLDoc} responceXML the xml document that has finished loading
161 * @private
162 */
163 GLGE.Document.prototype.loaded=function(url,responceXML){
164 	this.loadCount--;
165 	this.documents[url]={'xml':responceXML, 'url':url};
166 	var root=responceXML.getElementsByTagName("glge");
167 	if(root[0] && root[0].hasAttribute("version")) this.version=parseFloat(root[0].getAttribute("version"));
168 	var imports=responceXML.getElementsByTagName("import");
169 	for(var i=0; i<imports.length;i++){
170 		if(!this.documents[this.getAbsolutePath(imports[i].getAttribute("url"),url)]){
171 			this.documents[this.getAbsolutePath(imports[i].getAttribute("url"),url)]={};
172 			this.loadDocument(imports[i].getAttribute("url"),url);
173 		}
174 	}
175 	if(this.loadCount==0){
176 		this.finishedLoading();
177 	}
178 }

186 	}
187 	this["onLoad"]();
188 }

194 /**
195 * Use a preloader
196 * @param {object} [object]	This object contains optional parameters. Example1: {XMLQuota: 0.30, XMLBytes: 852605}, Example2: {XMLQuota: 0.13, numXMLFiles: 1}, Example3: true
197 */
198 GLGE.Document.prototype.usePreloader = function(args){
199 	this.preloader = new GLGE.DocumentPreloader(this, args);
200 	var that = this;
201 	this.addLoadListener(function(url){that.preloadImages.call(that);});	
202 }
203 /**
204 * Start preloading images. This function should be called when the document (xml) is loaded.
205 * @private
206 */
207 GLGE.Document.prototype.preloadImages = function(){
208 	var imageArrays = []; // 2 dimensional
209 	var docUrls = [];
210 	
211 	// create an array of all images
212 	for(var doc in this.documents){
213 		if(this.documents[doc].xml){
214 			imageArrays.push(this.documents[doc].xml.getElementsByTagName("texture"));
215 			docUrls.push(this.documents[doc].url);
216 		}
217 	}
218 	
219 	// add images to the preloader
220 	for(var a in imageArrays){
221 		for(var i=0; i<imageArrays[a].length; i++){
222 			var src = imageArrays[a][i].getAttribute("src");
223 			if(src)
224 				this.preloader.addImage(this.getAbsolutePath(src, docUrls[a]));
225 		}
226 	}
227 	this.preloader.loadImages();
228 }
229 /**
230 * Converts and attribute name into a class name
231 * @param {string} name attribute name to convert
232 * @private
233 */
234 GLGE.Document.prototype.classString=function(name){
235 	if(!name) return false;
236 	var names=name.split("_");
237 	var converted="";
238 	for(var i=0;i<names.length;i++){
239 		converted=converted+names[i][0].toUpperCase()+names[i].substr(1);
240 	}
241 	return converted;
242 }
243 /**
244 * Sets the properties of an object based on the attributes of the corresponding dom element
245 * @param {object} Obj the DOM element to apply the attributes of
246 * @private
247 */
248 GLGE.Document.prototype.setProperties=function(Obj){
249 	var set_method;
250 	var attribute_name;
251 	var value;
252 	for(var i=0; i<Obj.attributes.length; i++){
253 		value=false;
254 		set_method="set"+this.classString(Obj.attributes[i].nodeName);
255 
256 		if(Obj.attributes[i].value[0]=="#"){
257 			value=this.getElement(Obj.attributes[i].value.substr(1),true);
258 		}
259 		if(!value){
260 			//if this is a GLGE contsant then set the constant value otherwise just literal
261 			if(typeof(GLGE[Obj.attributes[i].value]) != "undefined"){
262 				value=GLGE[Obj.attributes[i].value];
263 			}
264 			else
265 			{
266 				value=Obj.attributes[i].value;
267 			}
268 		}
269 		
270 		if(Obj.object[set_method]) Obj.object[set_method](value);
271 		//if a uid is set in the xml doc then make sure it's registered correctly in the assets
272 		if(Obj.attributes[i].nodeName=="uid"){
273 			GLGE.Assets.unregisterAsset(Obj.object.uid);
274 			Obj.object.uid=Obj.attributes[i].value;
275 			GLGE.Assets.registerAsset(Obj.object,Obj.attributes[i].value);
276 		}
277 	}
278 }
279 /**
280 * Adds child objects 
281 * @param {object} Obj the DOM element to apply the children of
282 * @private
283 */
284 GLGE.Document.prototype.addChildren=function(Obj){
285 	//loop though and add the children
286 	var add_method;
287 	var child=Obj.firstChild;
288 	while(child){
289 		add_method="add"+this.classString(child.tagName);
290 		if(Obj.object[add_method]) Obj.object[add_method](this.getElement(child));
291 		child=child.nextSibling;
292 	}
293 }
294 /**
295 * Gets an object from the XML document based on the dom element 
296 * @param {string|domelement} ele the id of the element to get or the dom node
297 */
298 GLGE.Document.prototype.getElement=function(ele,noerrors){
299 	var docele,doc;
300 	if(typeof(ele)=="string"){
301 		for(var doc in this.documents){
302 			if(this.documents[doc].xml){
303 				docele=this.documents[doc].xml.getElementById(ele);
304 				if(docele){
305 					ele=docele;
306 					break;
307 				}
308 			}
309 		}
310 	}
311 	if(typeof(ele)=="string"){
312 		//if element is still a string at this point there there is an issue
313 		if(!noerrors) GLGE.error("Element "+ele+" not found in document");
314 		return false;
315 	}
316 	else
317 	{
318 		if(this["get"+this.classString(ele.tagName)]){
319 			return this["get"+this.classString(ele.tagName)](ele);
320 		}
321 		else
322 		{
323 			return this.getDefault(ele);
324 		}
325 	}
326 }
327 /**
328 * Parses the a data array
329 * @param {domelement} ele the element to create the objects from
330 * @private
331 */
332 GLGE.Document.prototype.getData=function(ele){
333 	if(!ele.object){
334 		ele.object=this.parseArray(ele);
335 		if(ele.hasAttribute("type")){
336 			var type=ele.getAttribute("type");
337 			switch(type){
338 				case "matrix":
339 					for(var i=0;i<ele.object.length;i++){
340 						ele.object[i]=GLGE.Mat4(ele.object[i].split(" "));
341 					}
342 					break;
343 				case "links":
344 					for(var i=0;i<ele.object.length;i++){
345 						ele.object[i]=this.getElement(ele.object[i].substr(1));
346 					}
347 					break;
348 			}
349 		}
350 	}
351 	return ele.object;
352 }
353 /**
354 * Parses the dom element and creates any objects that are required
355 * @param {domelement} ele the element to create the objects from
356 * @private
357 */
358 GLGE.Document.prototype.getDefault=function(ele){
359 	if(!ele.object){
360 		if(GLGE[this.classString(ele.tagName)]){
361 			ele.object=new GLGE[this.classString(ele.tagName)]();
362 			this.setProperties(ele);
363 			this.addChildren(ele);
364 		}
365 		else
366 		{
367 			GLGE.error("XML Parse Error: GLGE Object not found"); 
368 		}
369 	}
370 	return ele.object;
371 }

389 /**
390 * Parses a document node into an array
391 * @param {node} the node to parse
392 * @private
393 */
394 GLGE.Document.prototype.parseArray=function(node){
395 	var child=node.firstChild;
396 	var prev="";
397 	var output=[];
398 	var currentArray;
399 	var i;
400 	while(child){
401 		currentArray=(prev+child.nodeValue).split(",");
402 		child=child.nextSibling;
403 		if(currentArray[0]=="") currentArray.unshift();
404 		if(child) prev=currentArray.pop();
405 		for(var i=0;i<currentArray.length;i++) output.push(currentArray[i]);
406 	}
407 	return output;
408 }

410 /**
411 * Parses the mesh dom to create the mesh object
412 * @param {domelement} ele the element to create the mesh from
413 * @private
414 */
415 GLGE.Document.prototype.getMesh=function(ele){
416 	if(this.version>0) return this.getDefault(ele); //as of GLGE XML 1.0 the mesh is nothing special!
417 	
418 	if(!ele.object){
419 		ele.object=new GLGE.Mesh();
420 		this.setProperties(ele);
421 		var child=ele.firstChild;
422 		while(child){
423 			switch(child.tagName){
424 				case "positions":
425 					ele.object.setPositions(this.parseArray(child));
426 					break;
427 				case "normals":
428 					ele.object.setNormals(this.parseArray(child));
429 					break;				
430 				case "uv1":
431 					ele.object.setUV(this.parseArray(child));
432 					break;				
433 				case "uv2":
434 					ele.object.setUV2(this.parseArray(child));
435 					break;
436 				case "faces":
437 					ele.object.setFaces(this.parseArray(child));
438 					break;
439 				case "color":
440 					ele.object.setVertexColors(this.parseArray(child));
441 					break;
442 				case "joint_names":
443 					var names=this.parseArray(child);
444 					var jointObjects=[];
445 					for(var i=0;i<names.length;i++){
446 						if(names[i].substr(0,1)=="#"){
447 							jointObjects.push(this.getElement(names[i].substr(1)));
448 						}else{
449 							jointObjects.push(names[i]);
450 						}
451 					}
452 					ele.object.setJoints(jointObjects);
453 					break;
454 				case "bind_matrix":
455 					var mats=this.parseArray(child);
456 					var invBind=[];
457 					for(var i=0;i<mats.length;i++){
458 						invBind.push(GLGE.Mat4(mats[i].split(" ")));
459 					}
460 					ele.object.setInvBindMatrix(invBind);
461 					break;
462 				case "joints":
463 					ele.object.setVertexJoints(this.parseArray(child),child.getAttribute("count"));
464 					break;
465 				case "weights":
466 					ele.object.setVertexWeights(this.parseArray(child),child.getAttribute("count"));
467 					break;
468 			}
469 			child=child.nextSibling;
470 		}
471 	}
472 	return ele.object;
473 }

492 /**
493 * loads xml from a script tag
494 * @param {string} id the id of the element to load
495 */
496 GLGE.Document.prototype.parseScript=function(id){
497 	this.rootURL=window.location.toString();
498 	var xmlScript = document.getElementById(id);
499 	if (!xmlScript) {
500 		return null;
501 	}
502  
503 	var str = "";
504 	var k = xmlScript.firstChild;
505 	while (k) {
506 		if (k.nodeType == 3) {
507 			str += k.textContent;
508 		}
509 		k = k.nextSibling;
510 	}
511 	
512 	var parser=new DOMParser();
513 	var xmlDoc=parser.parseFromString(str,"text/xml");
514 	xmlDoc.getElementById=this.getElementById;
515 	
516 	this.documents["#"+id]={xml:xmlDoc};
517 
518 	var imports=xmlDoc.getElementsByTagName("import");
519 	for(var i=0; i<imports.length;i++){
520 		if(!this.documents[this.getAbsolutePath(imports[i].getAttribute("url"),url)]){
521 			this.documents[this.getAbsolutePath(imports[i].getAttribute("url"),url)]={};
522 			this.loadDocument(imports[i].getAttribute("url"));
523 		}
524 	}
525 	if(this.loadCount==0){
526 		this.finishedLoading();
527 	}
528 }
