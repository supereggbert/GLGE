 62 	var tags=this.getElementsByTagName("*");
 63 	for(var i=0; i<tags.length;i++){
 64 		if(tags[i].getAttribute("id")==id){

 67 		}
 68 	}

 71 /**
 72 * Gets the absolute path given an import path and the path it's relative to
 73 * @param {string} path the path to get the absolute path for
 74 * @param {string} relativeto the path the supplied path is relativeto
 75 * @returns {string} absolute path
 76 * @private
 77 */
 78 GLGE.Document.prototype.getAbsolutePath=function(path,relativeto){
 79 	if(path.substr(0,7)=="http://" || path.substr(0,7)=="file://"  || path.substr(0,8)=="https://"){
 80 		return path;
 81 	}
 82 	else
 83 	{
 84 		if(!relativeto){
 85 			relativeto=window.location.href;
 86 		}
 87 		//find the path compoents
 88 		var bits=relativeto.split("/");
 89 		var domain=bits[2];
 90 		var proto=bits[0];
 91 		var initpath=[];
 92 		for(var i=3;i<bits.length-1;i++){
 93 			initpath.push(bits[i]);
 94 		}
 95 		//relative to domain
 96 		if(path.substr(0,1)=="/"){
 97 			initpath=[];
 98 		}
 99 		var locpath=path.split("/");
100 		for(var i=0;i<locpath.length;i++){
101 			if(locpath[i]=="..") initpath.pop();
102 				else if(locpath[i]!="") initpath.push(locpath[i]);
103 		}
104 		return proto+"//"+domain+"/"+initpath.join("/");
105 	}
106 }
107 /**
108 * Loads the root document
109 * @param {string} url URL of the resource to load
110 * @param {object} preload Decides if a preloader is used. true: default preloader, object: specialized preloader
111 */
112 GLGE.Document.prototype.load=function(url, preload){
113 	
114 	if(preload)
115 		this.usePreloader(preload);
116 	
117 	this.documents=[];
118 	this.rootURL=url;
119 	this.loadDocument(url,null);
120 }
121 /**
122 * Loads an additional documents into the collection
123 * @param {string} url URL of the resource to load
124 * @param {string} relativeto the path the URL is relative to, null for default
125 */
126 GLGE.Document.prototype.loadDocument=function(url,relativeto){
127 	this.loadCount++;
128 	url=this.getAbsolutePath(url,relativeto);
129 	
130 	if(this.preloader)
131 	{
132 		this.preloader.loadXMLFile(url);
133 	}
134 	else
135 	{
136 		var req = new XMLHttpRequest();
137 		if(req) {
138 			req.docurl=url;
139 			req.docObj=this;
140 			req.overrideMimeType("text/xml");
141 			req.onreadystatechange = function() {
142 				if(this.readyState  == 4)
143 				{
144 					if(this.status  == 200 || this.status==0){
145 						this.responseXML.getElementById=this.docObj.getElementById;
146 						this.docObj.loaded(this.docurl,this.responseXML);
147 					}else{ 
148 						GLGE.error("Error loading Document: "+this.docurl+" status "+this.status);
149 					}
150 				}
151 			};
152 			req.open("GET", url, true);
153 			req.send("");
154 		}
155 	}	
156 }
157 /**
158 * Trigered when a document has finished loading
159 * @param {string} url the absolute url of the document that has loaded
160 * @param {XMLDoc} responceXML the xml document that has finished loading
161 * @private
162 */
163 GLGE.Document.prototype.loaded=function(url,responceXML){
164 	this.loadCount--;
165 	this.documents[url]={'xml':responceXML, 'url':url};
166 	var root=responceXML.getElementsByTagName("glge");
167 	if(root[0] && root[0].hasAttribute("version")) this.version=parseFloat(root[0].getAttribute("version"));
168 	var imports=responceXML.getElementsByTagName("import");
169 	for(var i=0; i<imports.length;i++){
170 		if(!this.documents[this.getAbsolutePath(imports[i].getAttribute("url"),url)]){
171 			this.documents[this.getAbsolutePath(imports[i].getAttribute("url"),url)]={};
172 			this.loadDocument(imports[i].getAttribute("url"),url);
173 		}
174 	}
175 	if(this.loadCount==0){
176 		this.finishedLoading();
177 	}
178 }

187 	}
188 	this["onLoad"]();
189 }

195 /**
196 * Use a preloader
197 * @param {object} [object]	This object contains optional parameters. Example1: {XMLQuota: 0.30, XMLBytes: 852605}, Example2: {XMLQuota: 0.13, numXMLFiles: 1}, Example3: true
198 */
199 GLGE.Document.prototype.usePreloader = function(args){
200 	this.preloader = new GLGE.DocumentPreloader(this, args);
201 	var that = this;
202 	this.addLoadListener(function(url){that.preloadImages.call(that);});	
203 }
204 /**
205 * Start preloading images. This function should be called when the document (xml) is loaded.
206 * @private
207 */
208 GLGE.Document.prototype.preloadImages = function(){
209 	var imageArrays = []; // 2 dimensional
210 	var docUrls = [];
211 	
212 	// create an array of all images
213 	for(var doc in this.documents){
214 		if(this.documents[doc].xml){
215 			imageArrays.push(this.documents[doc].xml.getElementsByTagName("texture"));
216 			docUrls.push(this.documents[doc].url);
217 		}
218 	}
219 	
220 	// add images to the preloader
221 	for(var a in imageArrays){
222 		for(var i=0; i<imageArrays[a].length; i++){
223 			var src = imageArrays[a][i].getAttribute("src");
224 			if(src)
225 				this.preloader.addImage(this.getAbsolutePath(src, docUrls[a]));
226 		}
227 	}
228 	this.preloader.loadImages();
229 }
230 /**
231 * Converts and attribute name into a class name
232 * @param {string} name attribute name to convert
233 * @private
234 */
235 GLGE.Document.prototype.classString=function(name){
236 	if(!name) return false;
237 	var names=name.split("_");
238 	var converted="";
239 	for(var i=0;i<names.length;i++){
240 		converted=converted+names[i][0].toUpperCase()+names[i].substr(1);
241 	}
242 	return converted;
243 }
244 /**
245 * Sets the properties of an object based on the attributes of the corresponding dom element
246 * @param {object} Obj the DOM element to apply the attributes of
247 * @private
248 */
249 GLGE.Document.prototype.setProperties=function(Obj){
250 	var set_method;
251 	var attribute_name;
252 	var value;
253 	for(var i=0; i<Obj.attributes.length; i++){
254 		value=false;
255 		set_method="set"+this.classString(Obj.attributes[i].nodeName);
256 
257 		if(Obj.attributes[i].value[0]=="#"){
258 			value=this.getElement(Obj.attributes[i].value.substr(1),true);
259 		}
260 		if(!value){
261 			//if this is a GLGE contsant then set the constant value otherwise just literal
262 			if(typeof(GLGE[Obj.attributes[i].value]) != "undefined"){
263 				value=GLGE[Obj.attributes[i].value];
264 			}
265 			else
266 			{
267 				value=Obj.attributes[i].value;
268 			}
269 		}
270 		
271 		if(Obj.object[set_method]) Obj.object[set_method]((value == parseFloat(value)) ? (parseFloat(value)) : (value));
272 		//if a uid is set in the xml doc then make sure it's registered correctly in the assets
273 		if(Obj.attributes[i].nodeName=="uid"){
274 			GLGE.Assets.unregisterAsset(Obj.object.uid);
275 			Obj.object.uid=Obj.attributes[i].value;
276 			GLGE.Assets.registerAsset(Obj.object,Obj.attributes[i].value);
277 		}
278 	}
279 }
280 /**
281 * Adds child objects 
282 * @param {object} Obj the DOM element to apply the children of
283 * @private
284 */
285 GLGE.Document.prototype.addChildren=function(Obj){
286 	//loop though and add the children
287 	var add_method;
288 	var child=Obj.firstChild;
289 	while(child){
290 		add_method="add"+this.classString(child.tagName);
291 		if(Obj.object[add_method]) Obj.object[add_method](this.getElement(child));
292 		child=child.nextSibling;
293 	}
294 }
295 /**
296 * Gets an object from the XML document based on the dom element 
297 * @param {string|domelement} ele the id of the element to get or the dom node
298 */
299 GLGE.Document.prototype.getElement=function(ele,noerrors){
300 	var docele,doc;
301 	if(typeof(ele)=="string"){
302 		for(var doc in this.documents){
303 			if(this.documents[doc].xml){
304 				docele=this.documents[doc].xml.getElementById(ele);
305 				if(docele){
306 					ele=docele;
307 					break;
308 				}
309 			}
310 		}
311 	}
312 	if(typeof(ele)=="string"){
313 		//if element is still a string at this point there there is an issue
314 		if(!noerrors) GLGE.error("Element "+ele+" not found in document");
315 		return false;
316 	}
317 	else
318 	{
319 		if(this["get"+this.classString(ele.tagName)]){
320 			return this["get"+this.classString(ele.tagName)](ele);
321 		}
322 		else
323 		{
324 			return this.getDefault(ele);
325 		}
326 	}
327 }
328 /**
329 * Parses the a data array
330 * @param {domelement} ele the element to create the objects from
331 * @private
332 */
333 GLGE.Document.prototype.getData=function(ele){
334 	if(!ele.object){
335 		ele.object=this.parseArray(ele);
336 		if(ele.hasAttribute("type")){
337 			var type=ele.getAttribute("type");
338 			switch(type){
339 				case "matrix":
340 					for(var i=0;i<ele.object.length;i++){
341 						ele.object[i]=GLGE.Mat4(ele.object[i].split(" "));
342 					}
343 					break;
344 				case "links":
345 					for(var i=0;i<ele.object.length;i++){
346 						ele.object[i]=this.getElement(ele.object[i].substr(1));
347 					}
348 					break;
349 			}
350 		}
351 	}
352 	return ele.object;
353 }
354 /**
355 * Parses the dom element and creates any objects that are required
356 * @param {domelement} ele the element to create the objects from
357 * @private
358 */
359 GLGE.Document.prototype.getDefault=function(ele){
360 	if(!ele.object){
361 		if(GLGE[this.classString(ele.tagName)]){
362 			ele.object=new GLGE[this.classString(ele.tagName)]();
363 			this.setProperties(ele);
364 			this.addChildren(ele);
365 		}
366 		else
367 		{
368 			GLGE.error("XML Parse Error: GLGE Object not found"); 
369 		}
370 	}
371 	return ele.object;
372 }

390 /**
391 * Parses a document node into an array
392 * @param {node} the node to parse
393 * @private
394 */
395 GLGE.Document.prototype.parseArray=function(node){
396 	var child=node.firstChild;
397 	var prev="";
398 	var output=[];
399 	var currentArray;
400 	var i;
401 	while(child){
402 		currentArray=(prev+child.nodeValue).split(",");
403 		child=child.nextSibling;
404 		if(currentArray[0]=="") currentArray.unshift();
405 		if(child) prev=currentArray.pop();
406 		for(var i=0;i<currentArray.length;i++) output.push(currentArray[i]);
407 	}
408 	return output;
409 }

411 /**
412 * Parses the mesh dom to create the mesh object
413 * @param {domelement} ele the element to create the mesh from
414 * @private
415 */
416 GLGE.Document.prototype.getMesh=function(ele){
417 	if(this.version>0) return this.getDefault(ele); //as of GLGE XML 1.0 the mesh is nothing special!
418 	
419 	if(!ele.object){
420 		ele.object=new GLGE.Mesh();
421 		this.setProperties(ele);
422 		var child=ele.firstChild;
423 		while(child){
424 			switch(child.tagName){
425 				case "positions":
426 					ele.object.setPositions(this.parseArray(child));
427 					break;
428 				case "normals":
429 					ele.object.setNormals(this.parseArray(child));
430 					break;				
431 				case "uv1":
432 					ele.object.setUV(this.parseArray(child));
433 					break;				
434 				case "uv2":
435 					ele.object.setUV2(this.parseArray(child));
436 					break;
437 				case "faces":
438 					ele.object.setFaces(this.parseArray(child));
439 					break;
440 				case "color":
441 					ele.object.setVertexColors(this.parseArray(child));
442 					break;
443 				case "joint_names":
444 					var names=this.parseArray(child);
445 					var jointObjects=[];
446 					for(var i=0;i<names.length;i++){
447 						if(names[i].substr(0,1)=="#"){
448 							jointObjects.push(this.getElement(names[i].substr(1)));
449 						}else{
450 							jointObjects.push(names[i]);
451 						}
452 					}
453 					ele.object.setJoints(jointObjects);
454 					break;
455 				case "bind_matrix":
456 					var mats=this.parseArray(child);
457 					var invBind=[];
458 					for(var i=0;i<mats.length;i++){
459 						invBind.push(GLGE.Mat4(mats[i].split(" ")));
460 					}
461 					ele.object.setInvBindMatrix(invBind);
462 					break;
463 				case "joints":
464 					ele.object.setVertexJoints(this.parseArray(child),child.getAttribute("count"));
465 					break;
466 				case "weights":
467 					ele.object.setVertexWeights(this.parseArray(child),child.getAttribute("count"));
468 					break;
469 			}
470 			child=child.nextSibling;
471 		}
472 	}
473 	return ele.object;
474 }

493 /**
494 * loads xml from a script tag
495 * @param {string} id the id of the element to load
496 */
497 GLGE.Document.prototype.parseScript=function(id){
498 	this.rootURL=window.location.toString();
499 	var xmlScript = document.getElementById(id);
500 	if (!xmlScript) {
501 		return null;
502 	}
503  
504 	var str = "";
505 	var k = xmlScript.firstChild;
506 	while (k) {
507 		if (k.nodeType == 3) {
508 			str += k.textContent;
509 		}
510 		k = k.nextSibling;
511 	}
512 	
513 	var parser=new DOMParser();
514 	var xmlDoc=parser.parseFromString(str,"text/xml");
515 	xmlDoc.getElementById=this.getElementById;
516 	
517 	this.documents["#"+id]={xml:xmlDoc};
518 
519 	var imports=xmlDoc.getElementsByTagName("import");
520 	for(var i=0; i<imports.length;i++){
521 		if(!this.documents[this.getAbsolutePath(imports[i].getAttribute("url"),url)]){
522 			this.documents[this.getAbsolutePath(imports[i].getAttribute("url"),url)]={};
523 			this.loadDocument(imports[i].getAttribute("url"));
524 		}
525 	}
526 	if(this.loadCount==0){
527 		this.finishedLoading();
528 	}
