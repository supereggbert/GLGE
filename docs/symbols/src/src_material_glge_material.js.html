616 GLGE.Material.prototype.addMaterialLayer=function(layer){
617   if(typeof layer=="string")  layer=GLGE.Assets.get(layer);
618   this.layers.push(layer);
619   var material=this;
620   var listener=function(event){
621     material.fireEvent("shaderupdate",{});

624   layer.addEventListener("shaderupdate",listener);
625   this.fireEvent("shaderupdate",{});

634   var idx=this.layers.indexOf(layer);

651 /**
652 * Generate the code required to calculate the texture coords for each layer
653 * @private
654 */
655 GLGE.Material.prototype.getLayerCoords=function(shaderInjection){
656     var shader=[];
657     shader.push("vec4 texturePos;\n");
658     for(var i=0; i<this.layers.length;i++){
659       shader.push("textureCoords"+i+"=vec3(0.0,0.0,0.0);\n");
660 
661       if(this.layers[i].mapinput==GLGE.UV1 || this.layers[i].mapinput==GLGE.UV2){
662         shader.push("texturePos=vec4(vec2(UVCoord["+(this.layers[i].mapinput*2)+"],(1.0-UVCoord["+(this.layers[i].mapinput*2+1)+"])),1.0,1.0);\n");
663       }
664 
665       if(this.layers[i].mapinput==GLGE.MAP_NORM){
666         shader.push("texturePos=vec4(normalize(n.xyz),1.0);\n");
667       }
668       if(this.layers[i].mapinput==GLGE.MAP_OBJ){
669         shader.push("texturePos=vec4(normalize(OBJCoord.xyz),1.0);\n");
670       }
671 
672       if(this.layers[i].mapinput==GLGE.MAP_REF){
673         //will need to do in fragment to take the normal maps into account!
674         shader.push("texturePos=vec4(reflect(normalize(eyevec.xyz),normalize(n.xyz)),1.0);\n");
675       }
676 
677 
678 
679       if(this.layers[i].mapinput==GLGE.MAP_ENV){
680         //will need to do in fragment to take the normal maps into account!
681         shader.push("texturePos=envMat * vec4(reflect(normalize(eyevec.xyz),normalize(n.xyz)),1.0);\n");
682       }
683 
684       shader.push("textureCoords"+i+"=(layer"+i+"Matrix * texturePos).xyz;\n");
685 
686       if(shaderInjection && ~shaderInjection.indexOf("GLGE_Texcoord")){
687         shader.push("textureCoords"+i+"=GLGE_Texcoord("+i+",textureCoords"+i+");\n");
688       }
689 
690     }
691 
692     return shader.join("");
693 }

713 /**
714 * Generate the fragment shader program for this material
715 * @private
716 */
717 GLGE.Material.prototype.getFragmentShader=function(lights,colors,shaderInjection){
718 	var shader="#ifdef GL_ES\nprecision mediump float;\n#endif\n#define GLGE_FRAGMENT\n";
719 	if(shaderInjection) shader+=shaderInjection;
720 	var tangent=false;
721 	for(var i=0; i<lights.length;i++){
722 		if(lights[i].type==GLGE.L_POINT || lights[i].type==GLGE.L_SPOT || lights[i].type==GLGE.L_DIR){
723 			shader=shader+"varying vec3 lightvec"+i+";\n"; 
724 			shader=shader+"varying float lightdist"+i+";\n";  
725 		}
726 	}
727 	shader=shader+"varying vec3 n;\n";  
728 	shader=shader+"varying vec3 t;\n";  
729 	shader=shader+"varying vec4 UVCoord;\n";
730 	shader=shader+"varying vec3 eyevec;\n"; 
731 	shader=shader+"varying vec3 OBJCoord;\n";
732 	if(colors) shader=shader+"varying vec4 vcolor;\n";
733 
734 	
735 
736 	//texture uniforms
737 	for(var i=0; i<this.textures.length;i++){
738 		if(this.textures[i].className=="Texture") shader=shader+"uniform sampler2D TEXTURE"+i+";\n";
739 		if(this.textures[i].className=="TextureCanvas") shader=shader+"uniform sampler2D TEXTURE"+i+";\n";
740 		if(this.textures[i].className=="TextureCanvasCube") shader=shader+"uniform samplerCube TEXTURE"+i+";\n";
741 		if(this.textures[i].className=="TextureVideo") shader=shader+"uniform sampler2D TEXTURE"+i+";\n";
742 		if(this.textures[i].className=="TextureCube") shader=shader+"uniform samplerCube TEXTURE"+i+";\n";
743 	}
744 	
745 	
746 	var cnt=1;
747 	var shadowlights=[];
748 	var num;
749 	for(var i=0; i<lights.length;i++){
750 	    if(lights[i].type==GLGE.L_OFF) continue;
751 			shader=shader+"uniform vec3 lightcolor"+i+";\n";  
752 			shader=shader+"uniform vec3 lightAttenuation"+i+";\n";  
753 			shader=shader+"uniform float spotCosCutOff"+i+";\n";  
754 			shader=shader+"uniform float spotExp"+i+";\n";  
755 			shader=shader+"uniform vec3 lightdir"+i+";\n";  
756 			shader=shader+"uniform mat4 lightmat"+i+";\n";
757 			shader=shader+"uniform float shadowbias"+i+";\n"; 
758 			shader=shader+"uniform int shadowsamples"+i+";\n";  
759 			shader=shader+"uniform float shadowsoftness"+i+";\n";  
760 			shader=shader+"uniform bool castshadows"+i+";\n";
