 40 /**
 41 * @class Class to describe and action on a skeleton
 42 * @param {string} uid a unique reference string for this object
 43 * @augments GLGE.QuickNotation
 44 * @augments GLGE.JSONLoader
 45 */
 46 GLGE.Action=function(uid){
 47 	this.channels=[];
 48 	GLGE.Assets.registerAsset(this,uid);
 49 };
 50 GLGE.augment(GLGE.QuickNotation,GLGE.Action);
 51 GLGE.augment(GLGE.JSONLoader,GLGE.Action);
 52 /**
 53  * @name Action#animFinished
 54  * @event
 55  * @param {object} data
 56  */
 57 GLGE.augment(GLGE.Events,GLGE.Action);
 58 
 59 /**
 60 * Starts playing the action
 61 */
 62 GLGE.Action.prototype.start=function(blendTime,loop,names){
 63 	if(!loop) loop=false;
 64 	if(!blendTime) blendTime=0;
 65 	var channels=this.channels;
 66 	var start=(new Date()).getTime();
 67 	this.animFinished=false;
 68 	
 69 	for(var i=0;i<channels.length;i++){
 70 		var animation=channels[i].getAnimation();
 71 		var action=this;
 72 		var channel=channels[i];
 73 		var target=channel.getTarget();
 74 		if(typeof target=="string"){
 75 			if(names && names[target]){
 76 				target=names[target];
 77 			}
 78 		}
 79 		var closure={};
 80 		closure.finishEvent=function(data){
 81 			target.removeEventListener("animFinished",closure.finishEvent);
 82 			if(!action.animFinished && target.animation==animation){
 83 				action.fireEvent("animFinished",{});
 84 				action.animFinished=true;
 85 			}
 86 		}
 87 		target.addEventListener("animFinished",closure.finishEvent);
 88 		
 89 		target.setAnimation(animation,blendTime,start);
 90 		target.setLoop(loop);
 91 
 92 	}
 93 };
 94 /**
 95 * Sets the start frame for all animations
 96 * @param {number} startFrame the starting frame for the animation
 97 */
 98 GLGE.Action.prototype.setStartFrame=function(startFrame){
 99 	for(var i=0;i<this.channels.length;i++){
100 		this.channels[i].getAnimation().setStartFrame(startFrame);
101 	}
102 	return this;
103 };
104 /**
105 * Sets the number of frames to play
106 * @param {number} frame the number of frames to play
107 */
108 GLGE.Action.prototype.setFrames=function(frames){
109 	for(var i=0;i<this.channels.length;i++){
110 		this.channels[i].getAnimation().setFrames(frames);
111 	}
112 	return this;
113 };
114 
115 
116 /**
117 * Adds and action channel to this action
118 * @param {GLGE.ActionChannel} channel the channel to be added
119 */
120 GLGE.Action.prototype.addActionChannel=function(channel){
121 	this.channels.push(channel);
122 	return this;
123 };
124 /**
125 * Removes and action channel to this action
126 * @param {GLGE.ActionChannel} channel the channel to be removed
127 */
128 GLGE.Action.prototype.removeActionChannel=function(channel){
129 	for(var i=0;i<this.channels.length;i++){
130 		if(this.channels[i]==channels){
131 			this.channels.splice(i,1);
132 			break;
133 		}
134 	}
135 };
