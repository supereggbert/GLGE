 40 /**
 41 * @name GLGE.Texture#downloadComplete

 48 /**
 49 * @class A texture to be included in a material
 50 * @param {string} uid the unique id for this texture
 51 * @see GLGE.Material
 52 * @augments GLGE.QuickNotation
 53 * @augments GLGE.JSONLoader
 54 * @augments GLGE.Events
 55 */
 56 GLGE.Texture=function(uid){
 57 	GLGE.Assets.registerAsset(this,uid);
 58 }
 59 GLGE.augment(GLGE.QuickNotation,GLGE.Texture);
 60 GLGE.augment(GLGE.JSONLoader,GLGE.Texture);
 61 GLGE.augment(GLGE.Events,GLGE.Texture);
 62 GLGE.Texture.prototype.className="Texture";
 63 GLGE.Texture.prototype.image=null;
 64 GLGE.Texture.prototype.glTexture=null;
 65 GLGE.Texture.prototype.url=null;
 66 GLGE.Texture.prototype.state=0;
 67 /**
 68 * Gets the textures used by the layer
 69 * @return {string} The textures image url
 70 */
 71 GLGE.Texture.prototype.getSrc=function(){
 72 	return this.url;
 73 };
 74 
 75 /**
 76 * Sets the textures image location
 77 * @param {string} url the texture image url
 78 */
 79 GLGE.Texture.prototype.setSrc=function(url){
 80 	this.url=url;
 81 	this.state=0;
 82 	this.image=new Image();
 83 	var texture=this;
 84 	this.image.onload = function(){
 85 		texture.state=1;
 86     	texture.fireEvent("downloadComplete");
 87 	}	
 88 	this.image.src=url;	
 89 	if(this.glTexture && this.gl){
 90 		this.glTexture=null;
 91 	}
 92 	return this;
 93 };
 94 
 95 /**
 96 * Sets the textures image location
 97 * @private
 98 **/
 99 GLGE.Texture.prototype.doTexture=function(gl){
100 	this.gl=gl;
101 	if(!gl.urlTextures) gl.urlTextures={};
102 	if(gl.urlTextures[this.url]){
103 		this.glTexture=gl.urlTextures[this.url];
104 		this.state=2;
105 	}
106 	//create the texture if it's not already created
107 	if(!this.image) this.setSrc(this.url);
108 	if(!this.glTexture) this.glTexture=gl.createTexture();
109 	//if the image is loaded then set in the texture data
110 	if(this.state==1){
111 		gl.bindTexture(gl.TEXTURE_2D, this.glTexture);
112 		//START... FRANCISCO REIS: to accept Non Power of Two Images
113 		var w = Math.pow( 2, Math.round( Math.log( this.image.width ) / Math.log( 2 ) ) );
114 		var h = Math.pow( 2, Math.round( Math.log( this.image.height ) / Math.log( 2 ) ) );
115 
116 		var imageOrCanvas;
117 		if(w == this.image.width && h == this.image.height)
118 			imageOrCanvas = this.image;
119 		else
120 		{
121 			imageOrCanvas = document.createElement("canvas");
122 			imageOrCanvas.width=w;
123 			imageOrCanvas.height=h;
124 			var context = imageOrCanvas.getContext("2d");
125 			context.drawImage(this.image,0,0,w,h);
126 		}
127 
128 		gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE,imageOrCanvas);//this line was replaced from ",this.image)" to ",imageOrCanvas)"
129 		//...END FRANCISCO REIS: to accept Non Power of Two Images
130 		gl.urlTextures[this.url]=this.glTexture;
131 		gl.generateMipmap(gl.TEXTURE_2D);
132 		gl.bindTexture(gl.TEXTURE_2D, null);
133 		this.state=2;
134 	}
135 	gl.bindTexture(gl.TEXTURE_2D, this.glTexture);
136 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
137 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR_MIPMAP_LINEAR);
138 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.REPEAT);
139 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.REPEAT);
140 	
141 	if(this.state==2) return true;
142 		else return false;
143 }

146 /**
147 * Determin if the image resource has been downloaded
148 **/
149 GLGE.Texture.prototype.isComplete=function(){
150     return this.state>0;
151 }
