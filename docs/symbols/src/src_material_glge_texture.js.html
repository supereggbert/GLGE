 48 /**
 49 * @class A texture to be included in a material
 50 * @param {string} uid the unique id for this texture
 51 * @see GLGE.Material
 52 * @augments GLGE.QuickNotation
 53 * @augments GLGE.JSONLoader
 54 * @augments GLGE.Events
 55 */
 56 GLGE.Texture=function(uid){
 57 	GLGE.Assets.registerAsset(this,uid);
 58 }
 59 GLGE.augment(GLGE.QuickNotation,GLGE.Texture);
 60 GLGE.augment(GLGE.JSONLoader,GLGE.Texture);
 61 GLGE.augment(GLGE.Events,GLGE.Texture);
 62 GLGE.Texture.prototype.className="Texture";
 63 GLGE.Texture.prototype.image=null;
 64 GLGE.Texture.prototype.glTexture=null;
 65 GLGE.Texture.prototype.url=null;
 66 GLGE.Texture.prototype.state=0;
 67 GLGE.Texture.prototype.preAlpha=true;

 94 /**
 95 * Sets the textures image location
 96 * @param {string} url the texture image url
 97 */
 98 GLGE.Texture.prototype.setSrc=function(url){
 99 	this.url=url;
100 	this.state=0;
101 	this.image=new Image();
102 	var texture=this;
103 	this.image.onload = function(){
104 		texture.state=1;
105     	texture.fireEvent("downloadComplete");
106 107 	}	
108 	this.image.src=url;	
109 	if(this.glTexture && this.gl){
110 		this.glTexture=null;
111 	}
112 	return this;
113 };
114 
115 /**
116 * Sets the textures image location
117 * @private
118 **/
119 GLGE.Texture.prototype.doTexture=function(gl){
120 	this.gl=gl;
121 	if(!gl.urlTextures) gl.urlTextures={};
122 	if(gl.urlTextures[this.url]){
123 		this.glTexture=gl.urlTextures[this.url];
124 		this.state=2;
125 	}
126 	//create the texture if it's not already created
127 	if(!this.image) this.setSrc(this.url);
128 	if(!this.glTexture) this.glTexture=gl.createTexture();
129 	//if the image is loaded then set in the texture data
130 	if(this.state==1){
131 		gl.bindTexture(gl.TEXTURE_2D, this.glTexture);
132 		//START... FRANCISCO REIS: to accept Non Power of Two Images
133 		var w = Math.pow( 2, Math.round( Math.log( this.image.width ) / Math.log( 2 ) ) );
134 		var h = Math.pow( 2, Math.round( Math.log( this.image.height ) / Math.log( 2 ) ) );
135 
136 		var imageOrCanvas;
137 		if(w == this.image.width && h == this.image.height)
138 			imageOrCanvas = this.image;
139 		else
140 		{
141 			imageOrCanvas = document.createElement("canvas");
142 			imageOrCanvas.width=w;
143 			imageOrCanvas.height=h;
144 			var context = imageOrCanvas.getContext("2d");
145 			context.drawImage(this.image,0,0,w,h);
146 		}
147 
148 		gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE,imageOrCanvas);//this line was replaced from ",this.image)" to ",imageOrCanvas)"
149 		//...END FRANCISCO REIS: to accept Non Power of Two Images
150 		gl.urlTextures[this.url]=this.glTexture;
151 		gl.generateMipmap(gl.TEXTURE_2D);
152 		gl.bindTexture(gl.TEXTURE_2D, null);
153 		this.state=2;
154 	}
155 	gl.bindTexture(gl.TEXTURE_2D, this.glTexture);
156 	if(this.preAlpha){
157 		gl.pixelStorei(gl.UNPACK_PREMULTIPLY_ALPHA_WEBGL, 1);
158 	}
159 	
160 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
161 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR_MIPMAP_LINEAR);
162 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.REPEAT);
163 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.REPEAT);
164 	
165 	if(this.state==2) return true;
166 		else return false;
167 }

172 **/
173 GLGE.Texture.prototype.isComplete=function(){
