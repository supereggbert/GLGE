209 /**
210 * Creates the shader program for the object
211 * @private
212 */
213 GLGE.Text.prototype.GLGenerateShader=function(gl){
214 	if(this.GLShaderProgram) gl.deleteProgram(this.GLShaderProgram);
215 
216 	//Vertex Shader
217 	var vertexStr="";
218 	vertexStr+="attribute vec3 position;\n";
219 	vertexStr+="attribute vec2 uvcoord;\n";
220 	vertexStr+="varying vec2 texcoord;\n";
221 	vertexStr+="uniform mat4 Matrix;\n";
222 	vertexStr+="uniform mat4 PMatrix;\n";
223 	vertexStr+="varying vec4 pos;\n";
224 	
225 	vertexStr+="void main(void){\n";
226 	vertexStr+="texcoord=uvcoord;\n";    
227 	vertexStr+="pos = Matrix * vec4(position,1.0);\n";
228 	vertexStr+="gl_Position = PMatrix * pos;\n";
229 	vertexStr+="}\n";
230 	
231 	//Fragment Shader
232 	var fragStr="#ifdef GL_ES\nprecision highp float;\n#endif\n";
233 	fragStr=fragStr+"uniform sampler2D TEXTURE;\n";
234 	fragStr=fragStr+"varying vec2 texcoord;\n";
235 	fragStr=fragStr+"uniform mat4 Matrix;\n";
236 	fragStr=fragStr+"varying vec4 pos;\n";
237 	fragStr=fragStr+"uniform float far;\n";
238 	fragStr=fragStr+"uniform bool depthrender;\n";
239 	fragStr=fragStr+"uniform float distance;\n";
240 	fragStr=fragStr+"uniform int picktype;\n";
241 	fragStr=fragStr+"uniform vec3 pickcolor;\n";
242 	fragStr=fragStr+"uniform vec3 color;\n";
243 	fragStr=fragStr+"uniform float alpha;\n";
244 	fragStr=fragStr+"void main(void){\n";
245 	fragStr=fragStr+"float ob=pow(min(1.0,abs(dot(normalize(Matrix[2].rgb),vec3(0.0,0.0,1.0)))*2.0),1.5);\n";
246 	fragStr=fragStr+"float a=texture2D(TEXTURE,texcoord).a*alpha*ob;\n";
247 	fragStr=fragStr+"if(picktype=="+GLGE.TEXT_BOXPICK+"){gl_FragColor = vec4(pickcolor,1.0);}"
248 	fragStr=fragStr+"else if(picktype=="+GLGE.TEXT_TEXTPICK+"){if(alpha<1.0) discard; gl_FragColor = vec4(pickcolor,alpha);}"
249 	fragStr=fragStr+"else{gl_FragColor = vec4(color.rgb,a);};\n";
250 	fragStr=fragStr+"if (depthrender) { if(a<0.5) discard; float depth = gl_FragCoord.z / gl_FragCoord.w;\n";
251 	fragStr=fragStr+"vec4 rgba=fract(depth/distance * vec4(16777216.0, 65536.0, 256.0, 1.0));\n";
252 	fragStr=fragStr+"gl_FragColor=rgba-rgba.rrgb*vec4(0.0,0.00390625,0.00390625,0.00390625);}\n";
253 	fragStr=fragStr+"}\n";
254 	
255 	this.GLFragmentShader=gl.createShader(gl.FRAGMENT_SHADER);
256 	this.GLVertexShader=gl.createShader(gl.VERTEX_SHADER);
257 
258 
259 	gl.shaderSource(this.GLFragmentShader, fragStr);
260 	gl.compileShader(this.GLFragmentShader);
261 	if (!gl.getShaderParameter(this.GLFragmentShader, gl.COMPILE_STATUS)) {
262 	      GLGE.error(gl.getShaderInfoLog(this.GLFragmentShader));
263 	      return;
264 	}
265 	
266 	//set and compile the vertex shader
267 	//need to set str
268 	gl.shaderSource(this.GLVertexShader, vertexStr);
269 	gl.compileShader(this.GLVertexShader);
270 	if (!gl.getShaderParameter(this.GLVertexShader, gl.COMPILE_STATUS)) {
271 		GLGE.error(gl.getShaderInfoLog(this.GLVertexShader));
272 		return;
273 	}
274 	
275 	this.GLShaderProgram = gl.createProgram();
276 	gl.attachShader(this.GLShaderProgram, this.GLVertexShader);
277 	gl.attachShader(this.GLShaderProgram, this.GLFragmentShader);
278 	gl.linkProgram(this.GLShaderProgram);	
279 }

292 /**
293 * Updates the canvas texture
294 * @private
295 */
296 GLGE.Text.prototype.updateCanvas=function(gl){
297 	var canvas = this.canvas;
298 	canvas.width=1;
299 	canvas.height=this.size*1.2;
300 	var ctx = canvas.getContext("2d");
301 	ctx.font = this.size+"px "+this.font;
302 	canvas.width=ctx.measureText(this.text).width;
303 	canvas.height=this.size*1.2;
304 	 ctx = canvas.getContext("2d");
305 	ctx.textBaseline="top";
306 	ctx.font = (this.extra||"") + " " + this.size+"px "+this.font;
307 	this.aspect=canvas.width/canvas.height;
308 	ctx.fillText(this.text, 0, 0);   
309 	
310 	var height=Math.pow(2,Math.ceil(Math.log(canvas.height))/(Math.log(2)));
311 	var width=Math.pow(2,Math.ceil(Math.log(canvas.width))/(Math.log(2)));
312 
313 	this.scaleCanvas.height=height;
314 	this.scaleCanvas.width=width;
315 
316 	this.scaleContext=this.scaleCanvas.getContext("2d");
317 	this.scaleContext.clearRect(0,0,width,height);
318 	this.scaleContext.drawImage(canvas, 0, 0, width, height);
319 	
320 	gl.bindTexture(gl.TEXTURE_2D, this.glTexture);
321 	//TODO: fix this when minefield is upto spec
322 	try{gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, this.scaleCanvas);}
323 	catch(e){gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, this.scaleCanvas,null);}
324 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
325 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
326 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR_MIPMAP_LINEAR);
327 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
328 	gl.generateMipmap(gl.TEXTURE_2D);
329 	
330 	gl.bindTexture(gl.TEXTURE_2D, null);
331 	this.dirty=false;
332 }

334 /**
335 * Renders the text to the render buffer
336 * @private
337 */
338 GLGE.Text.prototype.GLRender=function(gl,renderType,pickindex){
339 	if(!this.gl){
340 		this.GLInit(gl);
341 	}	
342 	if(this.dirty) this.updateCanvas(gl);
343 	if(renderType==GLGE.RENDER_DEFAULT || renderType==GLGE.RENDER_PICK || renderType==GLGE.RENDER_SHADOW){
344 		//if look at is set then look
345 		if(this.lookAt) this.Lookat(this.lookAt);
346 		
347 		if(gl.program!=this.GLShaderProgram){
348 			gl.useProgram(this.GLShaderProgram);
349 			gl.program=this.GLShaderProgram;
350 		}
351 					
352 		var attribslot;
353 		//disable all the attribute initially arrays - do I really need this?
354 		for(var i=0; i<8; i++) gl.disableVertexAttribArray(i);
355 		attribslot=GLGE.getAttribLocation(gl,this.GLShaderProgram, "position");
356 
357 		gl.bindBuffer(gl.ARRAY_BUFFER, this.posBuffer);
358 		gl.enableVertexAttribArray(attribslot);
359 		gl.vertexAttribPointer(attribslot, this.posBuffer.itemSize, gl.FLOAT, false, 0, 0);
360 		
361 		attribslot=GLGE.getAttribLocation(gl,this.GLShaderProgram, "uvcoord");
362 		gl.bindBuffer(gl.ARRAY_BUFFER, this.uvBuffer);
363 		gl.enableVertexAttribArray(attribslot);
364 		gl.vertexAttribPointer(attribslot, this.uvBuffer.itemSize, gl.FLOAT, false, 0, 0);
365 		
366 		gl.activeTexture(gl["TEXTURE0"]);
367 		gl.bindTexture(gl.TEXTURE_2D, this.glTexture);
368 		GLGE.setUniform(gl,"1i",GLGE.getUniformLocation(gl,this.GLShaderProgram, "TEXTURE"),0);
369 		
370 		if(!pickindex) pickindex=0;
371 		var b = pickindex >> 16 & 0xFF; 
372 		var g = pickindex >> 8 & 0xFF; 
373 		var r = pickindex & 0xFF;
374 		GLGE.setUniform3(gl,"3f",GLGE.getUniformLocation(gl,this.GLShaderProgram, "pickcolor"),r/255,g/255,b/255);
375 		
376 		if(renderType==GLGE.RENDER_PICK){
377 			GLGE.setUniform(gl,"1i",GLGE.getUniformLocation(gl,this.GLShaderProgram, "picktype"), this.pickType);	
378 		}else{
379 			GLGE.setUniform(gl,"1i",GLGE.getUniformLocation(gl,this.GLShaderProgram, "picktype"), 0);	
380 		}
381 		var distance=gl.scene.camera.getFar();
382 		GLGE.setUniform(gl,"1f",GLGE.getUniformLocation(gl,this.GLShaderProgram, "distance"), distance);
383 		if(renderType==GLGE.RENDER_SHADOW){
384 			GLGE.setUniform(gl,"1i",GLGE.getUniformLocation(gl,this.GLShaderProgram, "depthrender"), 1);
385 		}else{
386 			GLGE.setUniform(gl,"1i",GLGE.getUniformLocation(gl,this.GLShaderProgram, "depthrender"), 0);
387 		}
388 		
389 		
390 		if(!this.GLShaderProgram.glarrays) this.GLShaderProgram.glarrays={};
391 
392 		
393 		//generate and set the modelView matrix
394 		var scalefactor=this.size/100;
395 		var mMatrix=GLGE.mulMat4(gl.scene.camera.getViewMatrix(),GLGE.mulMat4(this.getModelMatrix(),GLGE.scaleMatrix(this.aspect*scalefactor,scalefactor,scalefactor)));
396 		var mUniform = GLGE.getUniformLocation(gl,this.GLShaderProgram, "Matrix");
397 		if(!this.GLShaderProgram.glarrays.mMatrix) this.GLShaderProgram.glarrays.mMatrix=new Float32Array(mMatrix);
398 			else GLGE.mat4gl(mMatrix,this.GLShaderProgram.glarrays.mMatrix);
399 		GLGE.setUniformMatrix(gl,"Matrix4fv",mUniform, true, this.GLShaderProgram.glarrays.mMatrix);
400 		
401 		var mUniform = GLGE.getUniformLocation(gl,this.GLShaderProgram, "PMatrix");
402 
403 		if(!this.GLShaderProgram.glarrays.pMatrix) this.GLShaderProgram.glarrays.pMatrix=new Float32Array(gl.scene.camera.getProjectionMatrix());
404 			else GLGE.mat4gl(gl.scene.camera.getProjectionMatrix(),this.GLShaderProgram.glarrays.pMatrix);
405 		GLGE.setUniformMatrix(gl,"Matrix4fv",mUniform, true, this.GLShaderProgram.glarrays.pMatrix);
406 				
407 		var farUniform = GLGE.getUniformLocation(gl,this.GLShaderProgram, "far");
408 		GLGE.setUniform(gl,"1f",farUniform, gl.scene.camera.getFar());
409 			
410 		var alphaUniform = GLGE.getUniformLocation(gl,this.GLShaderProgram, "alpha");
411 		GLGE.setUniform(gl,"1f",alphaUniform, this.alpha);
412 		
413 		//set the color
414 		GLGE.setUniform3(gl,"3f",GLGE.getUniformLocation(gl,this.GLShaderProgram, "color"), this.color.r,this.color.g,this.color.b);
415 		
416 		gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.GLfaces);
417 		gl.drawElements(gl.TRIANGLES, this.GLfaces.numItems, gl.UNSIGNED_SHORT, 0);
418 		gl.scene.lastMaterial=null;
419 	}
420 }
421 /**
422 * creates the plane mesh to draw
423 * @private
424 */
425 GLGE.Text.prototype.createPlane=function(gl){
426 	//create the vertex positions
427 	if(!this.posBuffer) this.posBuffer = gl.createBuffer();
428 	gl.bindBuffer(gl.ARRAY_BUFFER, this.posBuffer);
429 	gl.bufferData(gl.ARRAY_BUFFER, new Float32Array([1,1,0,-1,1,0,-1,-1,0,1,-1,0]), gl.STATIC_DRAW);
430 	this.posBuffer.itemSize = 3;
431 	this.posBuffer.numItems = 4;
432 	//create the vertex uv coords
433 	if(!this.uvBuffer) this.uvBuffer = gl.createBuffer();
434 	gl.bindBuffer(gl.ARRAY_BUFFER, this.uvBuffer);
435 	gl.bufferData(gl.ARRAY_BUFFER, new Float32Array([0,0,1,0,1,1,0,1]), gl.STATIC_DRAW);
436 	this.uvBuffer.itemSize = 2;
437 	this.uvBuffer.numItems = 4;
438 	//create the faces
439 	if(!this.GLfaces) this.GLfaces = gl.createBuffer();
440 	gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.GLfaces);
441 	gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array([2,1,0,0,3,2]), gl.STATIC_DRAW);
442 	this.GLfaces.itemSize = 1;
443 	this.GLfaces.numItems = 6;
444 }
