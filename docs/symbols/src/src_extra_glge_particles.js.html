 37 /**
 38 * @class A texture to be included in a material
 39 * @param {string} uid the unique id for this texture
 40 * @augments GLGE.Placeable
 41 * @augments GLGE.Animatable
 42 */
 43 GLGE.ParticleSystem=function(uid){
 44 	this.startTime=(new Date()).getTime();
 45 	this.texture={};
 46 	this.startMaxVelocity={x:0,y:0,z:0};
 47 	this.startMinVelocity={x:0,y:0,z:0};
 48 	this.startMaxAcceleration={x:0,y:0,z:0};
 49 	this.endMaxAcceleration={x:0,y:0,z:0};
 50 	this.startMinAcceleration={x:0,y:0,z:0};
 51 	this.endMinAcceleration={x:0,y:0,z:0};
 52 	this.startColor={r:0,g:0,b:0,a:1};
 53 	this.endColor={r:0,g:0,b:0,a:1};
 54 	GLGE.Assets.registerAsset(this,uid);
 55 }

452 /**
453 * The particles velocity function used to generate the initial particles velocities
454 */
455 GLGE.ParticleSystem.prototype.velocityFunction=function(i){
456 	return [
457 		(this.startMaxVelocity.x-this.startMinVelocity.x) * Math.random()+this.startMinVelocity.x,
458 		(this.startMaxVelocity.y-this.startMinVelocity.y) * Math.random()+this.startMinVelocity.y,
459 		(this.startMaxVelocity.z-this.startMinVelocity.z) * Math.random()+this.startMinVelocity.z
460 		];
461 }
462 /**
463 * The particles acceleration function used to generate the initial particles accelerations
464 */
465 GLGE.ParticleSystem.prototype.accelerationFunction=function(i){
466 	return [[
467 		(this.startMaxAcceleration.x-this.startMinAcceleration.x) * Math.random()+this.startMinAcceleration.x,
468 		(this.startMaxAcceleration.y-this.startMinAcceleration.y) * Math.random()+this.startMinAcceleration.y,
469 		(this.startMaxAcceleration.z-this.startMinAcceleration.z) * Math.random()+this.startMinAcceleration.z,
470 		],
471 		[
472 		(this.endMaxAcceleration.x-this.endMinAcceleration.x) * Math.random()+this.endMinAcceleration.x,
473 		(this.endMaxAcceleration.y-this.endMinAcceleration.y) * Math.random()+this.endMinAcceleration.y,
474 		(this.endMaxAcceleration.z-this.endMinAcceleration.z) * Math.random()+this.endMinAcceleration.z,
475 		]];
476 }

555 /**
556 * generates the particles
557 * @private
558 */
559 GLGE.ParticleSystem.prototype.generateParticles=function(gl){
560 	var num_particles=this.numParticles;
561 	this.attribute={initPos:[],initAcc:[],endAcc:[],initVel:[],initColor:[],endColor:[],sizeAndOffset:[]};
562 	this.faces=[];
563 	for(var i=0; i<num_particles;i++){
564 		var position=this.positionFunction(i);
565 		var velocity=this.velocityFunction(i);
566 		var acceleration=this.accelerationFunction(i);
567 		var color=this.colorFunction(i);
568 		var size=this.sizeFunction(i);
569 		var lifetime=this.lifeTimeFunction(i);
570 		var offsetTime=Math.floor(Math.random()*lifetime);
571 		for(var y=-1;y<=1;y=y+2){
572 			for(var x=-1;x<=1;x=x+2){
573 				this.attribute.initPos.push(parseFloat(position[0])+x,parseFloat(position[1])+y,parseFloat(position[2]));
574 				this.attribute.initAcc.push(acceleration[0][0],acceleration[0][1],acceleration[0][2]);
575 				this.attribute.endAcc.push(acceleration[1][0],acceleration[1][1],acceleration[1][2]);
576 				this.attribute.initVel.push(velocity[0],velocity[1],velocity[2]);
577 				this.attribute.initColor.push(color[0][0],color[0][1],color[0][2],color[0][3]);
578 				this.attribute.endColor.push(color[1][0],color[1][1],color[1][2],color[1][3]);
579 				this.attribute.sizeAndOffset.push(size[0],size[1],offsetTime,lifetime);
580 			}
581 		}
582 	}
583 	
584 	//create the face buffer
585 	for(var i=0; i<num_particles;i=i+4){
586 		this.faces.push(0+i,1+i,2+i);
587 		this.faces.push(1+i,2+i,3+i);
588 	}
589 	this.facesGL=gl.createBuffer();
590 	gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.facesGL);
591 	gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(this.faces), gl.STATIC_DRAW);
592 	this.facesGL.num=this.faces.length;
593 
594 	this.attribute.initPosGL=this.createBuffer(gl,this.attribute.initPos);
595 	this.attribute.initAccGL=this.createBuffer(gl,this.attribute.initAcc);
596 	this.attribute.endAccGL=this.createBuffer(gl,this.attribute.endAcc);
597 	this.attribute.initVelGL=this.createBuffer(gl,this.attribute.initVel);
598 	this.attribute.initColorGL=this.createBuffer(gl,this.attribute.initColor);
599 	this.attribute.endColorGL=this.createBuffer(gl,this.attribute.endColor);
600 	this.attribute.sizeAndOffsetGL=this.createBuffer(gl,this.attribute.sizeAndOffset);
601 }

621 	var vtxShader=[

632 	"uniform bool loop;",
633 	"uniform mat4 mvMatrix;",
634 	"uniform mat4 pMatrix;",
635 	//varying
636 	"varying vec2 UV;",
637 	"varying vec4 color;",
638 	//main
639 	"void main(){",
640 	
641 	"UV = (position.xy+1.0)/2.0;",
642 	"if((time>sizeTimeLife[2] && (time-sizeTimeLife[2])<sizeTimeLife[3]) || loop){",
643 	"float localTime = mod((time - sizeTimeLife[2]), sizeTimeLife[3]);",
644 	"color = (endColor-initColor)/sizeTimeLife[3]*localTime+initColor;",
645 	"float size = (sizeTimeLife[1]-sizeTimeLife[0])/sizeTimeLife[3]*localTime+sizeTimeLife[0];",
646 	"vec3 pos = (endAcc-initAcc)*(localTime*log(localTime)-localTime)+0.5*initAcc*localTime*localTime+initVel*localTime;",
647 	"pos = (mvMatrix*vec4(pos,1.0)).xyz;",
648 	"vec3 positions = pos+(position*size);",
649 	"gl_Position = pMatrix*vec4(positions,1.0);",
650 	"}else{",
651 	"gl_Position = vec4(0.0,0.0,0.0,1.0);",
652 	"}",
653 	"}"
654 	].join("");
655 	frgShader=[
656 	"#ifdef GL_ES\nprecision mediump float;\n#endif\n",
657 	//uniforms
658 	"uniform sampler2D texture;",
659 	//varying
660 	"varying vec2 UV;",
661 	"varying vec4 color;",
662 	//main
663 	"void main(){",
664 	"gl_FragColor=texture2D(texture,UV)*color;",
665 	"}"
666 	].join("");
667 
668 
669 	var vertexShader=gl.createShader(gl.VERTEX_SHADER);
670 	var fragmentShader=gl.createShader(gl.FRAGMENT_SHADER);
671 
672 	gl.shaderSource(vertexShader, vtxShader);
673 	gl.compileShader(vertexShader);
674 	if (!gl.getShaderParameter(vertexShader, gl.COMPILE_STATUS)) {
675 	      GLGE.error(gl.getShaderInfoLog(vertexShader));
676 	      return;
677 	}
678 
679 	gl.shaderSource(fragmentShader,frgShader);
680 	gl.compileShader(fragmentShader);
681 	if (!gl.getShaderParameter(fragmentShader, gl.COMPILE_STATUS)) {
682 	      GLGE.error(gl.getShaderInfoLog(fragmentShader));
683 	      return;
684 	}
685 
686 	if(this.program) gl.deleteProgram(this.Program);
687 	this.program = gl.createProgram();
688 	gl.attachShader(this.program, vertexShader);
689 	gl.attachShader(this.program, fragmentShader);
690 	gl.linkProgram(this.program);	
691 }
692 /**
693 * Creates the particle system buffers
694 * @private
695 */
696 GLGE.ParticleSystem.prototype.createBuffer=function(gl,array){
697 	var buffer = gl.createBuffer();
698 	gl.bindBuffer(gl.ARRAY_BUFFER, buffer);
699 	gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(array), gl.STATIC_DRAW);
700 	return buffer;

702 /**
703 * Sets the uniforms
704 * @private
705 */
706 GLGE.ParticleSystem.prototype.setUniforms=function(gl){
707 	var program=this.program;
708 	if(!program.glarrays) program.glarrays={};
709 	var cameraMatrix=gl.scene.camera.getViewMatrix();
710 	
711 	var pos=this.getPosition();
712 	//var camerapos=gl.scene.camera.getPosition();
713 	var mvMatrix=GLGE.mulMat4(cameraMatrix,[
714 		1,0,0,pos.x,
715 		0,1,0,pos.y,
716 		0,0,1,pos.z,
717 		0,0,0,1]);
718 
719 		
720 	var mvUniform = GLGE.getUniformLocation(gl,program, "mvMatrix");
721 
722 	if(!program.glarrays.mvMatrix) program.glarrays.mvMatrix=new Float32Array(mvMatrix);
723 		else GLGE.mat4gl(mvMatrix,program.glarrays.mvMatrix);
724 	gl.uniformMatrix4fv(mvUniform, true, program.glarrays.mvMatrix);
725 
726 	var pUniform = GLGE.getUniformLocation(gl,program, "pMatrix");
727 	if(!program.glarrays.pMatrix) program.glarrays.pMatrix=new Float32Array(gl.scene.camera.getProjectionMatrix());
728 			else GLGE.mat4gl(gl.scene.camera.getProjectionMatrix(),program.glarrays.pMatrix);	
729 	gl.uniformMatrix4fv(pUniform, true, program.glarrays.pMatrix);
730 
731 	gl.uniform1f(GLGE.getUniformLocation(gl,program, "time"), ((new Date()).getTime()-this.startTime));
732 	gl.uniform1i(GLGE.getUniformLocation(gl,program, "loop"), this.loop);
733 	
734 	
735 	gl.activeTexture(gl.TEXTURE0);
736 	//create the texture if it's not already created
737 	if(!this.glTexture) this.glTexture=gl.createTexture();
738 	//if the image is loaded then set in the texture data
739 	if(this.texture.state==1){
740 		gl.bindTexture(gl.TEXTURE_2D, this.glTexture);
741 		gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE,this.texture.image);
742 		gl.pixelStorei(gl.UNPACK_PREMULTIPLY_ALPHA_WEBGL, 1);
743 		gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
744 		gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR_MIPMAP_LINEAR);
745 		gl.generateMipmap(gl.TEXTURE_2D);
746 		gl.bindTexture(gl.TEXTURE_2D, null);
747 		this.texture.state=2;
748 		program.texture=true;
749 	}
750 	gl.bindTexture(gl.TEXTURE_2D, this.glTexture);
751 	if(program.texture){
752 		gl.uniform1i(GLGE.getUniformLocation(gl,program, "texture"), 0);
753 	}
754 
755 }

830 GLGE.ParticleSystem.prototype.GLRender=function(gl){
831 	if(!this.attribute) this.generateParticles(gl);
832 	if(!this.program) this.generateProgram(gl);
833 	
834 	gl.useProgram(this.program);
835 	this.setAttributes(gl);
836 	this.setUniforms(gl);
837 	gl.colorMask(0,0,0,0);
838 	gl.disable(gl.BLEND);
839 	gl.enable(gl.STENCIL_TEST);
840 	gl.stencilFunc(gl.ALWAYS, 1, 1);
841 	gl.stencilOp(gl.KEEP, gl.KEEP, gl.REPLACE);
842 	gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.facesGL);
843 	gl.drawElements(gl.TRIANGLES,this.facesGL.num, gl.UNSIGNED_SHORT, 0);
844 	gl.stencilFunc(gl.EQUAL, 1, 1);
845 	gl.stencilOp(gl.KEEP, gl.KEEP, gl.KEEP);
846 	gl.colorMask(1,1,1,1);
847 	gl.disable(gl.DEPTH_TEST);
848 	gl.enable(gl.BLEND);
849 	gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.facesGL);
850 	gl.drawElements(gl.TRIANGLES,this.facesGL.num, gl.UNSIGNED_SHORT, 0);
851 	gl.stencilOp(gl.REPLACE, gl.REPLACE, gl.REPLACE);
852 	gl.stencilFunc(gl.ALWAYS, 0, 0);
853 	gl.enable(gl.DEPTH_TEST);
854 	
855 	gl.scene.lastMaterial=null;
856 }
