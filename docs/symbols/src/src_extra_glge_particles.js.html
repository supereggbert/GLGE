 37 /**
 38 * @class A texture to be included in a material
 39 * @param {string} uid the unique id for this texture
 40 * @augments GLGE.Placeable
 41 * @augments GLGE.Animatable
 42 */
 43 GLGE.ParticleSystem=function(uid){
 44 	this.startTime=(new Date()).getTime();
 45 	this.texture={};
 46 	this.startMaxVelocity={x:0,y:0,z:0};
 47 	this.startMinVelocity={x:0,y:0,z:0};
 48 	this.startMaxAcceleration={x:0,y:0,z:0};
 49 	this.endMaxAcceleration={x:0,y:0,z:0};
 50 	this.startMinAcceleration={x:0,y:0,z:0};
 51 	this.endMinAcceleration={x:0,y:0,z:0};
 52 	this.startColor={r:0,g:0,b:0,a:1};
 53 	this.endColor={r:0,g:0,b:0,a:1};
 54 	GLGE.Assets.registerAsset(this,uid);
 55 }

454 /**
455 * The particles velocity function used to generate the initial particles velocities
456 */
457 GLGE.ParticleSystem.prototype.velocityFunction=function(i){
458 	return [
459 		(this.startMaxVelocity.x-this.startMinVelocity.x) * Math.random()+this.startMinVelocity.x,
460 		(this.startMaxVelocity.y-this.startMinVelocity.y) * Math.random()+this.startMinVelocity.y,
461 		(this.startMaxVelocity.z-this.startMinVelocity.z) * Math.random()+this.startMinVelocity.z
462 		];
463 }
464 /**
465 * The particles acceleration function used to generate the initial particles accelerations
466 */
467 GLGE.ParticleSystem.prototype.accelerationFunction=function(i){
468 	return [[
469 		(this.startMaxAcceleration.x-this.startMinAcceleration.x) * Math.random()+this.startMinAcceleration.x,
470 		(this.startMaxAcceleration.y-this.startMinAcceleration.y) * Math.random()+this.startMinAcceleration.y,
471 		(this.startMaxAcceleration.z-this.startMinAcceleration.z) * Math.random()+this.startMinAcceleration.z,
472 		],
473 		[
474 		(this.endMaxAcceleration.x-this.endMinAcceleration.x) * Math.random()+this.endMinAcceleration.x,
475 		(this.endMaxAcceleration.y-this.endMinAcceleration.y) * Math.random()+this.endMinAcceleration.y,
476 		(this.endMaxAcceleration.z-this.endMinAcceleration.z) * Math.random()+this.endMinAcceleration.z,
477 		]];
478 }

558 /**
559 * generates the particles
560 * @private
561 */
562 GLGE.ParticleSystem.prototype.generateParticles=function(gl){
563 	var num_particles=this.numParticles;
564 	this.attribute={initPos:[],initAcc:[],endAcc:[],initVel:[],initColor:[],endColor:[],sizeAndOffset:[]};
565 	this.faces=[];
566 	for(var i=0; i<num_particles;i++){
567 		var position=this.positionFunction(i);
568 		var velocity=this.velocityFunction(i);
569 		var acceleration=this.accelerationFunction(i);
570 		var color=this.colorFunction(i);
571 		var size=this.sizeFunction(i);
572 		var lifetime=this.lifeTimeFunction(i);
573 		var offsetTime=Math.floor(Math.random()*lifetime);
574 		for(var y=-1;y<=1;y=y+2){
575 			for(var x=-1;x<=1;x=x+2){
576 				this.attribute.initPos.push(parseFloat(position[0])+x,parseFloat(position[1])+y,parseFloat(position[2]));
577 				this.attribute.initAcc.push(acceleration[0][0],acceleration[0][1],acceleration[0][2]);
578 				this.attribute.endAcc.push(acceleration[1][0],acceleration[1][1],acceleration[1][2]);
579 				this.attribute.initVel.push(velocity[0],velocity[1],velocity[2]);
580 				this.attribute.initColor.push(color[0][0],color[0][1],color[0][2],color[0][3]);
581 				this.attribute.endColor.push(color[1][0],color[1][1],color[1][2],color[1][3]);
582 				this.attribute.sizeAndOffset.push(size[0],size[1],offsetTime,lifetime);
583 			}
584 		}
585 	}
586 	
587 	//create the face buffer
588 	for(var i=0; i<num_particles;i=i+4){
589 		this.faces.push(0+i,1+i,2+i);
590 		this.faces.push(1+i,2+i,3+i);
591 	}
592 	this.facesGL=gl.createBuffer();
593 	gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.facesGL);
594 	gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(this.faces), gl.STATIC_DRAW);
595 	this.facesGL.num=this.faces.length;
596 
597 	this.attribute.initPosGL=this.createBuffer(gl,this.attribute.initPos);
598 	this.attribute.initAccGL=this.createBuffer(gl,this.attribute.initAcc);
599 	this.attribute.endAccGL=this.createBuffer(gl,this.attribute.endAcc);
600 	this.attribute.initVelGL=this.createBuffer(gl,this.attribute.initVel);
601 	this.attribute.initColorGL=this.createBuffer(gl,this.attribute.initColor);
602 	this.attribute.endColorGL=this.createBuffer(gl,this.attribute.endColor);
603 	this.attribute.sizeAndOffsetGL=this.createBuffer(gl,this.attribute.sizeAndOffset);
604 }

624 	var vtxShader=[

635 	"uniform bool loop;",
636 	"uniform mat4 mvMatrix;",
637 	"uniform mat4 pMatrix;",
638 	//varying
639 	"varying vec2 UV;",
640 	"varying vec4 color;",
641 	//main
642 	"void main(){",
643 	
644 	"UV = (position.xy+1.0)/2.0;",
645 	"if((time>sizeTimeLife[2] && (time-sizeTimeLife[2])<sizeTimeLife[3]) || loop){",
646 	"float localTime = mod((time - sizeTimeLife[2]), sizeTimeLife[3]);",
647 	"color = (endColor-initColor)/sizeTimeLife[3]*localTime+initColor;",
648 	"float size = (sizeTimeLife[1]-sizeTimeLife[0])/sizeTimeLife[3]*localTime+sizeTimeLife[0];",
649 	"vec3 pos = (endAcc-initAcc)*(localTime*log(localTime)-localTime)+0.5*initAcc*localTime*localTime+initVel*localTime;",
650 	"pos = (mvMatrix*vec4(pos,1.0)).xyz;",
651 	"vec3 positions = pos+(position*size);",
652 	"gl_Position = pMatrix*vec4(positions,1.0);",
653 	"}else{",
654 	"gl_Position = vec4(0.0,0.0,0.0,1.0);",
655 	"}",
656 	"}"
657 	].join("");
658 	frgShader=[
659 	"#ifdef GL_ES\nprecision mediump float;\n#endif\n",
660 	//uniforms
661 	"uniform sampler2D texture;",
662 	//varying
663 	"varying vec2 UV;",
664 	"varying vec4 color;",
665 	//main
666 	"void main(){",
667 	"gl_FragColor=texture2D(texture,UV)*color;",
668 	"}"
669 	].join("");
670 
671 
672 	var vertexShader=gl.createShader(gl.VERTEX_SHADER);
673 	var fragmentShader=gl.createShader(gl.FRAGMENT_SHADER);
674 
675 	gl.shaderSource(vertexShader, vtxShader);
676 	gl.compileShader(vertexShader);
677 	if (!gl.getShaderParameter(vertexShader, gl.COMPILE_STATUS)) {
678 	      GLGE.error(gl.getShaderInfoLog(vertexShader));
679 	      return;
680 	}
681 
682 	gl.shaderSource(fragmentShader,frgShader);
683 	gl.compileShader(fragmentShader);
684 	if (!gl.getShaderParameter(fragmentShader, gl.COMPILE_STATUS)) {
685 	      GLGE.error(gl.getShaderInfoLog(fragmentShader));
686 	      return;
687 	}
688 
689 	if(this.program) gl.deleteProgram(this.Program);
690 	this.program = gl.createProgram();
691 	gl.attachShader(this.program, vertexShader);
692 	gl.attachShader(this.program, fragmentShader);
693 	gl.linkProgram(this.program);	
694 }
695 /**
696 * Creates the particle system buffers
697 * @private
698 */
699 GLGE.ParticleSystem.prototype.createBuffer=function(gl,array){
700 	var buffer = gl.createBuffer();
701 	gl.bindBuffer(gl.ARRAY_BUFFER, buffer);
702 	gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(array), gl.STATIC_DRAW);
703 	return buffer;
704 }
705 /**
706 * Sets the uniforms
707 * @private
708 */
709 GLGE.ParticleSystem.prototype.setUniforms=function(gl){
710 	var program=this.program;
711 	if(!program.glarrays) program.glarrays={};
712 	var cameraMatrix=gl.scene.camera.getViewMatrix();
713 	
714 	var pos=this.getPosition();
715 	//var camerapos=gl.scene.camera.getPosition();
716 	var mvMatrix=GLGE.mulMat4(cameraMatrix,[
717 		1,0,0,pos.x,
718 		0,1,0,pos.y,
719 		0,0,1,pos.z,
720 		0,0,0,1]);
721 
722 		
723 	var mvUniform = GLGE.getUniformLocation(gl,program, "mvMatrix");
724 
725 	if(!program.glarrays.mvMatrix) program.glarrays.mvMatrix=new Float32Array(mvMatrix);
726 		else GLGE.mat4gl(mvMatrix,program.glarrays.mvMatrix);
727 	gl.uniformMatrix4fv(mvUniform, true, program.glarrays.mvMatrix);
728 
729 	var pUniform = GLGE.getUniformLocation(gl,program, "pMatrix");
730 	if(!program.glarrays.pMatrix) program.glarrays.pMatrix=new Float32Array(gl.scene.camera.getProjectionMatrix());
731 			else GLGE.mat4gl(gl.scene.camera.getProjectionMatrix(),program.glarrays.pMatrix);	
732 	gl.uniformMatrix4fv(pUniform, true, program.glarrays.pMatrix);
733 
734 	gl.uniform1f(GLGE.getUniformLocation(gl,program, "time"), ((new Date()).getTime()-this.startTime));
735 	gl.uniform1i(GLGE.getUniformLocation(gl,program, "loop"), this.loop);
736 	
737 	
738 	gl.activeTexture(gl.TEXTURE0);
739 	//create the texture if it's not already created
740 	if(!this.glTexture) this.glTexture=gl.createTexture();
741 	//if the image is loaded then set in the texture data
742 	if(this.texture.state==1){
743 		gl.bindTexture(gl.TEXTURE_2D, this.glTexture);
744 		gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE,this.texture.image);
745 		gl.pixelStorei(gl.UNPACK_PREMULTIPLY_ALPHA_WEBGL, 1);
746 		gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
747 		gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR_MIPMAP_LINEAR);
748 		gl.generateMipmap(gl.TEXTURE_2D);
749 		gl.bindTexture(gl.TEXTURE_2D, null);
750 		this.texture.state=2;
751 		program.texture=true;
752 	}
753 	gl.bindTexture(gl.TEXTURE_2D, this.glTexture);
754 	if(program.texture){
755 		gl.uniform1i(GLGE.getUniformLocation(gl,program, "texture"), 0);
756 	}
757 
758 }

829 /**
830 * Renders the particle system
831 * @private
832 */
833 GLGE.ParticleSystem.prototype.GLRender=function(gl){
834 	if(!this.attribute) this.generateParticles(gl);
835 	if(!this.program) this.generateProgram(gl);
836 	
837 	gl.useProgram(this.program);
838 	this.setAttributes(gl);
839 	this.setUniforms(gl);
840 	gl.colorMask(0,0,0,0);
841 	gl.disable(gl.BLEND);
842 	gl.enable(gl.STENCIL_TEST);
843 	gl.stencilFunc(gl.ALWAYS, 1, 1);
844 	gl.stencilOp(gl.KEEP, gl.KEEP, gl.REPLACE);
845 	gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.facesGL);
846 	gl.drawElements(gl.TRIANGLES,this.facesGL.num, gl.UNSIGNED_SHORT, 0);
847 	gl.stencilFunc(gl.EQUAL, 1, 1);
848 	gl.stencilOp(gl.KEEP, gl.KEEP, gl.KEEP);
849 	gl.colorMask(1,1,1,1);
850 	gl.disable(gl.DEPTH_TEST);
851 	gl.enable(gl.BLEND);
852 	gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.facesGL);
853 	gl.drawElements(gl.TRIANGLES,this.facesGL.num, gl.UNSIGNED_SHORT, 0);
854 	gl.stencilOp(gl.REPLACE, gl.REPLACE, gl.REPLACE);
855 	gl.stencilFunc(gl.ALWAYS, 0, 0);
856 	gl.enable(gl.DEPTH_TEST);
857 	
858 	gl.scene.lastMaterial=null;
859 }
