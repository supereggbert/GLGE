 83 GLGE.Filter2d.prototype.createBuffer=function(gl,width,height){
 84 	if(!width) width=gl.canvas.width;
 85 	if(!height) height=gl.canvas.height;
 86 	var frameBuffer = gl.createFramebuffer();
 87 	var renderBuffer = gl.createRenderbuffer();
 88 	var texture = gl.createTexture();
 89 	gl.bindTexture(gl.TEXTURE_2D, texture);
 90 
 91 	var tex = new Uint8Array(width*height*4);
 92         gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, width, height, 0, gl.RGBA, gl.UNSIGNED_BYTE, tex);
 93     
 94 	gl.bindFramebuffer(gl.FRAMEBUFFER, frameBuffer);
 95 	gl.bindRenderbuffer(gl.RENDERBUFFER, renderBuffer);
 96 	gl.renderbufferStorage(gl.RENDERBUFFER, gl.DEPTH_COMPONENT16, width, height);
 97     
 98 	gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, renderBuffer);
 99 	gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_2D, texture, 0);
100 	
101 	gl.bindRenderbuffer(gl.RENDERBUFFER, null);
102 	gl.bindFramebuffer(gl.FRAMEBUFFER, null);
103 	gl.bindTexture(gl.TEXTURE_2D, null);
104 	return [frameBuffer,renderBuffer,texture];
105 }

233 //does all passes and renders result to screen
234 GLGE.Filter2d.prototype.GLRender=function(gl,buffer){
235 	gl.disable(gl.BLEND);
236 	if(!buffer) buffer=null;
237 	if(this.passes){
238 		for(var i=0;i<this.passes.length;i++){
239 			//set the frame buffer here
240 			if(this.passes.length-1==i){
241 				gl.bindFramebuffer(gl.FRAMEBUFFER, buffer);
242 			}else{
243 				if(!this.passes[i].buffer) this.passes[i].buffer=this.createBuffer(gl,this.passes[i].width,this.passes[i].height);
244 				gl.bindFramebuffer(gl.FRAMEBUFFER, this.passes[i].buffer[0]);
245 			}
246 			var width=(this.passes[i].width ? this.passes[i].width : gl.canvas.width);
247 			var height=(this.passes[i].height ? this.passes[i].height : gl.canvas.height);
248 			gl.viewport(0,0,width,height);
249 			gl.clearDepth(1.0);
250 			gl.depthFunc(gl.LEQUAL);
251 			gl.clearColor(0, 0, 0, 0);
252 			gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
253 			
254 			if(!this.passes[i].program){
255 				this.passes[i].program=this.GLCreateShader(gl,this.passes[i].GLSL);
256 			}
257 			gl.useProgram(this.passes[i].program);
258 			gl.program=this.passes[i].program;
259 			
260 			for(var j=0; j<8; j++) gl.disableVertexAttribArray(j);
261 			attribslot=GLGE.getAttribLocation(gl,this.passes[i].program, "position");
262 			if(!this.posBuffer) this.createPlane(gl);
263 
264 			gl.bindBuffer(gl.ARRAY_BUFFER, this.posBuffer);
265 			gl.enableVertexAttribArray(attribslot);
266 			gl.vertexAttribPointer(attribslot, this.posBuffer.itemSize, gl.FLOAT, false, 0, 0);
267 			this.GLSetUniforms(gl,i);
268 			gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.GLfaces);
269 			gl.drawElements(gl.TRIANGLES, this.GLfaces.numItems, gl.UNSIGNED_SHORT, 0);
270 		}
271 		if(this.persist){
272 			if(!this.persistTexture) this.createPersistTexture(gl);
273 			gl.bindTexture(gl.TEXTURE_2D, this.persistTexture);
274 			gl.copyTexImage2D(gl.TEXTURE_2D, 0, gl.RGBA, 0, 0, gl.canvas.width, gl.canvas.height, 0);
275 		}
276 	}
277 }

288 GLGE.Filter2d.prototype.GLSetUniforms=function(gl,pass){
289 	if(this.filterType==GLGE.FILTER_SKY){
290 		var invViewProj=GLGE.transposeMat4(GLGE.mulMat4(GLGE.inverseMat4(gl.scene.camera.matrix),GLGE.inverseMat4(gl.scene.camera.pMatrix)));
291 		GLGE.mat4gl(invViewProj,glmat)
292 		GLGE.setUniformMatrix(gl,"Matrix4fv",GLGE.getUniformLocation(gl,this.passes[pass].program, "invViewProj"),false,glmat);
293 	}
294 
295 	for(var key in this.uniforms){
296 		var uniform=this.uniforms[key];
297 		if(uniform.type=="Matrix4fv"){
298 			GLGE.setUniformMatrix(gl,"Matrix4fv",GLGE.getUniformLocation(gl,this.passes[pass].program, key),false,uniform.value);
299 		}else{
300 			GLGE.setUniform(gl,uniform.type,GLGE.getUniformLocation(gl,this.passes[pass].program, key),uniform.value);
301 		}
302 	}
303 
304 	
305 	var tidx=0;
306 	
307 	if(this.buffers){
308 		if(pass==0){
309 			gl.activeTexture(gl["TEXTURE"+tidx]);
310 			gl.bindTexture(gl.TEXTURE_2D, this.buffers[2]);
311 			gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
312 			gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
313 			gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
314 			gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
315 		}
316 		GLGE.setUniform(gl,"1i",GLGE.getUniformLocation(gl,this.passes[pass].program, "GLGE_RENDER"), tidx);
317 		tidx++;
318 		
319 		if(this.persist){
320 			if(pass==0){
321 				gl.activeTexture(gl["TEXTURE"+tidx]);
322 				gl.bindTexture(gl.TEXTURE_2D, this.persistTexture);
323 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
324 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
325 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
326 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
327 			}
328 			GLGE.setUniform(gl,"1i",GLGE.getUniformLocation(gl,this.passes[pass].program, "GLGE_PERSIST"), tidx);
329 			tidx++;
330 		}
331 		
332 		if(this.renderDepth){
333 			if(pass==0){
334 				gl.activeTexture(gl["TEXTURE"+tidx]);
335 				gl.bindTexture(gl.TEXTURE_2D, this.depthBuffers[2]);
336 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
337 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
338 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
339 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
340 			}
341 			GLGE.setUniform(gl,"1i",GLGE.getUniformLocation(gl,this.passes[pass].program, "GLGE_DEPTH"), tidx);
342 			tidx++;
343 		}
344 	    
345 	      if(this.renderEmit){
346 			if(pass==0){
347 				gl.activeTexture(gl["TEXTURE"+tidx]);
348 				gl.bindTexture(gl.TEXTURE_2D, this.emitBuffers[2]);
349 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
350 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
351 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
352 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
353 			}
354 			GLGE.setUniform(gl,"1i",GLGE.getUniformLocation(gl,this.passes[pass].program, "GLGE_EMIT"), tidx);
355 			tidx++;
356 	      }
357 	    
358 		
359 		if(this.renderNormal){
360 			if(pass==0){
361 				gl.activeTexture(gl["TEXTURE"+tidx]);
362 				gl.bindTexture(gl.TEXTURE_2D, this.normalBuffers[2]);
363 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
364 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
365 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
366 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
367 			}
368 			GLGE.setUniform(gl,"1i",GLGE.getUniformLocation(gl,this.passes[pass].program, "GLGE_NORMAL"), tidx);
369 			tidx++;
370 		}
371 		
372 		
373 		for(var i=0;i<this.passes.length;i++){
374 			if(this.passes[i].buffer){
375 				gl.activeTexture(gl["TEXTURE"+tidx]);
376 				gl.bindTexture(gl.TEXTURE_2D, this.passes[i].buffer[2]);
377 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
378 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
379 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
380 				gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
381 			}
382 			GLGE.setUniform(gl,"1i",GLGE.getUniformLocation(gl,this.passes[pass].program, "GLGE_PASS"+i), tidx);
383 			tidx++;
384 		}
385 	}
386 	
387 	if(!this.textures) this.textures=[];
388 	for(var i=0; i<this.textures.length;i++){
389 		gl.activeTexture(gl["TEXTURE"+(i+tidx)]);
390 		this.textures[i].doTexture(gl,null);
391 		var name = "TEXTURE"+i
392 		if(this.textures[i].name) name=this.textures[i].name;
393 		GLGE.setUniform(gl,"1i",GLGE.getUniformLocation(gl,this.passes[pass].program, name), i+tidx);
394 	}
395 }

397 /**
398 * creates the screen aligned plane mesh
399 * @private
400 */
401 GLGE.Filter2d.prototype.createPlane=function(gl){
402 	//create the vertex positions
403 	if(!this.posBuffer) this.posBuffer = gl.createBuffer();
404 	gl.bindBuffer(gl.ARRAY_BUFFER, this.posBuffer);
405 	gl.bufferData(gl.ARRAY_BUFFER, new Float32Array([1,1,0.5,-1,1,0.5,-1,-1,0.5,1,-1,0.5]), gl.STATIC_DRAW);
406 	this.posBuffer.itemSize = 3;
407 	this.posBuffer.numItems = 4;
408 	//create the faces
409 	if(!this.GLfaces) this.GLfaces = gl.createBuffer();
410 	gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.GLfaces);
411 	gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array([0,1,2,2,3,0]), gl.STATIC_DRAW);
412 	this.GLfaces.itemSize = 1;
413 	this.GLfaces.numItems = 6;
414 }

416 /**
417 * Creates a shader program
418 * @private
419 */
420 GLGE.Filter2d.prototype.GLCreateShader=function(gl,fragStr){
421 	//Vertex Shader
422 	var vertexStr=[];
423 	
424 	vertexStr.push("uniform mat4 invViewProj;\n");
425 	vertexStr.push("attribute vec3 position;\n");
426 	vertexStr.push("varying vec2 texCoord;\n");
427 	vertexStr.push("varying vec3 rayCoord;\n");
428 	
429 	vertexStr.push("void main(void){\n");
430 	vertexStr.push("vec4 near=invViewProj * vec4(position.xy,-1.0,1.0);\n");    
431 	vertexStr.push("near/=near.w;\n");    
432 	vertexStr.push("vec4 far=invViewProj * vec4(position.xy,1.0,1.0);\n");    
433 	vertexStr.push("far/=far.w;\n");    
434 	vertexStr.push("rayCoord=normalize(far.xyz-near.xyz);\n"); 
435 	vertexStr.push("texCoord=(position.xy+vec2(1.0,1.0))/2.0;\n");    
436 	vertexStr.push("gl_Position = vec4(position.xyz,1.0);\n");
437 	vertexStr.push("}\n");
438 	
439 	var GLVertexShader=GLGE.getGLShader(gl,gl.VERTEX_SHADER,vertexStr.join(""));
440 	var GLFragmentShader=GLGE.getGLShader(gl,gl.FRAGMENT_SHADER,fragStr);
441 
442 	return GLGE.getGLProgram(gl,GLVertexShader,GLFragmentShader);
443 }
