 34 (function(GLGE){
 35  
 36 /**
 37 * @class Postprocessing glow filter
 38 * @augments GLGE.Filter2d
 39 */
 40 GLGE.FilterGlow=function(uid){
 41 	this.setEmitBufferWidth(256);
 42 	this.setEmitBufferHeight(256);
 43 	GLGE.Assets.registerAsset(this,uid);
 44 };
 45 GLGE.augment(GLGE.Filter2d,GLGE.FilterGlow);
 46 GLGE.FilterGlow.prototype.renderEmit=true;
 47 GLGE.FilterGlow.prototype.blur=1.2;
 48 GLGE.FilterGlow.prototype.intensity=3;
 49 GLGE.FilterGlow.prototype.fxaacutoff=2;
 50 GLGE.FilterGlow.prototype.fxaastartintensity=0;
 51 
 52 GLGE.FilterGlow.prototype.setEmitBufferWidth=function(value){
 53 	GLGE.Filter2d.prototype.setEmitBufferWidth.call(this,value);
 54 	this.createPasses();
 55 	return this;
 56 }
 57 GLGE.FilterGlow.prototype.setEmitBufferHeight=function(value){
 58 	GLGE.Filter2d.prototype.setEmitBufferHeight.call(this,value);
 59 	this.createPasses();
 60 	return this;
 61 }
 62 GLGE.FilterGlow.prototype.setBlur=function(blur){
 63 	this.blur=blur;
 64 	this.createPasses();
 65 	return this;
 66 }
 67 GLGE.FilterGlow.prototype.setIntensity=function(intensity){
 68 	this.intensity=intensity;
 69 	this.createPasses();
 70 	return this;
 71 }
 72 GLGE.FilterGlow.prototype.setFXAA=function(value){
 73 	this.useFXAA=value;
 74 	this.createPasses();
 75 	return this;
 76 }
 77 GLGE.FilterGlow.prototype.setFXAACutoff=function(value){
 78 	this.fxaacutoff=value;
 79 	this.createPasses();
 80 	return this;
 81 }
 82 GLGE.FilterGlow.prototype.setFXAAStartIntensity=function(value){
 83 	this.fxaastartintensity=value;
 84 	this.createPasses();
 85 	return this;
 86 }
 87 GLGE.FilterGlow.prototype.createPasses=function(){
 88 	var pass1=[];
 89 	pass1.push("precision highp float;");
 90 	pass1.push("uniform sampler2D GLGE_EMIT;");
 91 	pass1.push("varying vec2 texCoord;");
 92 	pass1.push("float blurSize="+(1/this.emitBufferWidth*this.blur).toFixed(10)+";");
 93 	pass1.push("float rand(vec2 co){;");
 94 	pass1.push("return fract(sin(dot(co.xy ,vec2(12.9898,78.233))) * 43758.5453);");
 95 	pass1.push("}");
 96 	pass1.push("void main(void){");
 97 	pass1.push("vec4 color=vec4(0.0,0.0,0.0,0.0);");
 98 	pass1.push("float rnd=1.0-rand(texCoord.xy)*4.0*blurSize;");
 99 	pass1.push("color += texture2D(GLGE_EMIT, vec2(texCoord.x - 4.0*blurSize, texCoord.y)) * 0.05 * rnd;");
100 	pass1.push("color += texture2D(GLGE_EMIT, vec2(texCoord.x - 3.0*blurSize, texCoord.y)) * 0.09 * rnd;");
101 	pass1.push("color += texture2D(GLGE_EMIT, vec2(texCoord.x - 2.0*blurSize, texCoord.y)) * 0.12 * rnd;");
102 	pass1.push("color += texture2D(GLGE_EMIT, vec2(texCoord.x - blurSize, texCoord.y)) * 0.15 * rnd;");
103 	pass1.push("color += texture2D(GLGE_EMIT, vec2(texCoord.x, texCoord.y)) * 0.18 * rnd;");
104 	pass1.push("color += texture2D(GLGE_EMIT, vec2(texCoord.x + blurSize, texCoord.y)) * 0.15 * rnd;");
105 	pass1.push("color += texture2D(GLGE_EMIT, vec2(texCoord.x + 2.0*blurSize, texCoord.y)) * 0.12 * rnd;");
106 	pass1.push("color += texture2D(GLGE_EMIT, vec2(texCoord.x + 3.0*blurSize, texCoord.y)) * 0.09 * rnd;");
107 	pass1.push("color += texture2D(GLGE_EMIT, vec2(texCoord.x + 4.0*blurSize, texCoord.y)) * 0.05 * rnd;");
108 	pass1.push("gl_FragColor = vec4(color.rgb,1.0);");
109 	pass1.push("}");
110 	
111 	var pass2=[];
112 	pass2.push("precision highp float;");
113 	pass2.push("uniform sampler2D GLGE_PASS0;");
114 	pass2.push("uniform sampler2D GLGE_RENDER;");
115 	pass2.push("uniform sampler2D GLGE_EMIT;");
116 	pass2.push("varying vec2 texCoord;");
117 	pass2.push("float blurSize="+(1/this.emitBufferHeight*this.blur).toFixed(10)+";");
118 	pass2.push("float rand(vec2 co){;");
119 	pass2.push("return fract(sin(dot(co.xy ,vec2(12.9898,78.233))) * 43758.5453);");
120 	pass2.push("}");
121 	pass2.push("void main(void){");
122 	pass2.push("vec4 color=vec4(0.0,0.0,0.0,0.0);");
123 	pass2.push("float rnd=1.0-rand(texCoord.xy)*4.0*blurSize;");
124 	pass2.push("color += texture2D(GLGE_PASS0, vec2(texCoord.x, texCoord.y - 4.0*blurSize)) * 0.05 * rnd;");
125 	pass2.push("color += texture2D(GLGE_PASS0, vec2(texCoord.x, texCoord.y - 3.0*blurSize)) * 0.09 * rnd;");
126 	pass2.push("color += texture2D(GLGE_PASS0, vec2(texCoord.x, texCoord.y - 2.0*blurSize)) * 0.12 * rnd;");
127 	pass2.push("color += texture2D(GLGE_PASS0, vec2(texCoord.x, texCoord.y - blurSize)) * 0.15 * rnd;");
128 	pass2.push("color += texture2D(GLGE_PASS0, vec2(texCoord.x, texCoord.y)) * 0.18 * rnd;");
129 	pass2.push("color += texture2D(GLGE_PASS0, vec2(texCoord.x, texCoord.y + blurSize)) * 0.15 * rnd;");
130 	pass2.push("color += texture2D(GLGE_PASS0, vec2(texCoord.x, texCoord.y + 2.0*blurSize)) * 0.12 * rnd;");
131 	pass2.push("color += texture2D(GLGE_PASS0, vec2(texCoord.x, texCoord.y + 3.0*blurSize)) * 0.09 * rnd;");
132 	pass2.push("color += texture2D(GLGE_PASS0, vec2(texCoord.x, texCoord.y + 4.0*blurSize)) * 0.05 * rnd;");
133 	pass2.push("gl_FragColor = vec4(color.rgb*"+(this.intensity.toFixed(5))+"+texture2D(GLGE_RENDER,texCoord).rgb,1.0);");
134 	pass2.push("}");
135 	
136 	
137 	this.passes=[];
138 	this.addPass(pass1.join(""));
139 	this.addPass(pass2.join(""));
140 	
141 	if(this.useFXAA){
142 		var pass3=[]
143 		pass3.push("precision highp float;");
144 		pass3.push("uniform sampler2D GLGE_PASS1;");
145 		pass3.push("varying vec2 texCoord;");
146 		pass3.push("vec2 inverse_buffer_size=vec2(1.0/1280.0,1.0/720.0);");
147 		pass3.push("#define FXAA_REDUCE_MIN   (1.0/128.0)");
148 		pass3.push("#define FXAA_REDUCE_MUL   (1.0/16.0)");
149 		pass3.push("#define FXAA_SPAN_MAX     8.0");
150 		pass3.push("void  main(){");
151 		pass3.push("	vec3 rgbNW = texture2D(GLGE_PASS1,  (gl_FragCoord.xy + vec2(-1.0,-1.0)) * inverse_buffer_size).xyz;");
152 		pass3.push("	vec3 rgbNE = texture2D(GLGE_PASS1,  (gl_FragCoord.xy + vec2(1.0,-1.0)) * inverse_buffer_size).xyz;");
153 		pass3.push("	vec3 rgbSW = texture2D(GLGE_PASS1,  (gl_FragCoord.xy + vec2(-1.0,1.0)) * inverse_buffer_size).xyz;");
154 		pass3.push("	vec3 rgbSE = texture2D(GLGE_PASS1,  (gl_FragCoord.xy + vec2(1.0,1.0)) * inverse_buffer_size).xyz;");
155 		pass3.push("	vec3 rgbM  = texture2D(GLGE_PASS1,  gl_FragCoord.xy  * inverse_buffer_size).xyz;");
156 		pass3.push("	vec3 luma = vec3(0.299, 0.587, 0.114);");
157 		pass3.push("	float lumaNW = dot(rgbNW, luma);");
158 		pass3.push("	float lumaNE = dot(rgbNE, luma);");
159 		pass3.push("	float lumaSW = dot(rgbSW, luma);");
160 		pass3.push("	float lumaSE = dot(rgbSE, luma);");
161 		pass3.push("	float lumaM  = dot(rgbM,  luma);");
162 		pass3.push("	float lumaMin = min(lumaM, min(min(lumaNW, lumaNE), min(lumaSW, lumaSE)));");
163 		pass3.push("	float lumaMax = max(lumaM, max(max(lumaNW, lumaNE), max(lumaSW, lumaSE)));");
164 			
165 		pass3.push("	vec2 dir;");
166 		pass3.push("	dir.x = -((lumaNW + lumaNE) - (lumaSW + lumaSE));");
167 		pass3.push("	dir.y =  ((lumaNW + lumaSW) - (lumaNE + lumaSE));");
168 			
169 		pass3.push("	float dirReduce = max(");
170 		pass3.push("	(lumaNW + lumaNE + lumaSW + lumaSE) * (0.25 * FXAA_REDUCE_MUL),");
171 		pass3.push("	FXAA_REDUCE_MIN);");
172 			
173 		pass3.push("	float rcpDirMin = 1.0/(min(abs(dir.x), abs(dir.y)) + dirReduce);");
174 		pass3.push("	dir = min(vec2( FXAA_SPAN_MAX,  FXAA_SPAN_MAX),");
175 		pass3.push("	max(vec2(-FXAA_SPAN_MAX, -FXAA_SPAN_MAX),");
176 		pass3.push("	dir * rcpDirMin)) * inverse_buffer_size;");
177 			  
178 		pass3.push("	vec3 rgbA = 0.5 * (");
179 		pass3.push("	texture2D(GLGE_PASS1,   gl_FragCoord.xy  * inverse_buffer_size + dir * (1.0/3.0 - 0.5)).xyz +");
180 		pass3.push("	texture2D(GLGE_PASS1,   gl_FragCoord.xy  * inverse_buffer_size + dir * (2.0/3.0 - 0.5)).xyz);");
181 			
182 		pass3.push("	vec3 rgbB = rgbA * 0.5 + 0.25 * (");
183 		pass3.push("	texture2D(GLGE_PASS1,  gl_FragCoord.xy  * inverse_buffer_size + dir *  - 0.5).xyz +");
184 		pass3.push("	texture2D(GLGE_PASS1,  gl_FragCoord.xy  * inverse_buffer_size + dir * 0.5).xyz);");
185 		pass3.push("	float lumaB = dot(rgbB, luma);");
186 		pass3.push("	if((lumaB < lumaMin) || (lumaB > lumaMax)) gl_FragColor = vec4(rgbA,1.0);");
187 		pass3.push("	    else gl_FragColor = vec4(rgbB,1.0);");
188 		pass3.push("	if(length(rgbM)>"+this.fxaacutoff.toFixed(2)+") gl_FragColor = vec4(rgbM,1.0);");
189 		pass3.push("	if(length(rgbM)<"+this.fxaastartintensity.toFixed(2)+") gl_FragColor = vec4(rgbM,1.0);");
190 		pass3.push("}");
191 		this.addPass(pass3.join("\n"));
192 	}
193 	
194 }
195 
196 
197 
198 })(GLGE);
