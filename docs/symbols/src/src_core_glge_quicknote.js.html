 36 (function(GLGE){
 37 	
 38 /**
 39 * @class class to implelemnt quick notation
 40 */
 41 GLGE.QuickNotation=function(){
 42 }
 43 /**
 44 * Call to set properties and add children to an object
 45 * @example myObject._({LocX:10,LocY:20},child1,child2,.....);
 46 */
 47 GLGE.QuickNotation.prototype._=function(){
 48 	var argument;
 49 	for(var i=0; i<arguments.length;i++){
 50 		argument=arguments[i];
 51 		if(typeof argument=="object"){
 52 			if(argument.className && this["add"+argument.className]){
 53 				this["add"+argument.className](argument);
 54 			}else{
 55 				for(var key in argument){
 56 					if(this["set"+key]){
 57 						this["set"+key](argument[key]);
 58 					}
 59 				}
 60 			}
 61 		}
 62 	}
 63 	return this;
 64 }
 65 
 66 })(GLGE);
