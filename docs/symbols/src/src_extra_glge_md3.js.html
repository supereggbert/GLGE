 90 /**
 91 * Gets the absolute path given an import path and the path it's relative to
 92 * @param {string} path the path to get the absolute path for
 93 * @param {string} relativeto the path the supplied path is relativeto
 94 * @returns {string} absolute path
 95 * @private
 96 */
 97 GLGE.MD3.prototype.getAbsolutePath=function(path,relativeto){
 98 	if(path.substr(0,7)=="http://" || path.substr(0,7)=="file://"  || path.substr(0,7)=="https://"){
 99 		return path;
100 	}
101 	else
102 	{
103 		if(!relativeto){
104 			relativeto=window.location.href;
105 		}
106 		if (relativeto.indexOf("://")==-1){
107 			return relativeto.slice(0,relativeto.lastIndexOf("/"))+"/"+path;
108 		}
109 		//find the path compoents
110 		var bits=relativeto.split("/");
111 		var domain=bits[2];
112 		var proto=bits[0];
113 		var initpath=[];
114 		for(var i=3;i<bits.length-1;i++){
115 			initpath.push(bits[i]);
116 		}
117 		//relative to domain
118 		if(path.substr(0,1)=="/"){
119 			initpath=[];
120 		}
121 		var locpath=path.split("/");
122 		for(var i=0;i<locpath.length;i++){
123 			if(locpath[i]=="..") initpath.pop();
124 				else if(locpath[i]!="") initpath.push(locpath[i]);
125 		}
126 		return proto+"//"+domain+"/"+initpath.join("/");
127 	}
128 }

131 /**
132 * Sets the url of the MD3 model
133 * @param {string} url the url to the MD3 file
134 */
135 GLGE.MD3.prototype.setSrc=function(url,relativeTo){
136 	if(relativeTo) url=this.getAbsolutePath(url,relativeTo);
137 	this.url=url;
138 	
139 	var that=this;
140 	var xhr = new XMLHttpRequest();
141 	xhr.overrideMimeType('text/plain; charset=x-user-defined');
142 	xhr.open("GET", url, true);
143 	xhr.send(null);
144 	this.verts=[];
145 	this.normals=[];
146 	
147 	xhr.onreadystatechange = function (aEvt) {
148 		if (xhr.readyState == 4) {
149 			if(xhr.status == 200 || xhr.status == 0){
150 				response = xhr.responseText;
151 				if (response) {
152 					var buffer = new ArrayBuffer(response.length);
153 					var byteArray = new Uint8Array(buffer);
154 					var byteArray = [];
155 					for(var i=0;i<response.length;i++){
156 						byteArray[i]=response.charCodeAt(i) & 0xff;
157 					}
158 					that.bufferLoaded(byteArray);
159 				}
160 			}else{
161 				alert("Error loading page\n");
162 			}
163 		}
164 	};
165 }

303 /**
304 * Extract frame info
305 * @private
306 */
307 GLGE.MD3.prototype.parseFrames=function(){
308 	var start=this.headers.OFS_FRAMES+40;
309 	var frameSize=56;
310 	var animations={};
311 	var lastName=false;
312 	var firstFrame=0;
313 	for(var i=0;i<this.headers.NUM_FRAMES;i++){
314 		var name=this.getStringAt(start+i*frameSize,16).replace(/[0-9_]/g,'');
315 		if(lastName && lastName!=name){
316 			animations[lastName]=[firstFrame,i-1];
317 			firstFrame=i;
318 		}
319 		lastName=name;
320 	}
321 	animations[lastName]=[firstFrame,i-3];
322 
323 	this.MD3Animations[this.url]=animations;
324 }
325 /**
326 * Extract header info
327 * @private
328 */
329 GLGE.MD3.prototype.parseHeader=function(){
330 	this.headers={};
331 	for (var i=0; i<this.headerNames.length; i++) {
332 		this.headers[this.headerNames[i]]=this.getSint32At(i*4+76);
333 	}
334 }
335 /**
336 * Parse the surfaces 
337 * @private
338 */
339 GLGE.MD3.prototype.parseSurfaces=function(){
340 	this.surfaceHeaders=[];
341 	var offset=this.headers.OFS_SURFACES;
342 	for(var i=0;i<this.headers.NUM_SURFACES;i++){
343 		var start=offset+72;
344 		var sHeaders=this.surfaceHeaders[i]={offset:offset};
345 		var idx=0;
346 		for(var j=start;j<start+36;j=j+4){
347 			sHeaders[this.surfaceHeaderNames[idx++]]=this.getSint32At(j);
348 		}
349 		var normverts=this.parseNormVerts(i);
350 		var uvs=this.parseUVs(i);
351 		var faces=this.parseFaces(i);
352 		var mesh=this.createMesh(normverts[0],normverts[1],uvs,faces);
353 		var surface=new GLGE.Object;
354 		if(!this.MD3Materials[i]) this.MD3Materials[i]=new GLGE.Material;
355 		surface.setMaterial(this.MD3Materials[i]);
356 		surface.setMesh(mesh);
357 		this.surfaces.push(surface);
358 		offset+=this.surfaceHeaders[i].OFS_END;
359 	}
360 }

375 /**
376 * Parse the Faces
377 * @private
378 */
379 GLGE.MD3.prototype.parseFaces=function(surface){
380 	var header=this.surfaceHeaders[surface];
381 	var faces=[];
382 	var start=header.offset+header.OFS_TRIANGLES;
383 	var end=12*header.NUM_TRIANGLES;
384 	for(var i=start; i<start+end;i=i+12){
385 		faces.push(this.getSint32At(i));
386 		faces.push(this.getSint32At(i+4));
387 		faces.push(this.getSint32At(i+8));
388 	}
389 	return faces;
390 }
391 /**
392 * Parse the vertex UVs
393 * @private
394 */
395 GLGE.MD3.prototype.parseUVs=function(surface){
396 	var header=this.surfaceHeaders[surface];
397 	var uvs=[];
398 	var start=header.offset+header.OFS_ST;
399 	var end=8*header.NUM_VERTS;
400 	for(var i=start; i<start+end;i=i+8){
401 		uvs.push(this.getFloat32At(i));
402 		uvs.push(1-this.getFloat32At(i+4));
403 	}
404 	return uvs;
405 }
406 /**
407 * Parse the verts for each frame
408 * @private
409 */
410 GLGE.MD3.prototype.parseNormVerts=function(surface){
411 	var header=this.surfaceHeaders[surface];
412 	var verts=[];
413 	var normals=[];
414 	var start=header.offset+header.OFS_XYZNORMAL;
415 	var frameSize=8*header.NUM_VERTS;
416 	for(var frame=0; frame<header.NUM_FRAMES; frame++){
417 		var frameVerts=[];
418 		var frameNormals=[];
419 		for(var i=start+frame*frameSize; i<start+(frame+1)*frameSize;i=i+8){
420 			frameVerts.push(this.getSint16At(i)/64);
421 			frameVerts.push(this.getSint16At(i+2)/64);
422 			frameVerts.push(this.getSint16At(i+4)/64);
423 			var norm=this.decodeNormal(this.byteArray[i+6],this.byteArray[i+7]);
424 			frameNormals.push(norm[0]);
425 			frameNormals.push(norm[1]);
426 			frameNormals.push(norm[2]);
427 		}
428 		verts[frame]=frameVerts;
429 		normals[frame]=frameNormals;
430 	}
431 	return [verts,normals];
432 }

434 /**
435 * Decode the normal coords
436 * @private
437 */
438 GLGE.MD3.prototype.decodeNormal=function(zenith,azimuth){
439 	var lat = zenith * (2 * Math.PI) / 255;
440 	var lng = azimuth * (2 * Math.PI) / 255;
441 	var clat=Math.cos(lat);
442 	var slat=Math.sin(lat);
443 	var clng=Math.cos(lng);
444 	var slng=Math.sin(lng);
445 	return [-clat*slng,-slat*slng,clng];
446 }

458 /**
459 * get 16 bit int at location
460 * @private
461 */
462 GLGE.MD3.prototype.getSint16At=function(index){
463 	var value=this.byteArray[index]|(this.byteArray[index+1]<<8);
464 	if(value>0x8000) value=value-0x10000;
465 	return value;
466 }
467 /**
468 * get 32 bit signed int at location
469 * @private
470 */
471 GLGE.MD3.prototype.getSint32At=function(index){
472 	var value=this.byteArray[index]|(this.byteArray[index+1]<<8)|(this.byteArray[index+2]<<16)|(this.byteArray[index+3]<<24);
473 	if(value>0x80000000) value=value-0x100000000;
474 	return value;
475 }

477 /**
478 * Get 32 bit float at location
479 * @private
480 */
481 GLGE.MD3.prototype.getFloat32At=function(index){
482 	var b3=this.byteArray[index];
483 	var b2=this.byteArray[index+1];
484 	var b1=this.byteArray[index+2];
485 	var b0=this.byteArray[index+3];
486 	sign = 1 - (2 * (b0 >> 7)),
487 	exponent = (((b0 << 1) & 0xff) | (b1 >> 7)) - 127,
488 	mantissa = ((b1 & 0x7f) << 16) | (b2 << 8) | b3;
489 
490 	if (mantissa == 0 && exponent == -127) {
491 		return 0.0;
492 	}
493 
494 	if (exponent == -127) { // Denormalized
495 		return sign * mantissa * Math.pow(2, -126 - 23);
496 	}
497 
498 	return sign * (1 + mantissa * Math.pow(2, -23)) * Math.pow(2, exponent);
499 }

501 /**
502 * Get 32 bit float at location
503 * @private
504 */
505 GLGE.MD3.prototype.getStringAt=function(index,size){
506 	var name="";
507 	for(var i=index;i<index+size;i++){
508 		if(this.byteArray[i]==0) break;
509 		name+=String.fromCharCode(this.byteArray[i]);
510 	}
511 	return name;
512 }

549 /**
550 * Sets the MD3 frame number
551 * @param {string} frame the frame to display
552 */
553 GLGE.MD3.prototype.setMD3Frame=function(frame){
554 	var totalframes=this.MD3EndFrame-this.MD3StartFrame;
555 	if(totalframes==0) return;
556 	if(this.MD3Loop){
557 		frame=frame%totalframes;
558 		var frame2=((Math.floor(frame)+1)%totalframes);
559 	}else{
560 		frame=Math.min(totalframes,frame);
561 		frame2=Math.min(totalframes,Math.floor(frame)+1);
562 		if(frame==totalframes){
563 			this.fireEvent("md3AnimFinished",{});
564 		}
565 	}
566 	var framefrac=frame%1;
567 	if(frame<1 && this.MD3LastAnimFrame){
568 		frame=this.MD3LastAnimFrame-this.MD3StartFrame;
569 	}else{
570 		this.MD3LastAnimFrame=null;
571 		this.lastMD3Frame=Math.floor(frame)+this.MD3StartFrame;
572 	}
573 	for(var i=0;i<this.surfaces.length;i++){
574 		this.surfaces[i].setMeshFrame1(Math.floor(frame)+this.MD3StartFrame);
575 		this.surfaces[i].setMeshFrame2(frame2+this.MD3StartFrame);
576 		this.surfaces[i].setMeshBlendFactor(framefrac);
577 	}
578 }

584 		this.setMD3Frame(frame);
585 	}
586 	GLGE.Object.prototype.animate.call(this,now,nocache);
587 }
