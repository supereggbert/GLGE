 29 /**

 35 (function(GLGE){
 36 
 37 /**
 38 * @class A wrapping class for jiglib spheres
 39 * @augments GLGE.PhysicsAbstract
 40 */
 41 GLGE.PhysicsBox=function(uid){
 42 	this.jigLibObj=new jigLib.JBox(this,this.width,this.height,this.depth);
 43 	this.jigLibObj.GLGE=this;
 44 	this.jigLibObj.addEventListener(jigLib.JCollisionEvent.COLLISION, function(event){this.GLGE.fireEvent("collision",{obj:event.collisionBody.GLGE,impulse:event.collisionImpulse})});
 45 	GLGE.PhysicsAbstract.call(this,uid);
 46 }
 47 GLGE.augment(GLGE.PhysicsAbstract,GLGE.PhysicsBox);
 48 
 49 GLGE.PhysicsBox.prototype.width=1;
 50 GLGE.PhysicsBox.prototype.height=1;
 51 GLGE.PhysicsBox.prototype.depth=1;
 52 
 53 GLGE.PhysicsBox.prototype.className="PhysicsBox";
 54 /**
 55 * Sets the width of the box
 56 * @param {number} value The width to set
 57 */
 58 GLGE.PhysicsBox.prototype.setWidth=function(value){
 59 	this.width=value;
 60 	var sides=this.jigLibObj.get_sideLengths();
 61 	sides[0]=+value
 62 	this.jigLibObj.set_sideLengths(sides);
 63 	return this;
 64 }
 65 /**
 66 * Sets the height of the box
 67 * @param {number} value The height to set
 68 */
 69 GLGE.PhysicsBox.prototype.setHeight=function(value){
 70 	this.height=value;
 71 	var sides=this.jigLibObj.get_sideLengths();
 72 	sides[1]=+value
 73 	this.jigLibObj.set_sideLengths(sides);
 74 	return this;
 75 }
 76 
 77 /**
 78 * Sets the height of the box
 79 * @param {number} value The depth to set
 80 */
 81 GLGE.PhysicsBox.prototype.setDepth=function(value){
 82 	this.depth=value;
 83 	var sides=this.jigLibObj.get_sideLengths();
 84 	sides[2]=+value
 85 	this.jigLibObj.set_sideLengths(sides);
 86 	return this;
 87 }
 88 
 89 /**
 90 * Gets the width of the box
 91 * @returns {number} The width to set
 92 */
 93 GLGE.PhysicsBox.prototype.getWidth=function(){
 94 	return this.jigLibObj.get_sideLengths()[0];
 95 }
 96 
 97 /**
 98 * Gets the height of the box
 99 * @returns {number} The height to set
100 */
101 GLGE.PhysicsBox.prototype.getHeight=function(){
102 	return this.jigLibObj.get_sideLengths()[1];
103 }
104 
105 /**
106 * Gets the depth of the box
107 * @returns {number} The depth to set
108 */
109 GLGE.PhysicsBox.prototype.getDepth=function(){
110 	return this.jigLibObj.get_sideLengths()[2];
111 }
112 
113 })(GLGE);
