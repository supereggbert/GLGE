 36 (function(GLGE){

 41 /**
 42 * @class A events class
 43 **/
 44 GLGE.Events=function(){
 45 }
 46 /**
 47 * Fires an event
 48 * @param {string} event The name of the event to fire
 49 * @param {object} data the events data
 50 **/
 51 GLGE.Events.prototype.fireEvent=function(event,data){
 52 	if(this.events && this.events[event]){
 53 		var events=this.events[event];
 54 		for(var i=0;i<events.length;i++){
 55 			if(events[i] && events[i].call) events[i].call(this,data);
 56 		}
 57 	}
 58 }
 59 /**
 60 * Adds an event listener
 61 * @param {string} event The name of the event to listen for
 62 * @param {function} fn the event callback
 63 **/
 64 GLGE.Events.prototype.addEventListener=function(event,fn){
 65 	if(!this.events) this.events={};
 66 	if(!this.events[event]) this.events[event]=[];
 67 	this.events[event].push(fn);
 68 }
 69 /**
 70 * Removes an event listener
 71 * @param {function} fn the event callback to remove
 72 **/
 73 GLGE.Events.prototype.removeEventListener=function(event,fn){
 74     if(!this.events[event]) return;
 75 	var idx=this.events[event].indexOf(fn);
 76 	if(idx!=-1) this.events[event].splice(idx,1);
 77 }
 78 
 79 })(GLGE);
