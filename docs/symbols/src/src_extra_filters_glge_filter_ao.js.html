 34 (function(GLGE){
 35  
 36 /**
 37 * @class Postprocessing Ambient Occlusion filter
 38 * @augments GLGE.Filter2d
 39 */
 40 GLGE.FilterAO=function(){
 41 	this.setUniform("1f","cavitygamma",1/3);
 42 	this.setUniform("1f","whiteMul",2);
 43 	this.setUniform("1f","aogamma",1/3);
 44 	this.setUniform("1f","maxDist",0.025);
 45 	this.passes=[];
 46 };
 47 GLGE.augment(GLGE.Filter2d,GLGE.FilterAO);
 48 GLGE.FilterAO.prototype.renderNormal=true;
 49 GLGE.FilterAO.prototype.quality=1;
 50 GLGE.FilterAO.prototype.range=80;
 51 GLGE.FilterAO.prototype.samples=16;
 52 GLGE.FilterAO.prototype.useRender=true;
 53 
 54 GLGE.FilterAO.prototype.getNormalBufferHeight=function(){
 55 	return (this.normalBufferHeight ? this.normalBufferHeight : (this.gl.canvas.height*this.quality|0));
 56 }
 57 
 58 GLGE.FilterAO.prototype.getNormalBufferWidth=function(){
 59 	return (this.normalBufferWidth ? this.normalBufferWidth : (this.gl.canvas.width*this.quality|0));
 60 }
 61 
 62 GLGE.FilterAO.prototype.setUseRender=function(value){
 63 	this.useRender=value;
 64 	this.normalBuffers=null;
 65 	this.passes=[];
 66 	return this;
 67 }
 68 
 69 GLGE.FilterAO.prototype.setSamples=function(value){
 70 	this.samples=value;
 71 	this.normalBuffers=null;
 72 	this.passes=[];
 73 	return this;
 74 }
 75 
 76 GLGE.FilterAO.prototype.setQuality=function(value){
 77 	this.quality=value;
 78 	this.normalBuffers=null;
 79 	this.passes=[];
 80 	return this;
 81 }
 82 
 83 GLGE.FilterAO.prototype.setRange=function(value){
 84 	this.range=value;
 85 	if(this.gl){
 86 		this.setUniform("1f","blurX",this.range/this.getNormalBufferWidth()*this.quality/this.samples);
 87 		this.setUniform("1f","blurY",this.range/this.getNormalBufferHeight()/this.samples);
 88 	}
 89 	return this;
 90 }
 91 
 92 GLGE.FilterAO.prototype.setCavityGamma=function(value){
 93 	this.setUniform("1f","cavitygamma",1/value);
 94 	return this;
 95 }
 96 GLGE.FilterAO.prototype.setAmbientMultiplier=function(value){
 97 	this.setUniform("1f","whiteMul",value);
 98 	return this;
 99 }
100 GLGE.FilterAO.prototype.setAmbientGamma=function(value){
101 	this.setUniform("1f","aogamma",1/value);
102 	return this;
103 }
104 GLGE.FilterAO.prototype.setMaximumDistance=function(value){
105 	this.setUniform("1f","maxDist",value);
106 	return this;
107 }
108 
109 GLGE.FilterAO.prototype.GLRender=function(gl,buffer){
110 	this.gl=gl;
111 	if(this.passes.length==0){
112 		this.createPasses();
113 	}
114 	return GLGE.Filter2d.prototype.GLRender.call(this,gl,buffer)
115 }
116 
117 GLGE.FilterAO.prototype.createPasses=function(){
118 	if(!this.gl) return;
119 	
120 	
121 	var width=this.getNormalBufferWidth();
122 	var height=this.getNormalBufferHeight();
123 	
124 	
125 	var size=(this.samples/4)|0;
126 	var weights=[];
127 	for(var i=-size,cnt=0; i<=size;i++,cnt++){
128 		var n=size-Math.abs(i)+1;
129 		weights[cnt]=n/(size*size+size);
130 	}
131 	weights[size]=0;
132 	
133 	this.setUniform("1f","blurX",this.range/width*this.quality/this.samples);
134 	this.setUniform("1f","blurY",this.range/height/this.samples);
135 
136 
137 
138 	
139 	var pass1=[];
140 	pass1.push("precision highp float;");
141 	pass1.push("uniform sampler2D GLGE_NORMAL;");
142 	pass1.push("uniform float maxDist;");
143 	pass1.push("varying vec2 texCoord;");
144 	pass1.push("uniform float blurX;");
145 	pass1.push("float rand(vec2 co){");
146 	pass1.push("return (fract(sin(dot(co.xy ,vec2(12.9898,78.233))) * 43758.5453)-0.5)*2.0;");
147 	pass1.push("}");
148 	pass1.push("void main(void){");
149 	pass1.push("vec4 n=texture2D(GLGE_NORMAL,texCoord.xy).rgba;");
150 	pass1.push("vec4 color=vec4(0.0,0.0,0.0,n.a);");
151 	pass1.push("float blurSize=blurX/(n.a*n.a+1.0);");
152 	pass1.push("float offset=rand(texCoord.xy)*blurSize+texCoord.x;");
153 	pass1.push("vec3 samp;");
154 	pass1.push("float delta;");
155 	for(var i=-size,cnt=0;i<=size;i++,cnt++){
156 		if(i==0) continue;
157 		pass1.push("samp = texture2D(GLGE_NORMAL, vec2("+i+".0*blurSize+offset, texCoord.y)).rga;");
158 		pass1.push("delta=abs(n.a-samp.b);");
159 		pass1.push("if(delta<maxDist){");
160 		pass1.push("delta/=maxDist;");
161 		pass1.push("color.b -= (samp.r-0.5) * "+weights[cnt]+" * "+((2*i)/Math.abs(i) | 0)+".0;");
162 		pass1.push("color.rg += samp.rg * "+weights[cnt]+" * (1.0-delta);");
163 		pass1.push("color.rg += n.rg  * "+weights[cnt]+" * delta;");
164 		pass1.push("}else{");
165 		pass1.push("color.rg +=n.rg * "+weights[cnt]+";");
166 		pass1.push("}");
167 	}
168 	pass1.push("color.b = (color.b+1.0)*0.5;");
169 	pass1.push("gl_FragColor = color;");
170 	pass1.push("}");
171 	
172 	var pass2=[];
173 	pass2.push("precision highp float;");
174 	pass2.push("uniform sampler2D GLGE_PASS0;");
175 	pass2.push("uniform sampler2D GLGE_RENDER;");
176 	pass2.push("uniform sampler2D GLGE_NORMAL;");
177 	pass2.push("varying vec2 texCoord;");
178 	pass2.push("uniform float blurY;");
179 	
180 	
181 	pass2.push("uniform float cavitygamma;");
182 	pass2.push("uniform float whiteMul;");
183 	pass2.push("uniform float aogamma;");
184 	pass2.push("uniform float maxDist;");
185 	
186 	
187 	pass2.push("float rand(vec2 co){");
188 	pass2.push("return (fract(sin(dot(co.xy ,vec2(12.9898,78.233))) * 43758.5453)-0.5)*2.0;");
189 	pass2.push("}");
190 	pass2.push("void main(void){");
191 	pass2.push("vec4 color=vec4(0.0,0.0,0.0,1.0);");
192 	pass2.push("vec4 samp=vec4(0.0);");
193 	pass2.push("float random=rand(texCoord.xy);");
194 	if(this.quality<1){
195 		pass2.push("vec2 displace=vec2("+(0.5/width)+","+(0.5/height)+")*random;");
196 		pass2.push("vec4 n=texture2D(GLGE_PASS0, texCoord.xy+displace);");
197 	}else{
198 		pass2.push("vec4 n=texture2D(GLGE_PASS0, texCoord.xy);");
199 	}
200 	pass2.push("float delta;");
201 	pass2.push("float blurSize=blurY/(n.a*n.a+1.0);");
202 	pass2.push("float offset=random*blurSize+texCoord.y;");
203 	
204 	for(var i=-size,cnt=0;i<=size;i++,cnt++){
205 		if(i==0) continue;
206 		if(this.quality<1){
207 			pass2.push("samp = texture2D(GLGE_PASS0, vec2(texCoord.x, "+i+".0*blurSize + offset)+displace);");
208 		}else{
209 			pass2.push("samp = texture2D(GLGE_PASS0, vec2(texCoord.x, "+i+".0*blurSize + offset));");
210 		}
211 		pass2.push("delta=abs(n.a-samp.a);");
212 		pass2.push("if(delta<maxDist){");
213 		pass2.push("delta/=maxDist;");
214 		pass2.push("color.a -= (samp.g-0.5) * "+weights[cnt]+" * "+((i*2)/Math.abs(i) | 0)+".0;");
215 		pass2.push("color.rg += samp.rg  * "+weights[cnt]+" * (1.0-delta);");
216 		pass2.push("color.rg += n.rg * "+weights[cnt]+" * delta;");
217 		pass2.push("}else{");
218 		pass2.push("color.rg += n.rg * "+weights[cnt]+";");
219 		pass2.push("}");
220 	}
221 	pass2.push("color.a = (color.a+1.0)*n.b;");
222 	pass2.push("color.a = pow(color.a,cavitygamma);");
223 	if(this.quality<1){
224 		pass2.push("float dif =  length(color.rg-texture2D(GLGE_NORMAL, texCoord.xy+displace).rg);");
225 		pass2.push("samp =  texture2D(GLGE_NORMAL, texCoord.xy+displace+"+(1/this.gl.canvas.height)+").rgba;");
226 		pass2.push("if(abs(n.a-samp.a)<maxDist) dif =  max(length(color.rg-samp.rg),dif);");
227 		pass2.push("samp =  texture2D(GLGE_NORMAL, texCoord.xy+displace-"+(1/this.gl.canvas.height)+").rgba;");
228 		pass2.push("if(abs(n.a-samp.a)<maxDist) dif =  max(length(color.rg-samp.rg),dif);");
229 	}else{
230 		pass2.push("float dif =  length(color.rg-texture2D(GLGE_NORMAL, texCoord.xy).rg);");
231 	}
232 	
233 	pass2.push("float result = 1.0-((dif*(color.a-0.5)*2.0)+1.0)*0.5;");
234 	pass2.push("result = pow(min(result*whiteMul,1.0),aogamma);");
235 	pass2.push("gl_FragColor = vec4(vec3(result),1.0);");
236 	
237 
238 	if(this.useRender) pass2.push("gl_FragColor = vec4(texture2D(GLGE_RENDER, texCoord.xy).rgb*gl_FragColor.r,1.0);");
239 	pass2.push("}");
240 	
241 	var pass3=[]
242 		pass3.push("precision highp float;");
243 		pass3.push("uniform sampler2D GLGE_PASS1;");
244 		pass3.push("varying vec2 texCoord;");
245 		pass3.push("vec2 inverse_buffer_size=vec2(1.0/"+this.gl.canvas.width.toFixed(1)+",1.0/"+this.gl.canvas.height.toFixed(1)+");");
246 		pass3.push("#define FXAA_REDUCE_MIN   (1.0/128.0)");
247 		pass3.push("#define FXAA_REDUCE_MUL   (1.0/16.0)");
248 		pass3.push("#define FXAA_SPAN_MAX     8.0");
249 		pass3.push("void  main(){");
250 		pass3.push("	vec3 rgbNW = texture2D(GLGE_PASS1,  (gl_FragCoord.xy + vec2(-1.0,-1.0)) * inverse_buffer_size).xyz;");
251 		pass3.push("	vec3 rgbNE = texture2D(GLGE_PASS1,  (gl_FragCoord.xy + vec2(1.0,-1.0)) * inverse_buffer_size).xyz;");
252 		pass3.push("	vec3 rgbSW = texture2D(GLGE_PASS1,  (gl_FragCoord.xy + vec2(-1.0,1.0)) * inverse_buffer_size).xyz;");
253 		pass3.push("	vec3 rgbSE = texture2D(GLGE_PASS1,  (gl_FragCoord.xy + vec2(1.0,1.0)) * inverse_buffer_size).xyz;");
254 		pass3.push("	vec3 rgbM  = texture2D(GLGE_PASS1,  gl_FragCoord.xy  * inverse_buffer_size).xyz;");
255 		pass3.push("	vec3 luma = vec3(0.299, 0.587, 0.114);");
256 		pass3.push("	float lumaNW = dot(rgbNW, luma);");
257 		pass3.push("	float lumaNE = dot(rgbNE, luma);");
258 		pass3.push("	float lumaSW = dot(rgbSW, luma);");
259 		pass3.push("	float lumaSE = dot(rgbSE, luma);");
260 		pass3.push("	float lumaM  = dot(rgbM,  luma);");
261 		pass3.push("	float lumaMin = min(lumaM, min(min(lumaNW, lumaNE), min(lumaSW, lumaSE)));");
262 		pass3.push("	float lumaMax = max(lumaM, max(max(lumaNW, lumaNE), max(lumaSW, lumaSE)));");
263 			
264 		pass3.push("	vec2 dir;");
265 		pass3.push("	dir.x = -((lumaNW + lumaNE) - (lumaSW + lumaSE));");
266 		pass3.push("	dir.y =  ((lumaNW + lumaSW) - (lumaNE + lumaSE));");
267 			
268 		pass3.push("	float dirReduce = max(");
269 		pass3.push("	(lumaNW + lumaNE + lumaSW + lumaSE) * (0.25 * FXAA_REDUCE_MUL),");
270 		pass3.push("	FXAA_REDUCE_MIN);");
271 			
272 		pass3.push("	float rcpDirMin = 1.0/(min(abs(dir.x), abs(dir.y)) + dirReduce);");
273 		pass3.push("	dir = min(vec2( FXAA_SPAN_MAX,  FXAA_SPAN_MAX),");
274 		pass3.push("	max(vec2(-FXAA_SPAN_MAX, -FXAA_SPAN_MAX),");
275 		pass3.push("	dir * rcpDirMin)) * inverse_buffer_size;");
276 			  
277 		pass3.push("	vec3 rgbA = 0.5 * (");
278 		pass3.push("	texture2D(GLGE_PASS1,   gl_FragCoord.xy  * inverse_buffer_size + dir * (1.0/3.0 - 0.5)).xyz +");
279 		pass3.push("	texture2D(GLGE_PASS1,   gl_FragCoord.xy  * inverse_buffer_size + dir * (2.0/3.0 - 0.5)).xyz);");
280 			
281 		pass3.push("	vec3 rgbB = rgbA * 0.5 + 0.25 * (");
282 		pass3.push("	texture2D(GLGE_PASS1,  gl_FragCoord.xy  * inverse_buffer_size + dir *  - 0.5).xyz +");
283 		pass3.push("	texture2D(GLGE_PASS1,  gl_FragCoord.xy  * inverse_buffer_size + dir * 0.5).xyz);");
284 		pass3.push("	float lumaB = dot(rgbB, luma);");
285 		pass3.push("	if((lumaB < lumaMin) || (lumaB > lumaMax)) gl_FragColor = vec4(rgbA,1.0);");
286 		pass3.push("	    else gl_FragColor = vec4(rgbB,1.0);");
287 		pass3.push("	if(length(rgbM)>10.0) gl_FragColor = vec4(rgbM,1.0);");
288 		pass3.push("}");
289 		
290 	this.passes=[];
291 	this.addPass(pass1.join(""),width,height);
292 	this.addPass(pass2.join(""));
293 	this.addPass(pass3.join("\n"));
294 }
295 
296 
297 
298 })(GLGE);
