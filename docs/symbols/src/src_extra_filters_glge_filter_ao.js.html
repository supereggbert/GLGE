 34 (function(GLGE){
 35  
 36 /**
 37 * @class Postprocessing Ambient Occlusion filter
 38 * @augments GLGE.Filter2d
 39 */
 40 GLGE.FilterAO=function(){
 41 	this.setUniform("1f","cavitygamma",1/3);
 42 	this.setUniform("1f","whiteMul",2);
 43 	this.setUniform("1f","aogamma",1/3);
 44 	this.setUniform("1f","maxDist",0.025);
 45 	this.passes=[];
 46 };
 47  48 GLGE.augment(GLGE.Filter2d,GLGE.FilterAO);
 49 GLGE.FilterAO.prototype.renderNormal=true;
 50 GLGE.FilterAO.prototype.quality=1;
 51 GLGE.FilterAO.prototype.range=80;
 52 GLGE.FilterAO.prototype.samples=16;
 53 GLGE.FilterAO.prototype.useRender=true;
 54 
 55 GLGE.FilterAO.prototype.getNormalBufferHeight=function(){
 56 	return (this.normalBufferHeight ? this.normalBufferHeight : (this.gl.canvas.height*this.quality|0));
 57 }
 58 
 59 GLGE.FilterAO.prototype.getNormalBufferWidth=function(){
 60 	return (this.normalBufferWidth ? this.normalBufferWidth : (this.gl.canvas.width*this.quality|0));
 61 }
 62 
 63 GLGE.FilterAO.prototype.setUseRender=function(value){
 64 	this.useRender=value;
 65 	this.normalBuffers=null;
 66 	this.passes=[];
 67 	return this;
 68 }
 69 
 70 GLGE.FilterAO.prototype.setSamples=function(value){
 71 	this.samples=value;
 72 	this.normalBuffers=null;
 73 	this.passes=[];
 74 	return this;
 75 }
 76 
 77 GLGE.FilterAO.prototype.setQuality=function(value){
 78 	this.quality=value;
 79 	this.normalBuffers=null;
 80 	this.passes=[];
 81 	return this;
 82 }
 83 
 84 GLGE.FilterAO.prototype.setRange=function(value){
 85 	this.range=value;
 86 	if(this.gl){
 87 		this.setUniform("1f","blurX",this.range/this.getNormalBufferWidth()*this.quality/this.samples);
 88 		this.setUniform("1f","blurY",this.range/this.getNormalBufferHeight()/this.samples);
 89 	}
 90 	return this;
 91 }
 92 
 93 GLGE.FilterAO.prototype.setCavityGamma=function(value){
 94 	this.setUniform("1f","cavitygamma",1/value);
 95 	return this;
 96 }
 97 GLGE.FilterAO.prototype.setAmbientMultiplier=function(value){
 98 	this.setUniform("1f","whiteMul",value);
 99 	return this;
100 }
101 GLGE.FilterAO.prototype.setAmbientGamma=function(value){
102 	this.setUniform("1f","aogamma",1/value);
103 	return this;
104 }
105 GLGE.FilterAO.prototype.setMaximumDistance=function(value){
106 	this.setUniform("1f","maxDist",value);
107 	return this;
108 }
109 
110 GLGE.FilterAO.prototype.GLRender=function(gl,buffer){
111 	this.gl=gl;
112 	if(this.passes.length==0){
113 		this.createPasses();
114 	}
115 	return GLGE.Filter2d.prototype.GLRender.call(this,gl,buffer)
116 }
117 
118 GLGE.FilterAO.prototype.createPasses=function(){
119 	if(!this.gl) return;
120 	
121 	
122 	var width=this.getNormalBufferWidth();
123 	var height=this.getNormalBufferHeight();
124 	
125 	
126 	var size=(this.samples/4)|0;
127 	var weights=[];
128 	for(var i=-size,cnt=0; i<=size;i++,cnt++){
129 		var n=size-Math.abs(i)+1;
130 		weights[cnt]=n/(size*size+size);
131 	}
132 	weights[size]=0;
133 	
134 	this.setUniform("1f","blurX",this.range/width*this.quality/this.samples);
135 	this.setUniform("1f","blurY",this.range/height/this.samples);
136 
137 
138 
139 	
140 	var pass1=[];
141 	pass1.push("precision highp float;");
142 	pass1.push("uniform sampler2D GLGE_NORMAL;");
143 	pass1.push("uniform float maxDist;");
144 	pass1.push("varying vec2 texCoord;");
145 	pass1.push("uniform float blurX;");
146 	pass1.push("float rand(vec2 co){");
147 	pass1.push("return (fract(sin(dot(co.xy ,vec2(12.9898,78.233))) * 43758.5453)-0.5)*2.0;");
148 	pass1.push("}");
149 	pass1.push("void main(void){");
150 	pass1.push("vec4 n=texture2D(GLGE_NORMAL,texCoord.xy).rgba;");
151 	pass1.push("vec4 color=vec4(0.0,0.0,0.0,n.a);");
152 	pass1.push("float blurSize=blurX/(n.a*n.a+1.0);");
153 	pass1.push("float offset=rand(texCoord.xy)*blurSize+texCoord.x;");
154 	pass1.push("vec3 samp;");
155 	pass1.push("float delta;");
156 	for(var i=-size,cnt=0;i<=size;i++,cnt++){
157 		if(i==0) continue;
158 		pass1.push("samp = texture2D(GLGE_NORMAL, vec2("+i+".0*blurSize+offset, texCoord.y)).rga;");
159 		pass1.push("delta=abs(n.a-samp.b);");
160 		pass1.push("if(delta<maxDist){");
161 		pass1.push("delta/=maxDist;");
162 		pass1.push("color.b -= (samp.r-0.5) * "+weights[cnt]+" * "+((2*i)/Math.abs(i) | 0)+".0;");
163 		pass1.push("color.rg += samp.rg * "+weights[cnt]+" * (1.0-delta);");
164 		pass1.push("color.rg += n.rg  * "+weights[cnt]+" * delta;");
165 		pass1.push("}else{");
166 		pass1.push("color.rg +=n.rg * "+weights[cnt]+";");
167 		pass1.push("}");
168 	}
169 	pass1.push("color.b = (color.b+1.0)*0.5;");
170 	pass1.push("gl_FragColor = color;");
171 	pass1.push("}");
172 	
173 	var pass2=[];
174 	pass2.push("precision highp float;");
175 	pass2.push("uniform sampler2D GLGE_PASS0;");
176 	pass2.push("uniform sampler2D GLGE_RENDER;");
177 	pass2.push("uniform sampler2D GLGE_NORMAL;");
178 	pass2.push("varying vec2 texCoord;");
179 	pass2.push("uniform float blurY;");
180 	
181 	
182 	pass2.push("uniform float cavitygamma;");
183 	pass2.push("uniform float whiteMul;");
184 	pass2.push("uniform float aogamma;");
185 	pass2.push("uniform float maxDist;");
186 	
187 	
188 	pass2.push("float rand(vec2 co){");
189 	pass2.push("return (fract(sin(dot(co.xy ,vec2(12.9898,78.233))) * 43758.5453)-0.5)*2.0;");
190 	pass2.push("}");
191 	pass2.push("void main(void){");
192 	pass2.push("vec4 color=vec4(0.0,0.0,0.0,1.0);");
193 	pass2.push("vec4 samp=vec4(0.0);");
194 	pass2.push("float random=rand(texCoord.xy);");
195 	if(this.quality<1){
196 		pass2.push("vec2 displace=vec2("+(0.5/width)+","+(0.5/height)+")*random;");
197 		pass2.push("vec4 n=texture2D(GLGE_PASS0, texCoord.xy+displace);");
198 	}else{
199 		pass2.push("vec4 n=texture2D(GLGE_PASS0, texCoord.xy);");
200 	}
201 	pass2.push("float delta;");
202 	pass2.push("float blurSize=blurY/(n.a*n.a+1.0);");
203 	pass2.push("float offset=random*blurSize+texCoord.y;");
204 	
205 	for(var i=-size,cnt=0;i<=size;i++,cnt++){
206 		if(i==0) continue;
207 		if(this.quality<1){
208 			pass2.push("samp = texture2D(GLGE_PASS0, vec2(texCoord.x, "+i+".0*blurSize + offset)+displace);");
209 		}else{
210 			pass2.push("samp = texture2D(GLGE_PASS0, vec2(texCoord.x, "+i+".0*blurSize + offset));");
211 		}
212 		pass2.push("delta=abs(n.a-samp.a);");
213 		pass2.push("if(delta<maxDist){");
214 		pass2.push("delta/=maxDist;");
215 		pass2.push("color.a -= (samp.g-0.5) * "+weights[cnt]+" * "+((i*2)/Math.abs(i) | 0)+".0;");
216 		pass2.push("color.rg += samp.rg  * "+weights[cnt]+" * (1.0-delta);");
217 		pass2.push("color.rg += n.rg * "+weights[cnt]+" * delta;");
218 		pass2.push("}else{");
219 		pass2.push("color.rg += n.rg * "+weights[cnt]+";");
220 		pass2.push("}");
221 	}
222 	pass2.push("color.a = (color.a+1.0)*n.b;");
223 	pass2.push("color.a = pow(color.a,cavitygamma);");
224 	if(this.quality<1){
225 		pass2.push("float dif =  length(color.rg-texture2D(GLGE_NORMAL, texCoord.xy+displace).rg);");
226 		pass2.push("samp =  texture2D(GLGE_NORMAL, texCoord.xy+displace+"+(1/this.gl.canvas.height)+").rgba;");
227 		pass2.push("if(abs(n.a-samp.a)<maxDist) dif =  max(length(color.rg-samp.rg),dif);");
228 		pass2.push("samp =  texture2D(GLGE_NORMAL, texCoord.xy+displace-"+(1/this.gl.canvas.height)+").rgba;");
229 		pass2.push("if(abs(n.a-samp.a)<maxDist) dif =  max(length(color.rg-samp.rg),dif);");
230 231 	}else{
232 		pass2.push("float dif =  length(color.rg-texture2D(GLGE_NORMAL, texCoord.xy).rg);");
233 	}
234 	
235 	pass2.push("float result = 1.0-((dif*(color.a-0.5)*2.0)+1.0)*0.5;");
236 	pass2.push("result = pow(min(result*whiteMul,1.0),aogamma);");
237 	pass2.push("gl_FragColor = vec4(vec3(result),1.0);");
238 	
239 
240 	if(this.useRender) pass2.push("gl_FragColor = vec4(texture2D(GLGE_RENDER, texCoord.xy).rgb*gl_FragColor.r,1.0);");
241 	pass2.push("}");
242 	
243 	var pass3=[]
244 		pass3.push("precision highp float;");
245 		pass3.push("uniform sampler2D GLGE_PASS1;");
246 		pass3.push("varying vec2 texCoord;");
247 		pass3.push("vec2 inverse_buffer_size=vec2(1.0/"+this.gl.canvas.width.toFixed(1)+",1.0/"+this.gl.canvas.height.toFixed(1)+");");
248 		pass3.push("#define FXAA_REDUCE_MIN   (1.0/128.0)");
249 250 		pass3.push("#define FXAA_REDUCE_MUL   (1.0/16.0)");
251 		pass3.push("#define FXAA_SPAN_MAX     8.0");
252 		pass3.push("void  main(){");
253 		pass3.push("	vec3 rgbNW = texture2D(GLGE_PASS1,  (gl_FragCoord.xy + vec2(-1.0,-1.0)) * inverse_buffer_size).xyz;");
254 		pass3.push("	vec3 rgbNE = texture2D(GLGE_PASS1,  (gl_FragCoord.xy + vec2(1.0,-1.0)) * inverse_buffer_size).xyz;");
255 		pass3.push("	vec3 rgbSW = texture2D(GLGE_PASS1,  (gl_FragCoord.xy + vec2(-1.0,1.0)) * inverse_buffer_size).xyz;");
256 		pass3.push("	vec3 rgbSE = texture2D(GLGE_PASS1,  (gl_FragCoord.xy + vec2(1.0,1.0)) * inverse_buffer_size).xyz;");
257 		pass3.push("	vec3 rgbM  = texture2D(GLGE_PASS1,  gl_FragCoord.xy  * inverse_buffer_size).xyz;");
258 		pass3.push("	vec3 luma = vec3(0.299, 0.587, 0.114);");
259 		pass3.push("	float lumaNW = dot(rgbNW, luma);");
260 		pass3.push("	float lumaNE = dot(rgbNE, luma);");
261 		pass3.push("	float lumaSW = dot(rgbSW, luma);");
262 		pass3.push("	float lumaSE = dot(rgbSE, luma);");
263 		pass3.push("	float lumaM  = dot(rgbM,  luma);");
264 		pass3.push("	float lumaMin = min(lumaM, min(min(lumaNW, lumaNE), min(lumaSW, lumaSE)));");
265 		pass3.push("	float lumaMax = max(lumaM, max(max(lumaNW, lumaNE), max(lumaSW, lumaSE)));");
266 			
267 		pass3.push("	vec2 dir;");
268 		pass3.push("	dir.x = -((lumaNW + lumaNE) - (lumaSW + lumaSE));");
269 		pass3.push("	dir.y =  ((lumaNW + lumaSW) - (lumaNE + lumaSE));");
270 			
271 		pass3.push("	float dirReduce = max(");
272 		pass3.push("	(lumaNW + lumaNE + lumaSW + lumaSE) * (0.25 * FXAA_REDUCE_MUL),");
273 		pass3.push("	FXAA_REDUCE_MIN);");
274 			
275 		pass3.push("	float rcpDirMin = 1.0/(min(abs(dir.x), abs(dir.y)) + dirReduce);");
276 		pass3.push("	dir = min(vec2( FXAA_SPAN_MAX,  FXAA_SPAN_MAX),");
277 		pass3.push("	max(vec2(-FXAA_SPAN_MAX, -FXAA_SPAN_MAX),");
278 		pass3.push("	dir * rcpDirMin)) * inverse_buffer_size;");
279 			  
280 		pass3.push("	vec3 rgbA = 0.5 * (");
281 		pass3.push("	texture2D(GLGE_PASS1,   gl_FragCoord.xy  * inverse_buffer_size + dir * (1.0/3.0 - 0.5)).xyz +");
282 		pass3.push("	texture2D(GLGE_PASS1,   gl_FragCoord.xy  * inverse_buffer_size + dir * (2.0/3.0 - 0.5)).xyz);");
283 			
284 		pass3.push("	vec3 rgbB = rgbA * 0.5 + 0.25 * (");
285 		pass3.push("	texture2D(GLGE_PASS1,  gl_FragCoord.xy  * inverse_buffer_size + dir *  - 0.5).xyz +");
286 		pass3.push("	texture2D(GLGE_PASS1,  gl_FragCoord.xy  * inverse_buffer_size + dir * 0.5).xyz);");
287 		pass3.push("	float lumaB = dot(rgbB, luma);");
288 		pass3.push("	if((lumaB < lumaMin) || (lumaB > lumaMax)) gl_FragColor = vec4(rgbA,1.0);");
289 		pass3.push("	    else gl_FragColor = vec4(rgbB,1.0);");
290 		pass3.push("	if(length(rgbM)>10.0) gl_FragColor = vec4(rgbM,1.0);");
291 		pass3.push("}");
292 		
293 	this.passes=[];
294 	this.addPass(pass1.join(""),width,height);
295 	this.addPass(pass2.join(""));
296 	this.addPass(pass3.join("\n"));
297 }

301 })(GLGE);
