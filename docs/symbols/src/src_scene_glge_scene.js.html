289 /**
290 * z sorts the objects
291 * @private
292 */
293 GLGE.Scene.prototype.zSort=function(gl,objects){
294 	var cameraMatrix=gl.scene.camera.getViewMatrix();
295 	var transMatrix;
296 	for(var i=0;i<objects.length;i++){
297 		if(objects[i].object.getBoundingVolume){
298 			var center=objects[i].object.getBoundingVolume().getCenter();
299 		}else{
300 			var matrix=objects[i].object.getModelMatrix();
301 			var center=[matrix[3],matrix[7],matrix[11]];
302 		}
303 		objects[i].zdepth=center[0]*cameraMatrix[8]+center[1]*cameraMatrix[9]+center[2]*cameraMatrix[10]+cameraMatrix[11];
304 		if(objects[i].object.zDepth) {objects[i].zdepth=objects[i].object.zDepth;}
305 	}
306 	objects.sort(GLGE.Scene.sortFunc);
307 	return objects;
308 }

348 /**
349 * culls objects from the scene
350 * @private
351 */
352 GLGE.Scene.prototype.objectsInViewFrustum=function(renderObjects,cvp){
353 	var obj;
354 	var returnObjects=[];
355 	var planes=GLGE.cameraViewProjectionToPlanes(cvp);
356 	for(var i=0;i<renderObjects.length;i++){
357 		obj=renderObjects[i];
358 		if(obj.getBoundingVolume && obj.cull){
359 			var boundingVolume=obj.getBoundingVolume();
360 			var center=boundingVolume.getCenter();
361 			var radius=boundingVolume.getSphereRadius();
362 			if(GLGE.sphereInFrustumPlanes([center[0],center[1],center[2],radius],planes)){
363 				var points=boundingVolume.getCornerPoints();
364 				if(GLGE.pointsInFrustumPlanes(points,planes)){
365 					returnObjects.push(obj);
366 					if(obj.culled) obj.fireEvent("willRender",{});
367 					obj.culled=false;
368 				}else{
369 					if(!obj.culled) obj.fireEvent("willCull",{});
370 					obj.culled=true;
371 				}
372 			}else{
373 				if(!obj.culled) obj.fireEvent("willCull",{});
374 				obj.culled=true;
375 			}
376 		}else{
377 			returnObjects.push(obj);
378 		}
379 	}
380 	return returnObjects;	
381 }
382 /**
383 * Extracts all of the scene elements that need rendering
384 * @private
385 */
386 GLGE.Scene.prototype.unfoldRenderObject=function(renderObjects){
387 	var returnObjects=[];
388 	for(var i=0;i<renderObjects.length;i++){
389 		var renderObject=renderObjects[i];
390 		if(renderObject.getMultiMaterials){
391 			var multiMaterials=renderObject.getMultiMaterials();
392 			for(var j=0;j<multiMaterials.length;j++){
393 				var mat=multiMaterials[j].getMaterial();
394 				var mesh=multiMaterials[j].getMesh();
395 				if(!mat.meshIdx) mat.matIdx=j;
396 				if(!mat.meshIdx) mat.meshIdx=j;
397 				returnObjects.push({object:renderObject, multiMaterial:j});
398 			}
399 		}else{
400 			returnObjects.push({object:renderObject, multiMaterial:0});
401 		}
402 	}
403 	return returnObjects;
404 }

406 /**
407 * State sorting function
408 * @private
409 */
410 GLGE.Scene.prototype.stateSort=function(a,b){
411 	if(!a.object.GLShaderProgram) return 1;
412 	if(!b.object.GLShaderProgram) return -1;
413 	var aidx=a.object.GLShaderProgram.progIdx;
414 	var bidx=b.object.GLShaderProgram.progIdx;
415 	if(aidx>bidx){
416 		return 1;
417 	}else if(aidx<bidx){
418 		return -1;
419 	}else{
420 		if(!a.object.multimaterials || !b.object.multimaterials) return -1;
421 		var aidx=a.object.multimaterials[a.multiMaterial].getMaterial().matIdx;
422 		var bidx=b.object.multimaterials[b.multiMaterial].getMaterial().matIdx;
423 		if(aidx>bidx){
424 			return 1;
425 		}else if(aidx<bidx){
426 			return -1;
427 		}else{
428 			var amesh=a.object.multimaterials[a.multiMaterial].getMesh();
429 			var bmesh=a.object.multimaterials[a.multiMaterial].getMesh();
430 			if(!amesh) return -1;
431 			if(!bmesh) return 1;
432 			var aidx=amesh.meshIdx;
433 			var bidx=bmesh.meshIdx;
434 			if(aidx>bidx){
435 				return 1;
436 			}else if(aidx<bidx){
437 				return -1;
438 			}else{
439 				return 0;
440 			}
441 		}
442 	}
443 }

456 /**
457 * renders the scene
458 * @private
459 */
460 GLGE.Scene.prototype.render=function(gl){
461 	this.animate();
462 	//if look at is set then look
463 	if(this.camera.lookAt) this.camera.Lookat(this.camera.lookAt);	
464 
465 	gl.lights=this.getLights();
466 	
467 	var lights=gl.lights;
468 	gl.scene=this;
469 	this.lastMaterial=null;
470 	
471 	gl.disable(gl.BLEND);
472 	
473 	this.framebuffer=this.getFrameBuffer(gl);
474 	
475 
476 	var renderObjects=this.getObjects();
477 	var cvp=this.camera.getViewProjection();
478 	
479 	if(this.culling){
480 		var cvp=this.camera.getViewProjection();
481 		renderObjects=this.objectsInViewFrustum(renderObjects,cvp);
482 	}
483 	renderObjects=this.unfoldRenderObject(renderObjects);
484 	renderObjects=renderObjects.sort(this.stateSort);
485 
486 	
487 	//shadow stuff
488 	for(var i=0; i<lights.length;i++){
489 		if(lights[i].castShadows){
490 			if(!lights[i].gl) lights[i].GLInit(gl);
491 			var cameraMatrix=this.camera.matrix;
492 			var cameraPMatrix=this.camera.getProjectionMatrix();
493 			var projectedDistance=0;
494 			if(lights[i].getType()==GLGE.L_DIR){
495 				var mat=lights[i].getModelMatrix();
496 				var cmat=GLGE.inverseMat4(cameraMatrix);
497 				mat[3]=(mat[2])*lights[i].distance/2+cmat[3];
498 				mat[7]=(mat[6])*lights[i].distance/2+cmat[7];
499 				mat[11]=(mat[10])*lights[i].distance/2+cmat[11];
500 				lights[i].matrix=mat;
501 				var tvec=GLGE.mulMat4Vec4(cameraPMatrix,[0,0,lights[i].distance,1]);
502 				projectedDistance=tvec[3]/tvec[2]; //this is wrong?
503 			}
504 			
505 				gl.bindFramebuffer(gl.FRAMEBUFFER, lights[i].frameBuffer);
506 
507 				if(!lights[i].s_cache) lights[i].s_cache={};
508 				lights[i].s_cache.imvmatrix=GLGE.inverseMat4(lights[i].getModelMatrix());
509 				lights[i].s_cache.mvmatrix=lights[i].getModelMatrix();
510 				lights[i].s_cache.pmatrix=lights[i].getPMatrix(cvp,lights[i].s_cache.imvmatrix,projectedDistance,this.camera.far/2, this.camera);
511 				lights[i].s_cache.smatrix=GLGE.mulMat4(lights[i].s_cache.pmatrix,lights[i].s_cache.imvmatrix);
512 				lights[i].shadowRendered=false;
513 				
514 				
515 				gl.viewport(0,0,parseFloat(lights[i].bufferWidth),parseFloat(lights[i].bufferHeight));
516 				gl.clearDepth(1.0);
517 				gl.clearColor(1, 1, 1, 1);
518 				gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
519 					
520 				var height=parseFloat(lights[i].bufferHeight);
521 				var width=parseFloat(lights[i].bufferWidth);
522 				
523 
524 
525 				gl.viewport(0,0,width,height);						
526 
527 				this.camera.setProjectionMatrix(lights[i].s_cache.pmatrix);
528 				this.camera.matrix=lights[i].s_cache.imvmatrix;
529 				//draw shadows
530 				for(var n=0; n<renderObjects.length;n++){
531 					if(renderObjects[n].object.getCastShadows && !renderObjects[n].object.getCastShadows()) continue;
532 					if(renderObjects[n].object.className=="ParticleSystem") {continue;}
533 					if(lights[i].getType()==GLGE.L_SPOT){
534 						renderObjects[n].object.GLRender(gl, GLGE.RENDER_SHADOW,n,renderObjects[n].multiMaterial,lights[i].distance);
535 					}else{
536 						renderObjects[n].object.GLRender(gl, GLGE.RENDER_DEPTH,n,renderObjects[n].multiMaterial,lights[i].distance);
537 					}
538 				}
539 
540 				
541 				lights[i].s_cache.smatrix=GLGE.mulMat4(lights[i].s_cache.pmatrix,lights[i].s_cache.imvmatrix);
542 			
543 				lights[i].GLRenderSoft(gl);
544 						
545 			this.camera.matrix=null;
546 			this.camera.setProjectionMatrix(cameraPMatrix);
547 		}
548 	}
549 	
550 	if(this.culling){
551 		var cvp=this.camera.getViewProjection();
552 		renderObjects=this.objectsInViewFrustum(renderObjects,cvp);
553 	}
554 	
555 	gl.bindFramebuffer(gl.FRAMEBUFFER, this.framebuffer);
556 
557 	
558 	if(this.camera.animation) this.camera.animate();
559 	
560 	//null render pass to findout what else needs rendering
561 	this.getPasses(gl,renderObjects);	
562 	
563 	//first off render the passes
564 	var cameraMatrix=this.camera.matrix;
565 	var cameraPMatrix=this.camera.getProjectionMatrix();
566 	this.allowPasses=false;
567 	while(this.passes.length>0){
568 		var pass=this.passes.pop();
569 		gl.bindFramebuffer(gl.FRAMEBUFFER, pass.frameBuffer);
570 		this.camera.matrix=pass.cameraMatrix;
571 		this.camera.setProjectionMatrix(pass.projectionMatrix);
572 		this.mirror=pass.mirror;
573 		this.renderPass(gl,renderObjects,0,0,pass.width,pass.height,GLGE.RENDER_DEFAULT,pass.self);
574 	}
575 	this.mirror=false;
576 	
577 	this.camera.matrix=cameraMatrix;
578 	this.camera.setProjectionMatrix(cameraPMatrix);
579 	
580 	gl.bindFramebuffer(gl.FRAMEBUFFER, this.filter ? this.framebuffer : this.transbuffer);
581 	this.renderPass(gl,renderObjects,this.renderer.getViewportOffsetX(),this.renderer.getViewportOffsetY(),this.renderer.getViewportWidth(),this.renderer.getViewportHeight());	
582 
583 	
584 	this.applyFilter(gl,renderObjects, this.transbuffer);
585 	
586 	this.allowPasses=true;
587 }
588 /**
589 * gets the passes needed to render this scene
590 * @private
591 */
592 GLGE.Scene.prototype.getPasses=function(gl,renderObjects){
593 	for(var i=0; i<renderObjects.length;i++){
594 		renderObjects[i].object.GLRender(gl,GLGE.RENDER_NULL,0,renderObjects[i].multiMaterial);
595 	}
596 }

598 /**
599 * renders the scene
600 * @private
601 */
602 GLGE.Scene.prototype.renderPass=function(gl,renderObjects,offsetx,offsety,width,height,type,self){
603 	gl.clearDepth(1.0);
604 	gl.depthFunc(gl.LEQUAL);
605 	gl.viewport(offsetx,offsety,width,height);
606 	
607 	gl.clearColor(this.backgroundColor.r, this.backgroundColor.g, this.backgroundColor.b, this.backgroundColor.a);
608 
609 	gl.clear(gl.DEPTH_BUFFER_BIT | gl.COLOR_BUFFER_BIT | gl.STENCIL_BUFFER_BIT);
610 
611 	if(!type) type=GLGE.RENDER_DEFAULT;
612 	
613 	if(this.skyfilter && type==GLGE.RENDER_DEFAULT){
614 		this.skyfilter.GLRender(gl);
615 		gl.clear(gl.DEPTH_BUFFER_BIT);
616 		if(this.skyfilter && this.fogType==GLGE.FOG_SKYQUADRATIC || this.fogType==GLGE.FOG_SKYLINEAR){
617 			if(!this.skyTexture) this.createSkyBuffer(gl);
618 			gl.bindTexture(gl.TEXTURE_2D, this.skyTexture);
619 			gl.copyTexImage2D(gl.TEXTURE_2D, 0, gl.RGB, 0, 0, width, height, 0);
620 		}
621 	}
622 	
623 	var transObjects=[];
624 	gl.disable(gl.BLEND);
625 	for(var i=0; i<renderObjects.length;i++){
626 		if((!renderObjects[i].object.zTrans ||  type!=GLGE.RENDER_DEFAULT) && renderObjects[i].object!=self) renderObjects[i].object.GLRender(gl,type,0,renderObjects[i].multiMaterial);
627 			else if(renderObjects[i].object!=self) transObjects.push(renderObjects[i]);
628 	}
629 
630 	gl.enable(gl.BLEND);
631 	transObjects=this.zSort(gl,transObjects);
632 	for(var i=0; i<transObjects.length;i++){
633 		if(transObjects[i].object.blending){
634 			if(transObjects[i].object.blending.length==4){
635 				gl.blendFuncSeparate(gl[transObjects[i].object.blending[0]],gl[transObjects[i].object.blending[1]],gl[transObjects[i].object.blending[2]],gl[transObjects[i].object.blending[3]]);
636 			}else{
637 				gl.blendFunc(gl[transObjects[i].object.blending[0]],gl[transObjects[i].object.blending[1]]);
638 			}
639 		}
640 		if(transObjects[i].object.depthTest===false){

802 GLGE.Scene.prototype.makeRay=function(x,y){
803 	if(!this.camera){
804 		GLGE.error("No camera set for picking");
805 		return null;
806 	}else if(this.camera.matrix && this.camera.pMatrix){
807 		//correct xy account for canvas scaling
808 		var canvas=this.renderer.canvas;
809 		x=x/canvas.offsetWidth*canvas.width;
810 		y=y/canvas.offsetHeight*canvas.height;
811 		
812 		var height=this.renderer.getViewportHeight();
813 		var width=this.renderer.getViewportWidth();
814 		var offsetx=this.renderer.getViewportOffsetX();
815 		var offsety=this.renderer.getViewportHeight()-this.renderer.canvas.height+this.renderer.getViewportOffsetY();
816 		var xcoord =  ((x-offsetx)/width-0.5)*2;
817 		var ycoord = -((y+offsety)/height-0.5)*2;
818 		var invViewProj=GLGE.mulMat4(GLGE.inverseMat4(this.camera.matrix),GLGE.inverseMat4(this.camera.pMatrix));
819 		var origin =GLGE.mulMat4Vec4(invViewProj,[xcoord,ycoord,-1,1]);
820 		origin=[origin[0]/origin[3],origin[1]/origin[3],origin[2]/origin[3]];
821 		var coord =GLGE.mulMat4Vec4(invViewProj,[xcoord,ycoord,1,1]);
822 		coord=[-(coord[0]/coord[3]-origin[0]),-(coord[1]/coord[3]-origin[1]),-(coord[2]/coord[3]-origin[2])];
823 		coord=GLGE.toUnitVec3(coord);
824 		return {origin: origin, coord: coord};
825 		
826 	}else{
827 		return null;
828 	}
