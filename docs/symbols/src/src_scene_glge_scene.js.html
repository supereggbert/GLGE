288 /**
289 * z sorts the objects
290 * @private
291 */
292 GLGE.Scene.prototype.zSort=function(gl,objects){
293 	var cameraMatrix=gl.scene.camera.getViewMatrix();
294 	var transMatrix;
295 	for(var i=0;i<objects.length;i++){
296 		if(objects[i].object.getBoundingVolume){
297 			var center=objects[i].object.getBoundingVolume().getCenter();
298 		}else{
299 			var matrix=objects[i].object.getModelMatrix();
300 			var center=[matrix[3],matrix[7],matrix[11]];
301 		}
302 		objects[i].zdepth=center[0]*cameraMatrix[8]+center[1]*cameraMatrix[9]+center[2]*cameraMatrix[10]+cameraMatrix[11];
303 		if(objects[i].object.zDepth) {objects[i].zdepth=objects[i].object.zDepth;}
304 	}
305 	objects.sort(GLGE.Scene.sortFunc);
306 	return objects;
307 }

347 /**
348 * culls objects from the scene
349 * @private
350 */
351 GLGE.Scene.prototype.objectsInViewFrustum=function(renderObjects,cvp){
352 	var obj;
353 	var returnObjects=[];
354 	var planes=GLGE.cameraViewProjectionToPlanes(cvp);
355 	for(var i=0;i<renderObjects.length;i++){
356 		obj=renderObjects[i];
357 		if(obj.getBoundingVolume && obj.cull){
358 			var boundingVolume=obj.getBoundingVolume();
359 			var center=boundingVolume.getCenter();
360 			var radius=boundingVolume.getSphereRadius();
361 			if(GLGE.sphereInFrustumPlanes([center[0],center[1],center[2],radius],planes)){
362 				var points=boundingVolume.getCornerPoints();
363 				if(GLGE.pointsInFrustumPlanes(points,planes)){
364 					returnObjects.push(obj);
365 					if(obj.culled) obj.fireEvent("willRender",{});
366 					obj.culled=false;
367 				}else{
368 					if(!obj.culled) obj.fireEvent("willCull",{});
369 					obj.culled=true;
370 				}
371 			}else{
372 				if(!obj.culled) obj.fireEvent("willCull",{});
373 				obj.culled=true;
374 			}
375 		}else{
376 			returnObjects.push(obj);
377 		}
378 	}
379 	return returnObjects;	
380 }
381 /**
382 * Extracts all of the scene elements that need rendering
383 * @private
384 */
385 GLGE.Scene.prototype.unfoldRenderObject=function(renderObjects){
386 	var returnObjects=[];
387 	for(var i=0;i<renderObjects.length;i++){
388 		var renderObject=renderObjects[i];
389 		if(renderObject.getMultiMaterials){
390 			var multiMaterials=renderObject.getMultiMaterials();
391 			for(var j=0;j<multiMaterials.length;j++){
392 				var mat=multiMaterials[j].getMaterial();
393 				var mesh=multiMaterials[j].getMesh();
394 				if(!mat.meshIdx) mat.matIdx=j;
395 				if(!mat.meshIdx) mat.meshIdx=j;
396 				returnObjects.push({object:renderObject, multiMaterial:j});
397 			}
398 		}else{
399 			returnObjects.push({object:renderObject, multiMaterial:0});
400 		}
401 	}
402 	return returnObjects;
403 }

405 /**
406 * State sorting function
407 * @private
408 */
409 GLGE.Scene.prototype.stateSort=function(a,b){
410 	if(!a.object.GLShaderProgram) return 1;
411 	if(!b.object.GLShaderProgram) return -1;
412 	var aidx=a.object.GLShaderProgram.progIdx;
413 	var bidx=b.object.GLShaderProgram.progIdx;
414 	if(aidx>bidx){
415 		return 1;
416 	}else if(aidx<bidx){
417 		return -1;
418 	}else{
419 		if(!a.object.multimaterials || !b.object.multimaterials) return -1;
420 		var aidx=a.object.multimaterials[a.multiMaterial].getMaterial().matIdx;
421 		var bidx=b.object.multimaterials[b.multiMaterial].getMaterial().matIdx;
422 		if(aidx>bidx){
423 			return 1;
424 		}else if(aidx<bidx){
425 			return -1;
426 		}else{
427 			var amesh=a.object.multimaterials[a.multiMaterial].getMesh();
428 			var bmesh=a.object.multimaterials[a.multiMaterial].getMesh();
429 			if(!amesh) return -1;
430 			if(!bmesh) return 1;
431 			var aidx=amesh.meshIdx;
432 			var bidx=bmesh.meshIdx;
433 			if(aidx>bidx){
434 				return 1;
435 			}else if(aidx<bidx){
436 				return -1;
437 			}else{
438 				return 0;
439 			}
440 		}
441 	}
442 }

455 /**
456 * renders the scene
457 * @private
458 */
459 GLGE.Scene.prototype.render=function(gl){
460 	this.animate();
461 	//if look at is set then look
462 	if(this.camera.lookAt) this.camera.Lookat(this.camera.lookAt);	
463 
464 	gl.lights=this.getLights();
465 	
466 	var lights=gl.lights;
467 	gl.scene=this;
468 	this.lastMaterial=null;
469 	
470 	gl.disable(gl.BLEND);
471 	
472 	this.framebuffer=this.getFrameBuffer(gl);
473 	
474 
475 	var renderObjects=this.getObjects();
476 	var cvp=this.camera.getViewProjection();
477 	
478 	if(this.culling){
479 		var cvp=this.camera.getViewProjection();
480 		renderObjects=this.objectsInViewFrustum(renderObjects,cvp);
481 	}
482 	renderObjects=this.unfoldRenderObject(renderObjects);
483 	renderObjects=renderObjects.sort(this.stateSort);
484 
485 	
486 	//shadow stuff
487 	for(var i=0; i<lights.length;i++){
488 		if(lights[i].castShadows){
489 			if(!lights[i].gl) lights[i].GLInit(gl);
490 			var cameraMatrix=this.camera.matrix;
491 			var cameraPMatrix=this.camera.getProjectionMatrix();
492 			var projectedDistance=0;
493 			if(lights[i].getType()==GLGE.L_DIR){
494 				var mat=lights[i].getModelMatrix();
495 				var cmat=GLGE.inverseMat4(cameraMatrix);
496 				mat[3]=(mat[2])*lights[i].distance/2+cmat[3];
497 				mat[7]=(mat[6])*lights[i].distance/2+cmat[7];
498 				mat[11]=(mat[10])*lights[i].distance/2+cmat[11];
499 				lights[i].matrix=mat;
500 				var tvec=GLGE.mulMat4Vec4(cameraPMatrix,[0,0,lights[i].distance,1]);
501 				projectedDistance=tvec[3]/tvec[2]; //this is wrong?
502 			}
503 			
504 				gl.bindFramebuffer(gl.FRAMEBUFFER, lights[i].frameBuffer);
505 
506 				if(!lights[i].s_cache) lights[i].s_cache={};
507 				lights[i].s_cache.imvmatrix=GLGE.inverseMat4(lights[i].getModelMatrix());
508 				lights[i].s_cache.mvmatrix=lights[i].getModelMatrix();
509 				lights[i].s_cache.pmatrix=lights[i].getPMatrix(cvp,lights[i].s_cache.imvmatrix,projectedDistance,this.camera.far/2, this.camera);
510 				lights[i].s_cache.smatrix=GLGE.mulMat4(lights[i].s_cache.pmatrix,lights[i].s_cache.imvmatrix);
511 				lights[i].shadowRendered=false;
512 				
513 				
514 				gl.viewport(0,0,parseFloat(lights[i].bufferWidth),parseFloat(lights[i].bufferHeight));
515 				gl.clearDepth(1.0);
516 				gl.clearColor(1, 1, 1, 1);
517 				gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
518 					
519 				var height=parseFloat(lights[i].bufferHeight);
520 				var width=parseFloat(lights[i].bufferWidth);
521 				
522 
523 
524 				gl.viewport(0,0,width,height);						
525 
526 				this.camera.setProjectionMatrix(lights[i].s_cache.pmatrix);
527 				this.camera.matrix=lights[i].s_cache.imvmatrix;
528 				//draw shadows
529 				for(var n=0; n<renderObjects.length;n++){
530 					if(renderObjects[n].object.getCastShadows && !renderObjects[n].object.getCastShadows()) continue;
531 					if(renderObjects[n].object.className=="ParticleSystem") {continue;}
532 					if(lights[i].getType()==GLGE.L_SPOT){
533 						renderObjects[n].object.GLRender(gl, GLGE.RENDER_SHADOW,n,renderObjects[n].multiMaterial,lights[i].distance);
534 					}else{
535 						renderObjects[n].object.GLRender(gl, GLGE.RENDER_DEPTH,n,renderObjects[n].multiMaterial,lights[i].distance);
536 					}
537 				}
538 
539 				
540 				lights[i].s_cache.smatrix=GLGE.mulMat4(lights[i].s_cache.pmatrix,lights[i].s_cache.imvmatrix);
541 			
542 				lights[i].GLRenderSoft(gl);
543 						
544 			this.camera.matrix=null;
545 			this.camera.setProjectionMatrix(cameraPMatrix);
546 		}
547 	}
548 	
549 	if(this.culling){
550 		var cvp=this.camera.getViewProjection();
551 		renderObjects=this.objectsInViewFrustum(renderObjects,cvp);
552 	}
553 	
554 	gl.bindFramebuffer(gl.FRAMEBUFFER, this.framebuffer);
555 
556 	
557 	if(this.camera.animation) this.camera.animate();
558 	
559 	//null render pass to findout what else needs rendering
560 	this.getPasses(gl,renderObjects);	
561 	
562 	//first off render the passes
563 	var cameraMatrix=this.camera.matrix;
564 	var cameraPMatrix=this.camera.getProjectionMatrix();
565 	this.allowPasses=false;
566 	while(this.passes.length>0){
567 		var pass=this.passes.pop();
568 		gl.bindFramebuffer(gl.FRAMEBUFFER, pass.frameBuffer);
569 		this.camera.matrix=pass.cameraMatrix;
570 		this.camera.setProjectionMatrix(pass.projectionMatrix);
571 		this.mirror=pass.mirror;
572 		this.renderPass(gl,renderObjects,0,0,pass.width,pass.height,GLGE.RENDER_DEFAULT,pass.self);
573 	}
574 	this.mirror=false;
575 	
576 	this.camera.matrix=cameraMatrix;
577 	this.camera.setProjectionMatrix(cameraPMatrix);
578 	
579 	gl.bindFramebuffer(gl.FRAMEBUFFER, this.filter ? this.framebuffer : this.transbuffer);
580 	this.renderPass(gl,renderObjects,this.renderer.getViewportOffsetX(),this.renderer.getViewportOffsetY(),this.renderer.getViewportWidth(),this.renderer.getViewportHeight());	
581 
582 	
583 	this.applyFilter(gl,renderObjects, this.transbuffer);
584 	
585 	this.allowPasses=true;
586 }
587 /**
588 * gets the passes needed to render this scene
589 * @private

591 GLGE.Scene.prototype.getPasses=function(gl,renderObjects){
592 	for(var i=0; i<renderObjects.length;i++){
593 		renderObjects[i].object.GLRender(gl,GLGE.RENDER_NULL,0,renderObjects[i].multiMaterial);
594 	}
595 }

597 /**
598 * renders the scene
599 * @private
600 */
601 GLGE.Scene.prototype.renderPass=function(gl,renderObjects,offsetx,offsety,width,height,type,self){
602 	gl.clearDepth(1.0);
603 	gl.depthFunc(gl.LEQUAL);
604 	gl.viewport(offsetx,offsety,width,height);
605 	
606 	gl.clearColor(this.backgroundColor.r, this.backgroundColor.g, this.backgroundColor.b, this.backgroundColor.a);
607 	if(!type) {
608 		gl.scissor(offsetx,offsety,width,height);
609 		gl.enable(gl.SCISSOR_TEST);
610 		this.renderer.GLClear();
611 		gl.disable(gl.SCISSOR_TEST);
612 	}else{
613 		gl.clear(gl.DEPTH_BUFFER_BIT | gl.COLOR_BUFFER_BIT | gl.STENCIL_BUFFER_BIT);
614 	}
615 	if(!type) type=GLGE.RENDER_DEFAULT;
616 	
617 	if(this.skyfilter && type==GLGE.RENDER_DEFAULT){
618 		this.skyfilter.GLRender(gl);
619 		gl.clear(gl.DEPTH_BUFFER_BIT);
620 		if(this.skyfilter && this.fogType==GLGE.FOG_SKYQUADRATIC || this.fogType==GLGE.FOG_SKYLINEAR){
621 			if(!this.skyTexture) this.createSkyBuffer(gl);
622 			gl.bindTexture(gl.TEXTURE_2D, this.skyTexture);
623 			gl.copyTexImage2D(gl.TEXTURE_2D, 0, gl.RGB, 0, 0, width, height, 0);
624 		}
625 	}
626 	
627 	var transObjects=[];
628 	gl.disable(gl.BLEND);
629 	for(var i=0; i<renderObjects.length;i++){
630 		if((!renderObjects[i].object.zTrans ||  type!=GLGE.RENDER_DEFAULT) && renderObjects[i].object!=self) renderObjects[i].object.GLRender(gl,type,0,renderObjects[i].multiMaterial);
631 			else if(renderObjects[i].object!=self) transObjects.push(renderObjects[i]);
632 	}
633 
634 	gl.enable(gl.BLEND);
635 	transObjects=this.zSort(gl,transObjects);
636 	for(var i=0; i<transObjects.length;i++){
637 		if(transObjects[i].object.blending){
638 			if(transObjects[i].object.blending.length=4){
639 				gl.blendFuncSeparate(gl[transObjects[i].object.blending[0]],gl[transObjects[i].object.blending[1]],gl[transObjects[i].object.blending[2]],gl[transObjects[i].object.blending[3]]);
640 			}else{

808 		GLGE.error("No camera set for picking");

811 		//correct xy account for canvas scaling

814 		y=y/canvas.offsetHeight*canvas.height;
815 		
816 		var height=this.renderer.getViewportHeight();
817 		var width=this.renderer.getViewportWidth();
818 		var offsetx=this.renderer.getViewportOffsetX();
819 		var offsety=this.renderer.getViewportHeight()-this.renderer.canvas.height+this.renderer.getViewportOffsetY();
820 		var xcoord =  ((x-offsetx)/width-0.5)*2;
821 		var ycoord = -((y+offsety)/height-0.5)*2;
822 		var invViewProj=GLGE.mulMat4(GLGE.inverseMat4(this.camera.matrix),GLGE.inverseMat4(this.camera.pMatrix));
823 		var origin =GLGE.mulMat4Vec4(invViewProj,[xcoord,ycoord,-1,1]);

826 		coord=[-(coord[0]/coord[3]-origin[0]),-(coord[1]/coord[3]-origin[1]),-(coord[2]/coord[3]-origin[2])];
827 		coord=GLGE.toUnitVec3(coord);
